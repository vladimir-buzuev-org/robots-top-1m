    function validate(obj) {
        var errors = [];
        if (obj.contactForm_lastname == "") {
            errors.push({
                attr: "contactForm_lastname",
            })
        }
        if (obj.contactForm_firstname == "") {
            errors.push({
                attr: "contactForm_firstname",
            })
        }

        if (obj.contactForm_email != obj.contactForm_email2) {
            errors.push({
                attr: "contactForm_email2",
            })
        }

        if (obj.contactForm_email == "") {
            errors.push({
                attr: "contactForm_email",
            })
        }

        if( ! IsEmail(obj.contactForm_email)){
            errors.push({
                attr: "contactForm_email",
            })
        }

        if (obj.contactForm_msg == "") {
            errors.push({
                attr: "contactForm_msg",
            })
        }
