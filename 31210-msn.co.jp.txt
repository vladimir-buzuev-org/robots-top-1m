        var $cookieBanner;
        var locale;
        var cookieAcceptanceCookieName = "cac";
        var isCookieAcceptanceCookieBannerAccepted = false;

            function setCookie(cname, cvalue, exdays, domain, path)
            {
                var expiryDate;
                var builder = [cname, "=", cvalue];
                if (-1 == exdays) {
                    expiryDate = "Fri, 31 Dec 1999 23:59:59 GMT";
                }
                else if (exdays) {
                    var date = new Date();
                    date.setTime(date.getTime() + (exdays * 86400000));
                    expiryDate = date.toUTCString();
                }

                if (expiryDate) { builder.push(";expires=", expiryDate); }
                if (domain) { builder.push(";domain=", domain); }
                if (path) { builder.push(";path=", path); }

                document.cookie = builder.join("");
            }

            function getCookie(name)
            {
                var re = new RegExp('\\b' + name + '\\s*=\\s*([^;]*)', 'i');
                var match = re.exec(document.cookie);
                return (match && match.length > 1 ? match[1] : '');
            }

            function onAcceptClick(event)
            {
                if (isCookieAcceptanceCookieBannerAccepted)
                {
                    return;
                }

                var cookieDuration = 360;

                if (event.data && event.data.preventDefaultNavigation)
                {
                    event.preventDefault();
                }

                var cacCookieValue = createCacCookieValue();
                var existingCacCookieValue = getCookie(cookieAcceptanceCookieName);

                if (existingCacCookieValue)
                {
                    var updatedExistingCookieValue = false;

                    // update the existing value with new value e.g. nl-nl|v1 will be updated to nl-nl|v2
                    var localeDocumentVersionValues = existingCacCookieValue.split(",");
                    $.each(localeDocumentVersionValues, function (index, localeDocumentVersion) {
                        if (localeDocumentVersion.indexOf(locale + "|") !== -1) {
                            localeDocumentVersionValues[index] = cacCookieValue;
                            updatedExistingCookieValue = true;
                        }
                    });

                    // if cac value not set for current locale, add it. example existingCacCookieValue is cac=fr-fr|v1, current locale = nl-nl
                    // in this case we need to set cac=fr-fr|v1,nl-nl|v1
                    if (!updatedExistingCookieValue) {
                        localeDocumentVersionValues.push(cacCookieValue);
                    }

                    cacCookieValue = localeDocumentVersionValues.join();
                }

                // topDomain will be set to msn.com and empty-string in case domain is localhost and ipaddresses
                var topDomain = (window.location.hostname.match(/[^.]+\.[^.\d]+$/) || {})[0] || "";
                setCookie(cookieAcceptanceCookieName, cacCookieValue, cookieDuration, topDomain, "/");
                isCookieAcceptanceCookieBannerAccepted = true;
                $cookieBanner.hide();
            }

            function getSubDomains()
            {
                var subDomains = [];
                var hostname = location.hostname;
                subDomains.push(hostname);
                subDomains.push("." + hostname);
                var domainParts = hostname.split(".");
                if (domainParts.length > 2)
                {
                    for (var i = domainParts.length - 1; i > 1; i--)
                    {
                        subDomains.push(domainParts.slice(-1 * i).join("."));
                    }
                }

                return subDomains;
            }

            function removeUnknownCookies()
            {
                var cookies = $cookieBanner.data("unknownCookies");
                if (!cookies)
                {
                    return;
                }

                var subDomains = getSubDomains();
                var cookieList = cookies.split(",");

                $.each(cookieList, function(index, cname) {
                    setCookie(cname, "", -1);
                    $.each(subDomains, function (domainIndex, domain) {
                        setCookie(cname, "", -1, domain);
                    });
                });
            }

            function createCacCookieValue()
            {
                var latestCookieVersion = $cookieBanner.data("cacLatestVersion");
                var cacCookieValue = locale + "|" + latestCookieVersion;
                return cacCookieValue;
            }

            $(window).load(function () {
                $cookieBanner = $("#cookiebanner");
                locale = $cookieBanner.data("locale");
                var cacCookieValue = createCacCookieValue();
                var existingCacCookieValue = getCookie(cookieAcceptanceCookieName);

                // if valid cac cookie exist
                if (existingCacCookieValue && existingCacCookieValue.indexOf(cacCookieValue) != -1)
                {
                    // hide the banner
                    $cookieBanner.hide();
                }
                else
                {
                    $(".action a", $cookieBanner).click({ preventDefaultNavigation: true }, onAcceptClick);

                    // content clicks should accept the cookie banner
                    $("body").click(onAcceptClick);

                    $cookieBanner.click(function (event) { event.stopPropagation(); });
                }

                removeUnknownCookies();
            });
