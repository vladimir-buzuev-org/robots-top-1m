    {
        if(mt.matches)
        {
            if(clickmenu == id && doubleclick)
            {
                clickmenu = id;
                doubleclick = false;
            }
            else {
                e.preventDefault();
                $("#main-" + id).addClass("clicked");
                $("#sub-" + clickmenu).hide();
                $("#sub-" + id).show();
                clickmenu = id;
                doubleclick = true;
            }
        }
    }


    function mopen(id, e)
    {
        if(md.matches){
            mcancelclosetime();

            if(ddmenuitem)
            {
                dmenuitem.removeClass("active");
                ddmenuitem.hide();
            }

            dmenuitem = $("#main-" + id);
            ddmenuitem = $("#sub-" + id);
            dmenuitem.addClass("active");
            ddmenuitem.show();
        }

    }

    function mclose()
    {
        if(ddmenuitem)
        {
            dmenuitem.removeClass("active");
            ddmenuitem.hide();
        }
    }

    function mclosetime()
    {
        closetimer = window.setTimeout(mclose, timeout);
    }

    function mcancelclosetime()
    {
        if(closetimer)
        {
            window.clearTimeout(closetimer);
            closetimer = null;
        }
    }
