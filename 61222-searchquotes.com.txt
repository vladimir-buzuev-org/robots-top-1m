			var newNotifications = new Array();
			function checkForNewMessages() {
				$.ajax({
					url: 'http://www.searchquotes.com/sof/notifications/getnewnotifications.php?shown=0',

								alertsAr[alertsAr.length] = value;
								newNotifications[newNotifications.length] = value.notificationId;
							});
							startFloatingAlerts(alertsAr);
							updateNewMessagesCount();
						}
					},
					complete: function(a, b) {
						setTimeout(function() {
						      checkForNewMessages();
						}, 20000);	
					}

					url: 'http://www.searchquotes.com/sof/notifications/getnewnotifications.php?shown=1',
					cache: false,
					success: function(data) {
						var returnObj = JSON.parse(data);
						if (returnObj.count > 0) {
							$.each(returnObj.results, function(key, value) {
								newNotifications[newNotifications.length] = value.notificationId;
							});
							updateNewMessagesCount();
						}
					}
				});				
			}

			function startFloatingAlerts(alerts) {
				var i = 0;

					$(document).scroll();
					$('#floatingAlert'+value.notificationId).position({
						my: "right bottom",
						  at: "right bottom",
						  of: $(window),
						  offset: "-20 "+ vOffset
					});
					$(document).scroll(function() {
						$('#floatingAlert'+value.notificationId).position({
							my: "right bottom",
							  at: "right bottom",
							  of: $(window),
							  offset: "-20 "+ vOffset
						});
					});
					$('#floatingAlert'+value.notificationId).animate({

						setTimeout(function() {
							$('#floatingAlert'+value.notificationId).animate({
								opacity: 0
								}, 2000, function() {
							$('#floatingAlert'+value.notificationId).remove();
							});

					$('#newMessagesCountText').text(n);
					$('#newMessagesCount').css('background-image', "url('')"); //removed ./message_on.png
				}

			}

					success: function(data) {
						window.location="/inbox/";
					}
				});
