<script type='text/javascript'>
	$(function (){
		$('.orientacoes_submit').click(function (){

			var errors = 0;
			$('.orientacoes').each(function (){
				if ( $(this).val().length == 0 ) {
					$(this).css('border-color', '#ed3237');

					errors++
				}
				else
				{
					$(this).css('border-color', '#eee');
				}
			});

			if($("#g-recaptcha-response").val() == '') {
				errors++
			}

			if ( errors == 0 ) {
				$.ajax({
					url: 'enviar-orientacoes.php',
					data: {
						'nome': $('input[name=orientacoes_nome]').val(),
						'email': $('input[name=orientacoes_email]').val(),
						'cidade': $('input[name=orientacoes_cidade]').val(),
						'ddd': $('input[name=orientacoes_ddd]').val(),
						'telefone': $('input[name=orientacoes_telefone]').val(),
					},
					context: $(this)
				}).done(function (data){
					$('.orientacoes_mensagem').html(data);
					$('.orientacoes_mensagem').fadeIn(100);

					$('.orientacoes').each(function (){
						$(this).val(null);
					});
				})
			}
		});
	});
