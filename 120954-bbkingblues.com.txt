// When a form is submitted, post the data to the program to be processed

function doSubscribe() { 
   var e_status = document.getElementById('subscribeStatus');
   e_status.style.display='none';
   var status = '';
   var replyto = document.getElementById('replyto').value;
   if (!replyto) { 
      status = 'Please enter your e-mail address.';
   }
   else if ( replyto.search(/^\w+((-\w+)|(\.\w+))*\@[A-Za-z0-9]+((\.|-)[A-Za-z0-9]+)*\.[A-Za-z0-9]+$/) ) { 
      status = 'Please enter a valid e-mail address.';
   } 
   if (status) { 
      e_status.innerHTML = status;
      e_status.style.display='';
      return false;
   }
   // Post to program and get status back 
   $.post(
      "index.php",
      { "ajax": 1, "subscribe": 1, "replyto": replyto },
      function(data) { 
         status = data;
         e_status.innerHTML = data;
         e_status.style.display='';
      }
   );
   return false;
}
