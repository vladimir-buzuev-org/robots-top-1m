                $(this).parent().parent().addClass('dd');

                    return text.replace('[dd]', '').replace('[single]', '');
                })

                tarriffSwitch = true;
                $(this).parent().parent().addClass('single');
                $(this).text(function (index, text) {
                    return text.replace('[single]', '');
                })
            }
        });

            if (donatePage) {
                $('#monthly').change(function () { showDDorSingle2(); });
                showDDorSingle2();
            } else {
                $('.regularPayment').change(function () { showDDorSingle(); });
                showDDorSingle();
            }
        }
    });

function singleToDDSwitch() {
	var v = $(this).val();
	if ($(this).hasClass('active')) return;

	if (v == "Regular") {
		$('#direct-debit').click();
	}
	else {
		$('#direct-debit').click();
		if ($('#direct-debit').prop('checked') === true) $('#direct-debit').click();		
	}

	if (tarriffSwitch) showDDorSingle2();
	moveActive();
}

//Function to switch active
function moveActive() {
    if ($('#direct-debit').prop('checked') === true) {
        $('#Single').removeClass('active');
        $('#Regular').addClass('active');
    }
    else {
        $('#Single').addClass('active');
        $('#Regular').removeClass('active');
    }
}

        $('.donationAmount.single').hide();

        $('.donationAmount.single').show();

        defaultDonationPicked2($('.donationAmount.single')[0]);
    }
}

        defaultDonationPicked($('.donationAmount.single')[0]);
    }
}
