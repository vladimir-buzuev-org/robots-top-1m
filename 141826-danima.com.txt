        <script type="text/javascript">
            var fullHeight, winHeight, subnavHeight, doBounce, topBar, menuBar;
            $(document).ready(function(){
                var scroll = 0;
                fullHeight = $('#fullTop').height();
                menuBar = $('#menu').height();
                topBar = parseInt($('#topBar').height());
                winHeight = $(window).height();
                subnavHeight = $('#subnav').outerHeight();

                /** needed until you are done your paralax -- messes up the height and made it look bad on daves machine. **/
                $('#banner').height(winHeight-fullHeight);

                execBounce($('#banner .down'), 3, 10, 150);
                doBounce = setInterval(function() {execBounce($('#banner .down'), 3, 10, 150); }, 3000);

                $(window).scroll(function (e) {
                    scroll = $(window).scrollTop();
                    if(scroll > 50){
                        clearInterval(doBounce);
                        doBounce = null;
                    } else if(doBounce == null) {
                        //doBounce = setInterval(function(){ $('#banner .down').effect("bounce", { direction:'down', times:3 }, 1000); }, 3000);
                        doBounce = setInterval(function(){ execBounce($('#banner .down'), 3, 10, 150); }, 3000);
                    }

                    if(scroll > (winHeight - fullHeight)) {
                        $('#subnav').addClass('fixed').css({top: fullHeight+'px'});
                        $('body').css({paddingTop:subnavHeight + 'px'});
                    } else {
                        $('#subnav').removeClass('fixed').removeAttr('style');
                        $('body').removeAttr('style');
                    }
                });

                $('.down a').on('click', function(e){
                    e.preventDefault();
                    var offset = $($.attr(this, 'href')).offset();
                    //$('html,body').animate({scrollTop:(offset.top - (fullHeight - topBar))}, 550);
                    $('html,body').animate({scrollTop:(offset.top - (fullHeight))}, 550);
                });

                $('#subnav a').on('click', function(e){
                    if($(this).hasClass('noMove')) return true;
                    e.preventDefault();
                    var offset = $('#'+$(this).attr('class')).offset();
                    if(scroll > winHeight - (fullHeight*2)){
                        //$('html,body').animate({scrollTop: offset.top - (fullHeight - topBar + subnavHeight)}, 'slow');
                        $('html,body').animate({scrollTop: offset.top - (fullHeight + subnavHeight)}, 'slow');
                    } else {
                        //$('html,body').animate({scrollTop: (offset.top - subnavHeight)-(fullHeight - topBar + subnavHeight)}, 'slow');
                        $('html,body').animate({scrollTop: (offset.top - subnavHeight)-(fullHeight + subnavHeight)}, 'slow');
                    }
                });
            });

            function execBounce(element, times, distance, speed) {
                for(var i = 0; i < times; i++) {
                    element.animate({bottom: '-='+distance}, speed, 'swing')
                    .animate({bottom: '+='+distance}, speed,'swing');
                }
            }
