function do_contact_form(){
	if(!$('#contact_form').valid()) return false;
	$('#btn_send_contact').attr('disabled',true);
	
	$.ajax({
		url : './contact_form_form_handler.php',
		type : 'post',
		data : $('#contact_form').serialize(),
		success : function(res){
			$('#btn_send_contact').attr('disabled',false);
	
			res = jQuery.parseJSON(res);
			if(res.Status == 'OK'){
				alert(res.Msg);
				$('#ct_captcha').val('');
				$('#btn_reset').click();
			}else{
				alert(res.ErrMsg);
			}
		}
	});
	return false;
}
