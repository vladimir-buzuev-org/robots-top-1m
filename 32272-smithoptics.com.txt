function populateTrendingProducts(products, headerId, divId, totalItemsToShow, inStockOnly, showPrice, showDescription) {
    products = products != undefined ? products : new Array();
    totalItemsToShow = totalItemsToShow != undefined && !isNaN(totalItemsToShow) ? totalItemsToShow : 1;
    inStockOnly = inStockOnly != undefined ? inStockOnly : false;
    showPrice = showPrice != undefined ? showPrice : false;
    showDescription = showDescription != undefined ? showDescription : false;
    headerId = headerId != undefined && headerId.trim() != "" ? headerId : "trendingProductsHeader";
    divId = divId != undefined && divId.trim() != "" ? divId : "trendingProducts";
    counter = 0;
    var productGroups = groupTrendingProducts(products);
    if (totalItemsToShow > 0 && productGroups.length > 0) {
        for (i = 0; i < productGroups.length; i++) {
            if (counter >= totalItemsToShow) {
                break;
            }
            var productData = productGroups[i][0];
            if (!inStockOnly) {
                var reviewDivId = "trending_product_review_" + productData.skuGrouped.replace(" ", "");
                $("#" + divId).append(buildTrendingProductItem(productData, headerId, reviewDivId, showPrice, showDescription));
                if (productData.skuGrouped != "") {
                    getTrendingProductReviews(productData.skuGrouped, reviewDivId, 1, 150);
                }
                counter++;
            } else {
                checkInventoryThenPopulateTrendingProduct(productGroups[i], divId, headerId, showPrice, showDescription);
                counter++;
            }
        }
    }
}

function groupTrendingProducts(products) {
    var productGroups = new Array();
    var groupKeyArray = new Array();
    for (i = 0; i < products.length; i++) {
        var productData = products[i];
        var key = groupKeyArray.indexOf(productData.skuGrouped);
        if (key == -1) {
            groupKeyArray.push(productData.skuGrouped);
            productGroups.push(new Array());
            key = groupKeyArray.length - 1;
        }
        productGroups[key].push(productData);
    }
    return productGroups;
}

function checkInventoryThenPopulateTrendingProduct(products, divId, headerId, showPrice, showDescription) {
    if (products.length > 0) {
        var productData = products[0];
        //var stockUrl = rootPath + "p/" + productData.code + "/stock?akamai-feo=off";
        var stockUrl = rootPath + "p/" + productData.code + "/stock.json?akamai-feo=off";

        $.ajax({
            url: stockUrl,
            type: "GET",
            success: function(data) {
                //alert(data.inStock);
                var inStock = data.inStock != undefined ? data.inStock : false;
                //if (data.toLowerCase().trim() == "true") {
                //alert(inStock);
                if (inStock) {
                    reviewDivId = "trending_product_review_" + productData.skuGrouped.replace(" ", "");
                    $("#" + divId).append(buildTrendingProductItem(productData, headerId, reviewDivId, showPrice, showDescription));
                    if (productData.skuGrouped != "") {
                        getTrendingProductReviews(productData.skuGrouped, reviewDivId, 1, 150);
                    }
                } else {
                    products = products.slice(1);
                    checkInventoryThenPopulateTrendingProduct(products, divId, headerId, showPrice, showDescription);
                }
            },
            failure: function() {
                products = products.slice(1);
                checkInventoryThenPopulateTrendingProduct(products, divId, headerId, showPrice, showDescription);
            }
        });
    }
}

}

function trimText(text, length, wordBreak) {
    length = length != undefined ? length : 0;
    wordBreak = wordBreak != undefined ? wordBreak : false;
    if (text.length > length && length > 0) {
        while (text[length] != " " && text.length > length && wordBreak) {
            length++;
        }
        text = text.substr(0, length) + "...";
    }
    return text;
}

function drawStarRatings(rating, maxRating, showNumber, starImage, starWidth, starHeight) {
    rating = rating != undefined && !isNaN(rating) ? rating : 5;
    maxRating = maxRating != undefined && !isNaN(maxRating) ? maxRating : 5;
    showNumber = showNumber != undefined ? showNumber : false;
    starImage = starImage != undefined && starImage != "" ? starImage : "/_ui/desktop/smithoptics/images/REVIEW/review_star.png";
    starWidth = starWidth != undefined && !isNaN(starWidth) ? starWidth : 21.5;
    starHeight = starHeight != undefined && !isNaN(starHeight) ? starHeight : 15;
    if (rating > maxRating) {
        maxRating = Math.ceil(rating);
    }
    var ratingWidth = 100 / maxRating * rating;
    var s = "";
    s += '<div class="star_rating" style="width:' + (starWidth * maxRating) + "px; height:" + starHeight + 'px;">';
    s += '<div class="star_level" style="width:' + ratingWidth + '%;"></div>';
    s += '<div class="star_bar">';
    for (var i = 0; i < maxRating; i++) {
        s += '<img class="star-image" src="' + starImage + '" style="width:' + starWidth + "px;height:" + starHeight + 'px;" />';
    }
    s += "</div>";
    s += "</div>";
    if (showNumber) {
        s += '<div class="star_rating_number font_miller font_48">' + rating + "</div>";
    }
    return s;
}
