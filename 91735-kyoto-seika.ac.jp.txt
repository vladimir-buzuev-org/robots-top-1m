<script type="text/javascript">
$(function(){
    //setIdに#mainbanner

    var delayTime = 10000;

    ////currentCounts&Label
    var currentCounts = 1;
    var currentLabel = 'next';

    var randomDeleteList2 = ['3', '4'];
    var randomDeleteList3 = ['5', '6'];
    var randomDeleteList4 = ['7', '8'];
    var randomDeleteList5 = ['9', '10'];
    var randomDeleteList6 = ['11', '12'];

    // 削除
    var deleteId1 = randomDeleteList1[ Math.floor( Math.random() * randomDeleteList1.length ) ];
    var deleteId2 = randomDeleteList2[ Math.floor( Math.random() * randomDeleteList2.length ) ];
    var deleteId3 = randomDeleteList3[ Math.floor( Math.random() * randomDeleteList3.length ) ];

    $('#slide_list' + deleteId5).remove();
    $('#slide_list' + deleteId6).remove();
    $('#slide_navi' + deleteId1).remove();
    $('#slide_navi' + deleteId2).remove();
    $('#slide_navi' + deleteId3).remove();
    $('#slide_navi' + deleteId4).remove();
    $('#slide_navi' + deleteId5).remove();
    $('#slide_navi' + deleteId6).remove();

    $(setId +' #slides #slides_container .slides_control .slide ').css({left:'928px',display:'none'});

        $(setId +' #slides #slides_container .slides_control .slide:first ').css({zIndex:'5',display:'block'});
    });




    // 繰り返し実行設定する関数
    var setTimer;
    function timer() {
        if (setTimer) clearInterval(setTimer);//もしsetTimerに既に数値があったら、解除　

        //setTimerをセット
        setTimer = setInterval(function(){
            //全ての.activeに対して実行する
            $(setId + ' #slide_navi_container li.slide_navi.active').each(function(){
                var listLengh = $(setId + ' #slide_navi_container li.slide_navi').length;//li.slide_naviの数を取得
                var listIndex = $(setId + ' #slide_navi_container li.slide_navi').index(this);//現在のli.slide_naviの値を取得
                var listCount = listIndex+1;//listIndexに+1


                //currentCounts&Label
                currentCounts = currentCounts + 1;
                currentLabel = 'next';

                //もしcurrentCountsが7になったら
                if(currentCounts == 7){
                    currentCounts = 1;
                    currentLabel = 'next';
                } else {
                    currentCounts = currentCounts + 0;
                }

                //もしlistLenghがlistCountと同一の場合
                if(listLengh == listCount){
                    $(setId + ' #slide_navi_container li.slide_navi:first').click();//li.slide_naviの最初を表示？クリックする
                } else {
                    $(this).next('li').click();//それ以外の場合、次のlistを表示？クリックする
                }



                //alert(currentCounts);
            });
        }, delayTime);
    };

    $(setId + ' #slide_navi_container li.slide_navi').click(function(){

        var connectCont = $(setId + ' #slide_navi_container li.slide_navi').index(this);
        var showCont = connectCont+1;//+1にすることでshowContと一致


        if(showCont < currentCounts){
            currentLabel = 'prev';
            currentCounts = showCont;
        } else if(showCont > currentCounts) {
            currentLabel = 'next';
            currentCounts = showCont;
        } else {
            currentCounts = showCont;

        switch(currentLabel){

                $(setId + ' #slides #slides_container .slides_control #slide_list' + (showCont)).css({display:'block',left:'1856px'});
                $(setId + ' #slides #slides_container .slides_control #slide_list' + (showCont)).siblings().stop().animate({left:'0'},function(){$(this).css({zIndex:'0',display:'none'})});

                break;

        // クリックした要素に対する設定
        $(this).addClass('active');//.activeを追加
        $(this).siblings().removeClass('active');//それ以外の.activeを削除
        $(this).animate({opacity:'1'}, 200);//不透明度のアニメーション
        $(this).children('a').css('backgroundPosition','left bottom');//子要素の画像の位置を変更する

        // クリックされた要素以外の要素に対する設定
        $(setId + ' #slide_navi_container li.slide_navi:not(.active) a').css('backgroundPosition','left top');//子要素の画像の位置を変更する
        $(setId + ' #slide_navi_container li.slide_navi:not(.active)').animate({opacity:'0.5'},200);//不透明度のアニメーション

        // タイマーを実行
        timer();
    });

    // クリックされていない要素に対する設定
    $(setId + ' #slide_navi_container li.slide_navi:not(.active)').hover(//.activeが付いてないものにhoverした時
        function(){
            $(this).children('a').stop().css('backgroundPosition','left bottom');//子要素の画像の位置を変更する
            $(this).stop().animate({opacity:'1'}, 200);//不透明度のアニメーション
        },

            }
        }

    $(setId + ' #slide_navi_container li.slide_navi:first').addClass('active');//一番目に.active追加

    // 繰り返しタイマーの実行
    timer();

    // 最初のナビをクリック
    $(setId + ' #slide_navi_container li.slide_navi:first').click();



    // マウスオーバーでスライドショー動作（自動切替）を停止
    $(setId).mouseover(function(){
        clearInterval(setTimer);
    }).mousemove(function(){
            clearInterval(setTimer);
        }).mouseout(function(){
            timer();
        });

    // クリック範囲設定

        return false;

            var listLength  = $(setId + ' #slide_navi_container li.slide_navi').length;

            //currentCounts&Label
            currentCounts = currentCounts + 1;

            //もしcurrentCountsが7になったら
            if(currentCounts == 7){
                currentCounts = 1;
                currentLabel = 'next';
            } else {
                currentCounts = currentCounts + 0;
            }

            if (listLength == nextIndex) {
                $(setId + ' #slide_navi_container li.slide_navi:first').click();
            } else {
                $(this).next('li').click();
            }
        });

                $(this).prev('li').click();
            }
