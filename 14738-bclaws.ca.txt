	<script type="text/javascript">

function getFVList(){
	var xmlhttp = new XMLHttpRequest();
	var fvList, i, docs, docId, docTitle, url;
	xmlhttp.onreadystatechange=function(){
		if(xmlhttp.readyState==4 && xmlhttp.status==200){
			xmlDoc = xmlhttp.responseXML;
			fvList="";
			docs = xmlDoc.getElementsByTagName("doc");
			if(docs.length==0){
				fvList = "<p>No recently viewed.</p>";
			}
			else {
				for(i=0;i<docs.length;i++){
					docId = docs[i].getElementsByTagName("docid");
					docTitle = docs[i].getElementsByTagName("title");
					url = "href='/civix/document/id/complete/statreg/" + docId[0].firstChild.nodeValue;
					fvList = fvList + "<li><a " + url + "'>" + docTitle[0].firstChild.nodeValue + "</a></li>";
				}
			}
			document.getElementById("freq_visit").innerHTML= fvList;
		}
	}
	xmlhttp.open("GET","/civix/stats/complete/topDocs",true);
	xmlhttp.send();
}
