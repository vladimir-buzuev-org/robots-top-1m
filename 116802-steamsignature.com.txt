		function live(first) {

				dataType:'json',
				data: 'q=' + data,
				complete:function(e) { 
					ret=e.responseJSON;
					$(".gen").html(ret.count.toLocaleString() + ' Steam Signatures Generated');
					for(var i=0; i<ret.files.length; i++) {
						var found=false;
						
						for(var o=0; o < $('#last > div').length; o++) {
							if(typeof $('#last > div > a > img')[o] == 'undefined') $($('#last > div')[o]).remove(); else
							if('data:image/png;base64,' + ret.files[i].base64 == $('#last > div > a > img')[o].src) found=true; 
						}
						
						if(found === true) continue;

						var img=$('<img/>',{
							src: 'data:image/png;base64,' + ret.files[i].base64,
							class: 'img-responsive',
							alt: '',
						});
						
						var a=$('<a/>',{
							href: ret.files[i].id,
							target: '_blank'
						}).append(img);
						
						$('<div class="col col-xs-6"/>').append(a).prependTo($('#last'));
						
						if(!first)
						{
							$('#last').addClass('move');
							$('#last').toggleClass('move', false, 500, 'linear');
						}
						
						if($('#last > div').length > 4) $('#last > div').last().remove();
					}
					return setTimeout(live,1000);
				}
			});
