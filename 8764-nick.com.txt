  var TimeNow = (function(){
    var parseTime = function(t){
      //t format: 2014-04-02 13:00
      var d = new Date();
      try {
        d.setFullYear(parseInt(t.split("-")[0]));
        d.setMonth(parseInt(t.split("-")[1]) - 1);
        d.setDate(parseInt(t.split("-")[2]));
        d.setHours(parseInt(t.split(" ")[1].split(":")[0]));
        d.setMinutes(parseInt(t.split(" ")[1].split(":")[1]));
        d.setSeconds(0);
        d.setMilliseconds(0);
      } catch(error){
        console.log(error);
      }
      return d.getTime();
    };
    
    return {
      between: function(t1, t2){
        var now = new Date();
        var t1_stamp = parseTime(t1.trim());
        var t2_stamp = parseTime(t2.trim());
        return now > t1_stamp && now < t2_stamp;
      }
    }
  })();
</script>
