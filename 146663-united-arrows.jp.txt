    document.cookie = cookieName + "=" + encodeURIComponent(cookieValue) + "; expires=" + expire.toUTCString() + "; path=/";
  };

  /**
   * hashのkeyがparamsのkeyに存在していたらクッキーに保存
   * @param hash
   * @param params
   */
  var checkHash = function(hash, params) {
    var _keyValue; //[0]=key [1]=value
    for (var i=0;i<hash.length;i++){
      _keyValue = hash[i].split("=");
      if(params.hasOwnProperty(_keyValue[0])){
        SetCookie(params[_keyValue[0]], _keyValue[1], smcExpireDays);
      }
    }
  };

  checkHash(getQueryHash(document.location.search), smcParams);

}());
