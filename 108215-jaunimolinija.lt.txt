    function update_yt_players() {

        $('iframe[src*="youtube.com"]').each(
                function (index, item) {
                    var $embed = $(item);
                    console.log("$embed: ", $embed);
                    var ratio = $embed.attr('height') / $embed.attr('width');
                    $embed.height($embed.width() * ratio);
                    }
        );

        }

    function app_callback_handler(type) {
        if (type == 'resize') {
            update_yt_players();
            }
        }


    function allImagesLoaded() {

        $('.body_container img, .img_autoresize').not('.img_manual_size').each(function (index, el) {

            var $element = $(el);
            var target_width = el.naturalWidth || $element.css('width') || $element.width();

            if (target_width != '0px') {
                $element.css("maxWidth", target_width);
                $element.css({width: '100%', height: 'auto'});
                } else {
//                console.log( "IMG target_width: ", target_width, $element );
                }

            });
