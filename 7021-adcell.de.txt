<script>

// Auf Click des Login-Buttons erscheint das Login-Formular
$(".login-button").click(function(){
    $(".container").find(".customer-login").fadeToggle("fast", "linear");
});

// MOBILE - Auf Click des Customer-Buttons erscheinen die Kundendaten
$(".customer-btn").click(function(){
    $(".customer-info").fadeToggle("fast", "linear");
});

// Navigation Pfeil entfernen bzw. hinzufügen
$('#topmenu li').on('mouseenter', function(ev){
    var href = $(this).find('a').attr('href');
    $('#topmenu li').removeClass('navi-arrow');
    var numSubNaviElements = $('#submenu li[data-parent="' + href + '"]').length;
    if( numSubNaviElements > 0 ){
        $(this).addClass('navi-arrow');
    }

    var $activeListElement = $('#topmenu li.active');
    if($(this).find('a').text() != $activeListElement.find('a').text()){
        $activeListElement.addClass('no-arrow');
    }
});

$('.main-navi').on('mouseleave', function(ev){
    $('#topmenu li').removeClass('navi-arrow');
	var href = $('#topmenu li.active').find('a').attr('href');
    var numSubNaviElements = $('#submenu li[data-parent="' + href + '"]').length;
    if( numSubNaviElements > 0 ){
        $('#topmenu li.active').removeClass('no-arrow');
    }
    $('#submenu li').removeClass('menuvisible').addClass('menuhidden');
    $('#submenu li.menuhidden[data-parent="' + href + '"]').removeClass('menuhidden').addClass('menuvisible');
});

// Subnavi Box-shadow hinzufügen, wenn seite scrollt
$(window).scroll(function() {
    if ($(this).scrollTop() > 70){
        $('.subnavi').addClass("sub-shadow");
      }
      else{
        $('.subnavi').removeClass("sub-shadow");
      }
    });
</script>

//Formatierung von Zahlen
  function formatNumber(num) {
      return num.toString().replace(/(\d)(?=(\d{3})+(?!\d))/g, "$1.")
  }
//Keine Formatierung bei Zahlen
  function unformatNumber(num) {
      return num.replace(/\./g, "")
  }
//Counter werden hochgezählt
  $(function(){
      $('.customer-count').text("0");
      $('.customer-count').each( function(){
          var counter = $(this);
          var number = counter.attr("data-count");
          var userCountInterval = setInterval(function(){
              var bottom_of_object = counter.offset().top + counter.outerHeight();
              var bottom_of_window = $(window).scrollTop() + $(window).height();
              if( bottom_of_window > (bottom_of_object + 10) ){
                  var text = parseInt(unformatNumber(counter.text()));
                  if ( text < number){
                      var sum = text + (number / 15);
                      text = formatNumber((sum).toFixed(0));
                      counter.text(text);
                  }
                  else{
                      counter.text(formatNumber(number));
                      $( ".customer-register-btn" ).addClass( "call-action" );
                      clearInterval(userCountInterval);
                  };
              };
          }, 100);
      });
  });

var NewsLoader = {

    loadSuccess: function (data) {
        if (data.success == 1) {
            if (!jQuery.isEmptyObject(data.data)){
                $('#loadNews').attr('data-offset', parseInt($('#loadNews').attr('data-offset')) + 10);
                for (var i in data.data) {
                    row = data.data[i]

                    tr1 = $('<tr></tr>');
                    td = $('<td class="first-td"  colspan="2" style="border: none; font-size: 70%;">' + row.pubDate + ' - ' + row.removeDate + '</td>');
                    $(tr1).append(td);

                    tr2 = $('<tr></tr>');
                    th = $('<td style="width: 80px; vertical-align: middle;"></td>');
                    a = $('<a href="'+row.url+'" title="'+row.programName+'">');
                    td2 = $('<td style="vertical-align:middle;"></td>');
                    a2 = $('<a href="'+row.url+'" title="'+row.programName+'"></a>');
                    span = $('<span>'+row.programName+'</span><br/>');
                    span2 = $('<span>'+row.headline+'</span>');

                    $(a).append('<img src="//media.adcell.de/partner/' + row.programId + '.png" class="img-format" style="width:72px;height:auto;margin:5px 0px 0px;"/>');
                    $(a2).append(span, span2);
                    $(th).append(a);
                    $(tr2).append(th, td2);
                    $(td2).append(a2);

                    $('#newslistleft').append(tr1);
                    $('#newslistleft').append(tr2);
                };
            } else {
                $('#loadNews').remove();
            }
        }
        $('#loadNews').removeClass('processing');
    },

    loadError: function(data){
        console.log('ein fehler');
    },

    loadNews: function (el) {
        if(el.length != 0){
            if(!el.hasClass('processing')){
               el.addClass('processing');
                var scope = this;
                // URL anpassen
                var url = '/default/index/newsboxloader/newstype/news/offset/' + $(el).attr('data-offset');
                // Daten holen
                $.ajax({
                    type: "POST",
                    url: url,
                    success: scope.loadSuccess,
                    error: scope.loadError,
                    dataType: 'json'
                });
            }
        }
        return false;
    }

}

<script>
    var PartnerFocusBox = {
        init: function () {
            this.bindClick();
            this.resize();
        },

        bindClick: function () {
            var scope = this;
            $('#nextPartner').click($.proxy(scope.onClick, scope));
        },

        resize: function () {
            $('#partnerFocusBox').find('.ct').css('width', '295px');
            $('#partnerFocusBox').find('.partner-focus').css('width', '173px');

            //resize text
            $('#partnerFocusBox').find('.hellip').click(function(){
                $(this).hide();
                $('#partnerFocusBox').find('.hellip-after').show();
            });
        },

        onClick: function () {
            this.addLoading();
            var params = {
                'random' : 1
            };
            var scope = this;
            adcell.fn.send(
                '/default/index/partnerfocus',
                params,
                {
                    success: $.proxy(scope.changePartner, scope),
                    error: $.proxy(scope.loadCompleteError, scope)
                }
            );
            return false;
        },

        changePartner: function (data) {
            this.removeLoading();
            $('#partnerFocusBox').html(data);

            $('#partnerFocusBox').find('.btn-medium').button({
                icons: {
                    primary: 'ui-icon-arrowthick-1-e'
                }
            });

            this.resize();
        },

        loadCompleteError: function () {
            this.removeLoading();
        },

        addLoading: function () {
            this.removeLoading();

            var headline = $('#partnerFocusBox').find('h3').first();
            var loading = $('<img style="margin-left: 10px;" id="partnerFocusLoading" src="/themes/adcell-default/images/loading.gif" />');
            $(headline).append(loading);
        },

        removeLoading: function () {
            $('#partnerFocusLoading').remove();
        },
    }

    PartnerFocusBox.init();
</script>
