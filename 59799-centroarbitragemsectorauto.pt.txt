function addCommas(nStr){
    nStr += '';
    x = nStr.split('.');
    x1 = x[0];
    x2 = x.length > 1 ? ',' + x[1] : '';
    var rgx = /(\d+)(\d{3})/;
    while (rgx.test(x1)) {
        x1 = x1.replace(rgx, '$1' + '.' + '$2');
    }
    return x1 + x2;
}

function advanceAlert(){
	var element=$("div.alertBox ul li:visible").first();
	var next=element.next("li");
	if (next.size()==0){
		next=$("div.alertBox ul li").first();
	}
	element.fadeOut(300,function(){
		next.fadeIn(300);
	})
}

$(document).ready(function() {	
	$("#selectSubjectForm").submit(function(event){
		document.location=$("#selectSubject").val();
		return false;
	});

	
	$("div.alertBox").mouseenter(function(event){
		if (alertIntervalHandler!=null){
			clearInterval(alertIntervalHandler);
		}
	});
	$("div.alertBox").mouseleave(function(event){
		if ($("div.alertBox ul li").size()>1){
			alertIntervalHandler=setInterval("advanceAlert();",TTL_ALERTAS);		
		}		
	});
	if ($("div.alertBox ul li").size()>1){
		alertIntervalHandler=setInterval("advanceAlert();",TTL_ALERTAS);		
	}


	graphic();
});
