    var open_elements = Array.from(document.querySelectorAll('.sidemenuWrapper.categoryMenu ul.content li.menu-item.open'));

    function openParentElementWhenChildrenIsActive(){
        if(open_elements !== undefined && open_elements !== null && open_elements.length > 0){
            open_elements.forEach(function(element){
                var id = element.dataset.id;
                var child_level = parseInt(element.dataset.level) + 1;
                showChildElements(child_level,id);
            });

        }
    }

    function showChildElements(child_level, parent_id){
        var selector = '.sidemenuWrapper.categoryMenu ul.content li.menu-item.level-'+child_level+'[data-parentid="'+parent_id+'"]';
        var children = Array.from(document.querySelectorAll(selector));
        if(children.length){
            children.forEach(function(child){
                child.style.display = 'block';
            });
        }
    }

            this.removeClass('clicked');
            return hideAllChildElements();
        }else{
            hideAllChildElements();
            this.addClass('clicked');

<p><strong>LEGO</strong> er en privatejet virksomhed som blev grundlagt af Ole Kirk Kristiansen, i 1932. Sidenhen er virksomheden g&aring;et i arv og ejes nu af barnebarnet, Kjeld Kirk Kristiansen. <strong>LEGO</strong> har hovedkontor i Billund, hvor de ogs&aring; har deres kendte forlystelsespark, Legoland. Ideen bag <strong>LEGO</strong> er at udvikle b&oslash;rns kreativitet igennem leg, hvoraf navnet kom, da <strong>LEGO</strong> er en sammentr&aelig;kning af ordende &rdquo;Leg godt&rdquo;. <strong>LEGO</strong> er i dag et verdenskendt m&aelig;rke, som s&aelig;lger leget&oslash;j mm. til over 130 lande, og er dermed verdens tredjest&oslash;rste producent af legematerialer. Vision er at v&aelig;re banebrydende indenfor nye m&aring;der at t&aelig;nke leg p&aring;. Derfor udvikler og producere <strong>LEGO</strong> l&oslash;bende nye legematerialer, som svare til forbrugerenes konstant skiftende &oslash;nsker og behov. Den kendte legoklods er det absolut vigtigste produkt og <strong>LEGO</strong> er p&aring; baggrund af denne, blevet k&aring;ret til &Aring;rhundredets Leget&oslash;j, ikke mindre end to gange. Selvom produkterne konstant er under udvikling, er fundamentet for de mange nye tiltag, stadig den traditionelle legoklods. Disse legoklodser er helt unikke og giver uendelige muligheder, n&aring;r der skal bygges. Det er kun fantasien der s&aelig;tter gr&aelig;nserne, n&aring;r der skal leges.</p>
