        $(function() {
            $("#clube_empresarial").on({
                "change": function() {
                    if ($(this).val() == 1) {
                        $(".clubeEmpresarial").show().removeClass("hidden");
                        $(".pessoaFisica").hide().addClass("hidden");
                    } else {
                        $(".pessoaFisica").show().removeClass("hidden");
                        $(".clubeEmpresarial").hide().addClass("hidden");
                    }
                }
            });

            $("#responsavel").on({
                "click": function() {
                    if ($(this).is(":checked")) {
                        $("#nome_segurado").val($("#nome").val());
                        $("#cpf_segurado").val($("#CPF").val());
                        $("#sexo_segurado").val($("#sexo").val());
                    } else {
                        $("#nome_segurado").val("");
                        $("#cpf_segurado").val("");
                        $("#sexo_segurado").val("");
                    }
                }
            });

            $("#responsavel_sinistro").on({
                "click": function() {
                    if ($(this).is(":checked")) {
                        $("#sinistro_nome_segurado").val($("#nome").val());
                        $("#sinistro_cpf_segurado").val($("#CPF").val());
                        $("#sinistro_sexo_segurado").val($("#sexo").val());
                    } else {
                        $("#sinistro_nome_segurado").val("");
                        $("#sinistro_cpf_segurado").val("");
                        $("#sinistro_sexo_segurado").val("");
                    }
                }
            });

            $("[data-modal-reference=\"SeguroVida\"]").on({
                "click": function() {

                    $("[data-modal-id=\"SeguroVida\"]").modal({
                        backdrop: false
                    });
                }
            });
        });
