<script type="text/javascript">
function toggleGeo()
{	
	if ($('#filter_mylocation').prop('checked'))
	{
		navigator.geolocation.getCurrentPosition(displayPosition);			
	}
	else
	{
		$('#google_x').val('');
		$('#google_y').val('');
		$('#filterForm').submit();	   
	}	
	return false;
}

function displayPosition(position) {
	  	
	  $('#google_x').val(position.coords.latitude);
	  $('#google_y').val(position.coords.longitude);
	  $('#filterForm').submit();	   
	}

		  $("#searchResult").fadeIn('fast');
		  
		  return false;
	});

  
  jQuery(document).live("click", function(e) { 
      var $clicked = $(e.target);
      if (!$clicked.hasClass("search")){
      	jQuery("#searchResult").fadeOut('fast');      
      }
});
