        var answerId = $("input[name=answer]:checked").val();

        if (isNaN(answerId)) {
            $('#poll-question-block-message').addClass('text-danger');
            $('#poll-question-block-message').html("Please choose your opinion and vote.");
            return;
        }

        $.ajax({
            type: 'GET',
            url: Routing.generate('nagarik_poll_save_vote', {'answer_id': answerId}),
            success: function (data) {

                $('#poll-result-block').show();
                $('#poll-result-block-message').html(data.message);

                $('#poll-question-block-message').html('');

                if (data.status == 'success') {
                    $('#poll-result-block-message').addClass('text-success');
                } else {
                    $('#poll-result-block-message').addClass('text-danger');
                }

                $('#poll-question-block').removeClass('col-md-12');
                $('#poll-question-block').addClass('col-md-8');
            }
        });

    });
