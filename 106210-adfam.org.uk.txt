function checkoutRecalc() {
	var basketSubtotal=weightTotal=0;
	var large=false;
	//SPECIAL OFFER
	var specialOfferDiscount=0;
	$('#basket>div.body>div.row').each(function(){
		var thisQuantity=$(this).children('div.quantity').children('p').children('input').val();
		var thisPrice=parseFloat($(this).children('div.price').children('p').html().substr(1));
		var thisSubtotal=thisPrice*thisQuantity;
		basketSubtotal+=thisSubtotal;
		weightTotal+=thisQuantity*$(this).children('input.weight').val();
		if ($(this).children('input.large').val()=='1'&&thisQuantity!=0) {
			large=true;
		}
		//SPECIAL OFFER
		// if ($(this).children('div.item').children('p').html().match(/^Adfam Annual Conference/)&&thisQuantity>1) {
			// specialOfferDiscount=thisSubtotal*0.1;
		// }
		if ($(this).hasClass('publication')) {
			specialOfferDiscount+=thisSubtotal*0.33;
		}
	});
	//SPECIAL OFFER
	// var specialOfferDiscount=0;
	// if (basketSubtotal>=100) {
		// specialOfferDiscount=basketSubtotal*0.15;
	// } else if (basketSubtotal>=50) {
		// specialOfferDiscount=basketSubtotal*0.1;
	// }
	$('#basketSubtotal').html('Â£'+basketSubtotal.toFixed(2));
	$('#basketSpecialOfferDiscount').html('Â£'+specialOfferDiscount.toFixed(2))
	$('#basketSubtotalDiscount').html('Â£'+(basketSubtotal-specialOfferDiscount).toFixed(2));
	if (weightTotal>0) {
		$.ajax({
			url:'/assets/script/postageCalc.php',
			type:'POST',
			data:{
				weight:weightTotal,
				large:large
			},
			success:function(data){
				var basketPostage=parseFloat(data);
				$('#basketPostage').html('Â£'+basketPostage.toFixed(2));
				var basketTotal=(basketSubtotal-specialOfferDiscount+basketPostage).toFixed(2);

function submitPost(caller) {
	if (checkDefaults(caller)) {
		return false;
	}
	$(caller).off();
	var data={
		action:'post',
		title:$(caller).prev().prev().children('input').val(),
		post:$(caller).prev().children('textarea').val()
	};
	if ($(caller).parent().hasClass('forum')) {
		data.section='forum';
		data.forum_topic_id=$(caller).parent().parent().attr('id').match(/\d+$/)[0];
	} else {
		data.section='blog';
	}
	$.ajax({
		url:'/assets/script/forum_blog.php',
		type:'POST',
		data:data,
		success:function(data){
			if (data) {
				location.assign(data);
			} else {
				alert('Sorry, there was an error submitting your post, please try again.');
				$(caller).on('click',function(){
					submitPost(caller);
				});
			}
		}
	});
