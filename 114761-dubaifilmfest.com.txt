$(window).scroll(function() {
	close_header_nav();	
	var st = $(this).scrollTop();
	if (st > lastScrollTop){
		if (st > 70){
			// downscroll

			$('.toggle_nav').css('visibility', 'visible').fadeIn('slow');
		}

		// upscroll

		$('.toggle_nav').fadeOut('slow');
    }
	lastScrollTop = st;
});
