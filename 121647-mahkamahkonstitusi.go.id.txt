<script>
function openTab(evt, tabName) {
    // Declare all variables
    var i, tabcontent, tablinks;

    // Get all elements with class="tabcontent" and hide them
    tabcontent = document.getElementsByClassName("tabcontent");
    for (i = 0; i < tabcontent.length; i++) {
        tabcontent[i].style.display = "none";
    }

    // Get all elements with class="tablinks" and remove the class "active"
    tablinks = document.getElementsByClassName("tablinks");
    for (i = 0; i < tablinks.length; i++) {
        tablinks[i].className = tablinks[i].className.replace(" active", "");
    }

    // Show the current tab, and add an "active" class to the link that opened the tab
    document.getElementById(tabName).style.display = "block";
    evt.currentTarget.className += " active";
}
// Get the element with id="defaultOpen" and click on it
document.getElementById("defaultOpen").click();
</script>

<script>
function openTabbawah(evt, tabNamebawah) {
    // Declare all variables
    var i, tabcontentbawah, tablinksbawah;

    // Get all elements with class="tabcontent" and hide them
    tabcontentbawah = document.getElementsByClassName("tabcontentbawah");
    for (i = 0; i < tabcontentbawah.length; i++) {
        tabcontentbawah[i].style.display = "none";
    }

    // Get all elements with class="tablinks" and remove the class "active"
    tablinksbawah = document.getElementsByClassName("tablinksbawah");
    for (i = 0; i < tablinksbawah.length; i++) {
        tablinksbawah[i].className = tablinksbawah[i].className.replace(" active", "");
    }

    // Show the current tab, and add an "active" class to the link that opened the tab
    document.getElementById(tabNamebawah).style.display = "block";
    evt.currentTarget.className += " active";
}
// Get the element with id="defaultOpen" and click on it
document.getElementById("defaultOpenbawah").click();
