        function OpenModal(url) {
            $.ajax({
                type: "POST",

                data: "{Param1: '" + url + "'}",

                    RenderResult(data);
                }
            });
        }

        function RenderResult(data) {
            var str = data.d.split('^');
            document.getElementById('ucHeader1_spnModalTitle').innerHTML = str[0];
            document.getElementById('ucHeader1_spnModalText').innerHTML = str[1];
            $('#myModal').modal('show');
        }

        var substringMatcher = function () {
            return function findMatches(q, add) {
                var suggestions = [];
                var keyword = $("#txtKeyword").val().replace("'", "");
                //pass request to server
                var myVars = { 'Keyword': keyword };
                $.ajax({
                    url: 'https://your.googlemerchandisestore.com/Products/ProductListing.aspx/GetProductData',
                    data: "{'Keyword': '" + keyword + "' }",
                    dataType: "json",
                    type: "POST",
                    contentType: "application/json; charset=utf-8",
                    cache: false,
                    async: true
                }).done(function (data) {
                    //create array for response objects
                    if (data.d == "")
                    { add(suggestions); return false; }
                    var json = $.parseJSON(data.d);
                    //process response

                    $.each(json, function (i, val) {
                        suggestions.push(val.ProductCode + '-' + val.ProductName);
                    });
                    //pass array to callback
                    add(suggestions);
                });
            };
        };

                return url + "#" + uriEncodedQuery;

            },
            remote: {
                url: "https://your.googlemerchandisestore.com/Products/ProductListing.aspx?q=%QUERY",

            }
