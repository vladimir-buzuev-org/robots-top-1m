    <script>
    $(document).ready(function () {

        var xhr;
        var active = false;
        var lvl;
        var screenWidth = $(window).width();
        var startX, startY, tap;
        var linktype;

        function getCoord(e, c) {
            return /touch/.test(e.type) ? (e.originalEvent || e).changedTouches[0]['page' + c] : e['page' + c];
        }

        function setTap() {
            tap = true;
            setTimeout(function () {
                tap = false;
            }, 500);
        }

        var selectedMenu;

        function ajaxCallback(item) {

            linktype = $(item).attr("data-menu");


            if (screenWidth > 480 && linktype == "normal") {
                $('html, body').animate({ scrollTop: 0 }, 800);
            } else {
                $('html, body').animate({ scrollTop: 0 }, 800);
            }

            lvl = $(item).attr("data-level");

            if (active) { xhr.abort(); }
            active = true;

            var id = $(item).attr("data-nodeid");
            var header = '<li class=\"parent-page"\><a href=\"' + $(item).attr("href") + '\"><span class=\"link-name\">' + $(item).text() + ' Home</span></a></li>';
            var nxtlvl = (+lvl + +1);

            $(".mnu" + nxtlvl).html("<i class=\"fa fa-cog fa-spin\"></i> Loading Menu");

            xhr = $.ajax({
                url: "/services/menu.asmx/GetMenuItems",
                type: "post",
                dataType: "json",
                contentType: "application/json",
                data: '{ "nodeID": "' + id + '" }',
                success: function (data) {

                    var buildHTML = '';

                    if (data.d.length == 0) {
                        document.location = $(item).attr("href");
                    }

                    for (var i = 0; i < data.d.length; i++) {
                        buildHTML += '<li><a href=\"' + data.d[i].MenuUrl + '\" class=\"menu-link\" data-nodeid=\"' + data.d[i].MenuID + '\" data-level=\"' + nxtlvl + '\"><span class=\"link-name\">' + data.d[i].MenuTitle + '</span> <i class="fa fa-arrow-right"></i></a></li>';
                    }

                    if (screenWidth <= 480) {

                        var parentLink = '<li><a href=\"' + $(item).attr("href") + '\"><span class=\"link-name\">' + $(item).text() + ' Home</span></a></li>';

                        if (selectedMenu == $(item).attr("data-nodeid")) {
                            selectedMenu = 0;
                        } else {
                            selectedMenu = $(item).attr("data-nodeid");
                        }

                        if (selectedMenu == 0) {
                            $(".mnu2").empty();
                        } else {
                            $(".mobmnu-" + id).html('<ul class=\"submeganav\">' + parentLink + buildHTML + '</ul>');
                        }

                    } else {

                        $(".mnu" + nxtlvl).html('<ul>' + header + buildHTML + '</ul>');

                        $(".mnu" + nxtlvl + " .menu-link").on('click touchend', function () {

                            ajaxCallback($(this));

                            if (lvl <= 2) {
                                return false;
                            }

                        });
                    }

                },
                error: function (xhr) {
                }
            });
        }

        //var event = (screenWidth <= 480) ? "touchend" : "click";

        $(".mnu1 .menu-link").on('touchstart', function (ev) {
            startX = getCoord(ev, 'X');
            startY = getCoord(ev, 'Y');
        }).on('touchend', function (ev) {
            // If movement is less than 20px, execute the handler
            if (Math.abs(getCoord(ev, 'X') - startX) < 20 && Math.abs(getCoord(ev, 'Y') - startY) < 20) {

                if (screenWidth <= 480) {
                    $("[class^=mobmnu]").empty();
                } else {
                    $(".mnu2").empty();
                    $(".mnu3").empty();
                }

                ajaxCallback($(this));

                // Prevent emulated mouse events
                ev.preventDefault();
            }
            setTap();
        }).on('click', function (ev) {
            if (!tap) {
                // If handler was not called on touchend, call it on click;
                if (screenWidth <= 480) {
                    $("[class^=mobmnu]").empty();
                } else {
                    $(".mnu2").empty();
                    $(".mnu3").empty();
                }

                ajaxCallback($(this));
            }
            ev.preventDefault();
        });

        if (screenWidth > 480) {
            $("#collapseMenu").click(function (e) {
                if ($(e.target).parents("a").length < 1) {
                    $("#collapseMenu").collapse("hide");
                }
            });
        }

        if (screenWidth > 480) {
            $("#collapseSubMenu").click(function (e) {
                if ($(e.target).parents("a").length < 1) {
                    $("#collapseSubMenu").collapse("hide");
                }
            });
        }

    });

    </script>
