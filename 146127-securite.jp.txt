						$(selector).html(content);
						//選択した地域ではファンドがない場合、「…ありません」メッセージを出す
						if($($.parseHTML(resData)).find('#fund-count').text() === '0')
						{
							$('.tabContentsShow').html('<div id="fund-empty" class= "center">この地域では受付中ファンドがありません</div>');
						}
					}
				});

				//地域メニューの色が変わる
				$(this).addClass('blue');
				$(this).siblings('li').removeClass('blue');
			}
		});

		//もっと見る機能
		$(document).on('click','#load-more-button', function () {
			//ファンドリストDOM
			var prev_fund_list = $(this).parent().prev('.tabContentsShow');
			//地域名
			var area_key = $('#area-key').text();
			//jsonファイルのインデックス
			var json_file_index = $('#json-file-index').text();

			$.ajax({
				url: '/?area_key='+area_key+'&json_file_index='+json_file_index,
				type: 'GET',
				success: function (resData) 
				{
					var ajax_DOM = $($.parseHTML(resData));
					//新なファンドリストを追加する
					prev_fund_list.append(ajax_DOM.find('.tabContentsShow').html());
					//パラメータを更新する
					$('.load-more').html(ajax_DOM.find('.load-more').html());
					$('#params').html(ajax_DOM.find('#params').html());
				}
			});
		});
	});
</script>
