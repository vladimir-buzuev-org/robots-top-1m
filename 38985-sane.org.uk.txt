<script type="text/javascript">
    $(document).ready(function() {
         var maxxfix_login = function(username, password, divName) {
            var success = false;
            
            $.ajax({
                url: '/support_forum/custom_login.php?mode=login',
                type: 'POST',
                async:   false,
                data: {
                    'username': username, 
                    'password': password,  
                    'autologin':true,
                    'login' : 'Login',
                    'redirect' : 'index.php',
                    'sid' : ''
                },
                success: function(xml){
					if( xml != '' ){
						if( divName == 'login' ){ 
							$('#loginform_error').html( xml ); 
							$('#loginform_error').addClass("error_message");
						}else{ 
							$('#forum_error').html( xml ); 
							$('#forum_error').addClass("error_message");
						}
						success = false;
					}else{
						success = true;
					}
                },
                error: function() {
					success = false;
                }
            });
            
            return success;
        }
        
        $('.login-form').submit(function(e) {
            return maxxfix_login($(this).find('#user').val(), $(this).find('#password').val() , 'login');
        });
        
        $('#adminlogin').submit(function(e) {
            return maxxfix_login($(this).find('#username').val(), $(this).find('#password2').val() , 'forum');
        });
        
        
    });    

    function logout(){
        $.get('fix_logoutscript.php', function() {
            window.location.reload();
        });
    }
	
</script>
