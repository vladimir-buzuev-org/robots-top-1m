        function ShowPrintPreview() {
            window.location.href = (window.location.search == "") ? window.location.href + "?IsPrintPrev=1" : window.location.href + "&IsPrintPrev=1";
        }

        function loadPrintIcon() {

            //in IE 8,7 use custom function to get element by class name since document.getElementsByClassName will not work
            var socialElement;
            if (isIE() == false) {
                socialElement = document.getElementsByClassName('social');
            }
            else {
                if (isIE() > 8) {
                    socialElement = document.getElementsByClassName('social');
                }
                else {
                    socialElement = getElementsByClassName(document.body, 'social');
                }
            }

            var printIconUrl = (window.location.search == "") ? window.location.href + "?IsPrintPrev=1" : window.location.href + "&IsPrintPrev=1";

            if (socialElement.length > 0) {

                //build print preview icon div
                var printPreviewDiv = document.createElement("div");
                //printPreviewDiv.innerHTML = '<div class="print" style="float: right;"><a href="' + printIconUrl + '" alt="Print Page"><img src="/Style Library/eGov/Common/Images/print-icon.png" alt="Print Page"></a></div>';
                printPreviewDiv.innerHTML = '<div class="print" style="float: right;"><a href="#" onClick="ShowPrintPreview();" alt="Print Page"><img src="/Style Library/eGov/Common/Images/print-icon.png" alt="Print Page"></a></div>';

                socialElement[0].innerHTML += printPreviewDiv.innerHTML;
            }
        }

        function getElementsByClassName(node, classname) {
            var a = [];
            var re = new RegExp('(^| )' + classname + '( |$)');
            var els = node.getElementsByTagName("*");
            for (var i = 0, j = els.length; i < j; i++)
                if (re.test(els[i].className)) a.push(els[i]);
            return a;
        }

        function isIE() {
            var myNav = navigator.userAgent.toLowerCase();
            return (myNav.indexOf('msie') != -1) ? parseInt(myNav.split('msie')[1]) : false;
        }
