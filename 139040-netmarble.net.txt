			function getSortedGames(originArrays) {
				var channelingGames = []
					, wholeGames = []
					, _len = originArrays.length;
				for(var i = 0; i < _len; i++) {
					var icon = originArrays[i].icon[0];
					if (icon != undefined && icon.iconCd == 'C') {
						channelingGames.push(originArrays[i]);
					} else {
						wholeGames.push(originArrays[i]);
					}
				}
				
				return wholeGames.concat(channelingGames);
			}
