				loader			= wrap.find(".loader"),
				newsConfirm		= wrap.find(".confirmation")

				var
					$v				= $(v),
					val 			= $v.val(),
					data 			= $v.data(),
					def 			= data.preset,
					type 			= $v.attr("type"),
					classError 		= "error",
					classFilled 	= "postfill",
					label 			= wrap.find("." + data.label);
				if (type == "checkbox") {
					if ($v.is(":checked")) {
						label.removeClass(classError);
					} else {
						valid = false;
						label.addClass(classError);
					}
				} else if($v.is("select")) {
					if(val == null) {
						$v.addClass(classError);
						valid = false;
					} else {
						$v.removeClass(classError);
					}
				}
				else {
					var checkValid = true;
					try {
						if (data.validator) {
							var xp = new RegExp(data.validator.slice(1, -1));
							checkValid = xp.test(val);

						loader.hide();
						alert("An error has occured, please try again later!");
					}
				});
			}
		});
