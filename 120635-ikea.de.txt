      ;(function() {
          // helpers
          var regExp = function(name) {
              return new RegExp('(^| )'+ name +'( |$)');
          };
          var forEach = function(list, fn, scope) {
              for (var i = 0; i < list.length; i++) {
                  fn.call(scope, list[i]);
              }
          };

          // class list object with basic methods
          function ClassList(element) {
              this.element = element;
          }

          ClassList.prototype = {
              add: function() {
                  forEach(arguments, function(name) {
                      if (!this.contains(name)) {
                          this.element.className += ' '+ name;
                      }
                  }, this);
              },
              remove: function() {
                  forEach(arguments, function(name) {
                      this.element.className =
                          this.element.className.replace(regExp(name), '');
                  }, this);
              },
              toggle: function(name) {
                  return this.contains(name)
                      ? (this.remove(name), false) : (this.add(name), true);
              },
              contains: function(name) {
                  return regExp(name).test(this.element.className);
              },
              // bonus..
              replace: function(oldName, newName) {
                  this.remove(oldName), this.add(newName);
              }
          };

          // IE8/9, Safari
          if (!('classList' in Element.prototype)) {
              Object.defineProperty(Element.prototype, 'classList', {
                  get: function() {
                      return new ClassList(this);
                  }
              });
          }

          // replace() support for others
          if (window.DOMTokenList && DOMTokenList.prototype.replace == null) {
              DOMTokenList.prototype.replace = ClassList.prototype.replace;
          }

      function rotateHeadlines(el, headlineSelector, inactiveClass) {
        var wrapper = document.body.querySelectorAll(el);
        var headlines = wrapper[0].querySelectorAll(headlineSelector);

        var headlinesLength = headlines.length;
        var currentHeadline = 0;

        function nextHeadlineCalc() {
          if (currentHeadline + 1 === headlinesLength) return 0;
          return currentHeadline + 1;
        }

        function fadeToNextHeadline () {
          var nextHeadline = nextHeadlineCalc();

          TweenMax.set(headlines[currentHeadline], {
            opacity: 0,
            x: -20,
          });

          TweenMax.to(headlines[currentHeadline], 0.2, {

            onComplete: function () {
              headlines[currentHeadline].classList.add(inactiveClass);
              headlines[nextHeadline].classList.remove(inactiveClass);

              currentHeadline = nextHeadline;

              fadeToNextHeadline();
            },
          });
        }

      function animateMarker(el) {
        var wrapper = document.body.querySelectorAll(el);
        var markers = wrapper[0].querySelectorAll('.marker__element');

        var markerTimeline = {};

        function enterMarker(id) {
          return function (event) {
            if (event.target.classList.contains('marker__area')) {
              markerTimeline[id].play();
            }
          }
        }

        function leaveMarker(id) {
          return function (event) {
            if (event.target.classList.contains('marker__area')) {
              markerTimeline[id].reverse();
            }
          }
        }

        for (var markerId = 0; markerId < markers.length; markerId += 1) {
          // search for marker elements

          var markerLine = markers[markerId].getElementsByClassName('marker__line')[0];
