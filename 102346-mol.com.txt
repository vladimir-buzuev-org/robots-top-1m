    $(document).ready(function ($) {

        // Clear search text box after dialog box hidden
        $('#search').focusout(function () { $('#search').val(''); });

        var games = new Array;

        // Workaround for bug in mouse item selection
        $.fn.typeahead.Constructor.prototype.blur = function () {
            var that = this;
            setTimeout(function () { that.hide(); }, 2500);
        };
        
        $('#search').typeahead({
            source: function (query, process) {
                if (query.length < 3)
                    return;

                $.ajax({
                    url: '/Product/Search?q=' + query,
                    type: 'get',
                    contentType: "application/json; charset=utf-8",
                    data: { query: query },
                    dataType: 'json',
                    success: function (json) {
                        var t;
                        games = new Array;
                        $.each(json, function (i, val) {
                            $.each(val.Items, function (y, item) {
                                t = { id: item.Key, name: item.Value };
                                games.push(t);
                            });
                        });
                        var results = _.map(games, function (fg) {
                            return fg.name;
                        });
                        process(results);
                    }
                });
            },

            displayField: "name",

            minLength: 3,

            items: 100,

            delay: 500,

            matcher: function (item) {
                for (var i = 0; i < item.length; i++) {
                    if (item[i].toLowerCase().indexOf(this.query.trim().toLowerCase()) != -1)
                        return true;
                }

                return false;
            },

            sorter: function (items) {
                if (items.length == 0) {
                    items.push("NOT FOUND");
                }
                return items;
            },

            highlighter: function (name) {
                
                if (name == 'NOT FOUND') {
                    return "<span>No results found.</span>";
                }
                
                var product = _.find(games, function (p) {
                    return p.name == name;
                });
                return product.name[0];
            },

            updater: function (name) {
                var product = _.find(games, function (p) {
                    return p.name == name;
                });
                window.location.href = "/Product/GamehiveMicro/" + product.id;
                return product.name[0];
            }

        });
    })

    $(document).ready(function () {

    var loadedTabs = [];

    function LoadContent(tabIndex) {
        
        var loaded = false;

        for (var i = 0; i < loadedTabs.length; i++) {
            if (loadedTabs[i] == tabIndex) {
                loaded = true;
                break;
            }
        }

        if (loaded == true) {
            return;
        }
        else {
            loadedTabs.push(tabIndex);
        }

        $('#loading').show();
        
        var url = $("#Tab" + tabIndex.toString()).val();
        
        $.ajax(
        {
            type: 'get',
            url: url,
            dataType: 'html',
            success: function (result) {                
                $('#t' + tabIndex.toString()).html(result);                
            },
            error: function (error) {
                $('#t' + tabIndex.toString()).load('/error.html');
                loadedTabs.pop();
            },
            complete: function () {                
                $('#loading').hide();
            }
        });
    }

    function GetParameterByName(name) {
        name = name.replace(/[\[]/, "\\[").replace(/[\]]/, "\\]");
        var regex = new RegExp("[\\?&]" + name + "=([^&#]*)"),
            results = regex.exec(location.search);
        return results === null ? "" : decodeURIComponent(results[1].replace(/\+/g, " "));
    }
