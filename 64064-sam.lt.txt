<script type="text/javascript">
    var baseHref = '//sam.lrv.lt/lt/';
    function changeEventsMonth(month)
    {
        $('#loader').show();
        jQuery.ajax({
            url: baseHref + 'events/getCalendarAjax/calendarDate.' + month+'/type.'+["agenda","events","meetings"],
            success: function (data) {
                jQuery('#events_calendar').html(data);

                var day = jQuery('#day-events').data('day');
                if(day)
                {
                    day = day.replace(/-/g, '');
                    jQuery('.day_' + day + ' .ui-state-default').addClass('ui-state-active');
                }
            }
        });
    }

    function currenDayEvents(obj, day)
    {
        jQuery.ajax({
            url: baseHref + 'events/getDayEventsAjax/calendarDate.' + day + '/type.'+["agenda","events","meetings"],
            success: function (data) {
                jQuery('#day-events').html(data).data('day', day);
            }
        });

        jQuery('#events_calendar .ui-state-default').removeClass('ui-state-active');
        jQuery(obj).addClass('ui-state-active');

        return false;
    }

    function showContent(category, element)
    {
        $('#day-events .schedule').hide();
        $('#day_ev_'+category+'.schedule').show();
        $('.all_schedule a').hide();
        $('.all_schedule a.day_ev_'+category).show();
        $('#day-events .category a').removeClass('active');
        $(element).addClass('active');
    }
