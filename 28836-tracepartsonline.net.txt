        width,
        height
    )
    {

        var left = (screen.width/2)-(width/2);

    function saveSearch_Command(){

            var searchType = "0";
            var searchClass = "TRACEPARTS";
            var searchNode = "TRACEPARTS";

            if
            (
                searchType == "1" ||
                searchType == "4"
            )

               searchMode_sSimple = $("#searchSaveMode").val();
            }

                         "&sType=" + searchType + 
                         "&sMode=" + searchMode_sSimple + 
                         "&sClass=" + searchClass + 
                         "&sNode=" + searchNode;

            saveSearch_JqCommand(params);
        }

    }

                closeSearchSaveForm();
                showUserLoginCtrl(json.errors, true, function () {  showSearchSaveForm(); });
            }
            else if(json.errors) 
            {
                $("#searchSaveInfo").html(json.errors);
            }
            else
            {
                if (json.results)
                {
                   closeSearchSaveForm();
                } 
            }
        }
    }
