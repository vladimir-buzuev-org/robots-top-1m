function requestTrial(email){
	var baseURL =    "https://www.clearslide.com/rest/api/auth/signup";
	var pageURL = jQuery(location).attr('href');
	var days = jQuery('#trial-days').val();

	  }
	});
}

function requestTrial(email){

	});
}

	  return true;
