    var newsletterformCaptchaStartTime = 0;

    function newsletterformCaptchaRefresh() {

        var captchaimg = document.getElementById("newsletterform_captchaimg");
        var captchakeyelem = document.getElementById("newsletterform_captchakey");

        newsletterformCaptchaStartTime = new Date().getTime();

    function newsletterformCaptchaTimer() {
        var now = new Date().getTime();
        if ((now - newsletterformCaptchaStartTime) > 60000 && $('.captcha:visible').length != 0) {
        	newsletterformCaptchaRefresh();
        }
        var captchatimer = document.getElementById("newsletterform_captchatimer");
        if (!captchatimer) {
            // captcha has been removed
            return;
        }
        var width = Math.floor((60000 - (now - newsletterformCaptchaStartTime)) / 60000 * 64);
        captchatimer.innerHTML = "<div class=\"form_captchatimer_bar\" style=\"width:" + width + "px;\"></div>";
        window.setTimeout(newsletterformCaptchaTimer, 500);
    }

<script type="text/javascript">

	/* attach a submit handler to the form */
	$("#formnewsletter").submit(function(event) {
	
		/* stop form from submitting normally */
		event.preventDefault();
		/* Validate form */
		var formInstance = $('#formnewsletter').parsley();
		if (formInstance.isValid()) {

			/* get some values from elements on the page: */
			var url = $(this).attr('action');
			$.ajax({
				type: "POST",
				url: url,
				data: $( "#formnewsletter" ).serialize(),
				dataType: "json"
			}).done(function( response ) {

				/* Initialize UI */  
				$('.js-newsletterform-error-messages').addClass('h-hidden');
				$('.js-newsletterform-error-captcha').addClass('h-hidden');
				$('.js-newsletterform-error-fatal').addClass('h-hidden');
			
				if (response.errorData === 'true') {
					$('.js-newsletterform-error-messages').removeClass('h-hidden');
					if (response.errorCaptcha === 'true') {
						$('.js-newsletterform-error-captcha').removeClass('h-hidden');
					}
					if (response.errorFatal === 'true') {
						$('.js-newsletterform-error-fatal').removeClass('h-hidden');
					}
				}
				else {
					if (response.redirect != undefined) {
						location.href = response.redirect;
					}
				}
			})
			.fail(function(jqXHR, textStatus) {
				$('.js-newsletterform-error-messages').removeClass('h-hidden');
				$('.js-newsletterform-error-fatal').removeClass('h-hidden');
			});
		}
	});
</script>
