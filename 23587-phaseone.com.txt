<script>
    function toggleSubDropdownMenu(event, submenu) {
        event = event || window.event;
        event.preventDefault();

        var windowWidth = $(window).width();
        if (windowWidth < 1024) {
            window.location.href = submenu.href;
        } else {
            var activeSubdropdownClass = submenu.className;

            $.each($(".sub-dropdown-menu"), function () {
                var dropdownId = $(this).attr("id");
                if (dropdownId !== activeSubdropdownClass) {
                    $(this).hide();
                }
            });

            var activeSubdropdown = $("#" + activeSubdropdownClass);
            if (activeSubdropdown.length) {
                activeSubdropdown.slideToggle();
            } else {
                window.location.href = submenu.href;
            }
        }
    }
