<script>

    $(document).ready(function() {

        $('#txtPhone').mask('(00) 00000-0000');
        $('#txtCep').mask('00000-000');

        var sending = false;
        var $btnSubmit = $('#btnSubmit');

        $('#frmContact').submit(function (e) {

            if (sending) {
                return false;
            }

            sending = true;
            $btnSubmit.text('Enviando...');

            e.preventDefault();

            var data = {
                name: $('#txtName').val(),
                email: $('#txtEmail').val(),
                phone: $('#txtPhone').val(),
                cep: $('#txtCep').val(),
                utm_source: $('#txtUtmSource').val(),
                utm_content: $('#txtUtmContent').val()
            };

            var $form = $(this);

            $.ajax({
                type: 'POST',
                url: '/submit-contact.php',
                dataType: 'json',
                data: data,
                success: function (response) {

                    sending = false;

                    $btnSubmit.text('Enviar');

                    $('input.error').removeClass('error');

                    if (response.success) {

                        $('#message').removeClass('msg-error').addClass('msg-success').text(response.message).fadeIn();

                        $form[0].reset();

                    } else {

                        if (response.errors) {
                            $.each(response.errors, function(i, error) {
                                $('input[name="' + error.field + '"]').addClass('error');
                            });

                            $('#message').removeClass('msg-success').addClass('msg-error').text('Preencha os campos em vermelho').fadeIn();

                        } else if(response.message) {

                            $('#message').removeClass('msg-success').addClass('msg-error').text(response.message).fadeIn();

                        }

                    }

                },
                error: function () {

                    sending = false;

                    $btnSubmit.text('Enviar');

                    console.log('erro');

                }

            });

        });


    });

</script>
