function mobileBanners(){
    imgs = jQuery("#banners_wrapper.slider img");
    // more that one image do a slideshow
    if(imgs.length>1){
    	// add slider navigation
    	for(var x = 0; x < imgs.length; x++){
    		jQuery("#mobileBannerNav").append("<div class='navigation' id='navball"+x+"' onclick='mobileBannersJump("+x+")'>"+parseInt(x+1)+"</div>");
    	}
    	jQuery(imgs[0]).fadeIn();
        jQuery("#navball0").addClass("active");
        // execute this every 5 seconds
    	setInterval(function(){
                mobileBannersJump();
    	},5000);
    }
    else{
        jQuery("#banners_wrapper.slider img").fadeIn("fast");
    }
}

/* function that handles the jumping of banners */
function mobileBannersJump(value){
        // if force jump
	if( value>=0 ) noJumping = false;
	if( !noJumping ){
		// reset index
		if(indexBanner==imgs.length) indexBanner = 0;
		// if force jump
		if( value>=0 ) indexBanner = value;
		jQuery(".navigation").removeClass("active");
		jQuery("#navball"+indexBanner).addClass("active");
		jQuery(imgs[prevBanner]).fadeOut("fast");
		jQuery(imgs[indexBanner]).fadeIn("slow");
		prevBanner = indexBanner;
		indexBanner++;
	}
}

/* while mouse over the banners in mobile no transitions*/
function noJumpingAllowed(){
	jQuery("#banners_wrapper.slider").mouseover(function(){ noJumping = true });
	jQuery("#banners_wrapper.slider").mouseleave(function(){ noJumping = false });
}
