    $('#newsletter-subscribe-button').on('click', function (e) {
		e.preventDefault();

		var honeypotField = $('#newsletter-email-first').val();

		if (honeypotField == "")
        {
		    var btn = $(this);
		    btn.attr("disabled", true);

		    var subscribe = 'true';
		    if($('#newsletter-unsubscribe').is(':checked')) {
			    subscribe = 'false';
		    }
		    var email = $("#newsletter-email").val();

		    $.ajax({
			    cache: false,
			    type: "POST",
		        //url: btn.data("href"),
			    url: "/Subscribe",
			    data: { "subscribe": subscribe, "email": email },
			    success: function (data) {
				    OnNewsletterSubscribeResponse(data.Result, !data.Success);
				    if (!data.Success) {
					    btn.removeAttr("disabled");
				    }
			    },
			    error:function (xhr, ajaxOptions, thrownError) {
				    OnNewsletterSubscribeResponse(thrownError, true);
				    btn.removeAttr("disabled");
			    }
		    });
		}
		return false;
	});

    function OnNewsletterSubscribeResponse(msg, isError) {
        displayNotification(msg, isError ? "error" : "success", isError /* sticky */);
    };

</script>
