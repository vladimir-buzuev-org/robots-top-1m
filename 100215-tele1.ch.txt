            didScroll = true;

            if($(document).scrollTop() > pos_mobile.top)
            {
                if($('#header_mobile').data('size') == 'big')
                {
                    $('#header_mobile').data('size','small');

                  {
                    $('#header_mobile').data('size','big');
                    //$('#mainlogo').stop().animate({
                    //    top:'17px'

                        width:'54px'
                    },600);
                    $('.mobile').css('position', 'relative');
                    $('#header_mobile .h_lower').fadeIn(600);
                  }  
              }
            // desktop
            if($(document).scrollTop() > height_maxi) {
                if($('#header_desktop').data('size') == 'big')
                  {
                    $('#header_desktop').data('size','small');
                    $('.desktop').css('position', 'fixed');
                    $('#content').css('padding-top', '120px')
                    //$('#header_desktop').css('top', '0px');
                 }
            } else {
                if($('#header_desktop').data('size') == 'small')

                    $('#content').css('padding-top', '30px')
                    //$('#header_desktop').css('top', '0px');
                }  
            }
        });

        // Hide Header on on scroll down
        var didScroll;
        var lastScrollTop = 0;
        var delta = 300;
        var ddelta = 100;
        var navbarHeight = 45;

        setInterval(function() {
            if (didScroll) {
                hasScrolled();
                didScroll = false;
            }
        }, 250);

        function hasScrolled() {
            var st = $(this).scrollTop();

            // MOBILE
            // If they scrolled down and are past the navbar, add class .nav-up.
            // This is necessary so you never see what is "behind" the navbar.
            if (st > lastScrollTop && st > (navbarHeight + pos_mobile.top)) {
                // Scroll Down
                if (lastScrollTop + delta <= st) {
                    $('#header_mobile').removeClass('nav-down').addClass('nav-up');
                }
            } else {
                // Scroll Up
                if(st + $(window).height() < $(document).height()) {
                    $('#header_mobile').removeClass('nav-up').addClass('nav-down');
                }
            }
            
            // DESKTOP
            if (st > lastScrollTop && st > pos_desktop.top) {
                // Scroll Down
                if (lastScrollTop + ddelta <= st) {
                    $('#header_desktop').removeClass('nav-down').addClass('nav-up');
                }
            } else {
                // Scroll Up
                if(st + $(window).height() < $(document).height()) {
                    $('#header_desktop').removeClass('nav-up').addClass('nav-down');
                }
            }

            lastScrollTop = st;
        }

                } else {
                    window.location.href = "/search";
                }
            })
        })

                                        $content.find('.vex-dialog-input').html(data);
                                        if ($content.find('.vex-dialog-input input').length > 2) {
                                            $content.find('.vex-dialog-button-primary').show();
                                        }else {
                                            $content.find('.vex-dialog-button-secondary').val('Ok').css('float', 'right');
                                        }
                                    });

                                    //e.preventDefault();
                                    $content.find('.vex-dialog-input').html('<div class="loading">&nbsp;</div>');
                                    $content.find('.vex-dialog-button-primary').hide();
