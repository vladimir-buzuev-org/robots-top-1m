		  function doChannel($this) {
				var channels = new Array('twitterchannel','facebookchannel','youtubechannel','blogchannel','presschannel','linkedinchannel');
				if (currentChannel!="") {
						if ($this.data('portlet')===currentChannel) return;
						
						var oldChannel = currentChannel;
						currentChannel = $this.data('portlet');
						var currHeight = 175; //jq('#socialmedia div.portlet.' + currentChannel).height();
						jq('#socialmedia div.portlet.' + currentChannel).css('top',currHeight);
						jq('#socialmedia div.portlet.' + currentChannel).css('opacity',0);
						jq('#socialmedia div.portlet.' + currentChannel).show();
						//if (jq.support.opacity) jq('#socialmedia div.portlet.' + currentChannel).fadeIn();
						//else jq('#socialmedia div.portlet.' + currentChannel).show();

						jq('#socialmedia div.portlet.' + currentChannel).animate({top: 0}, { duration: 500, queue: false });
						jq('#socialmedia div.portlet.' + currentChannel).animate({opacity: 1}, { duration: 500, queue: false });
						jq('#socialmedia div.portlet.' + oldChannel).animate({top: (currHeight * -2)}, { duration: 500, queue: false });
						jq('#socialmedia div.portlet.' + oldChannel).animate({opacity: 0}, { duration: 500, queue: false, complete: function(){
							jq('#socialmedia div.portlet.' + oldChannel).hide();
							jq('#socialmedia div.portlet.' + oldChannel).css('top',0);
							jq('#socialmedia div.portlet.' + oldChannel).css('opacity',1);
						}});
						/*
						if (jq.support.opacity) jq('#socialmedia div.portlet.' + oldChannel).fadeOut();
						else jq('#socialmedia div.portlet.' + oldChannel).hide();
						*/					
				} else {
						jq('#socialmedia div.wrapper').animate({left: '0'}, { duration: 500, queue: false });
						jq('#belt').animate({left: '-430'}, { duration: 500, queue: false });
						jq('#belt').animate({right: '505'}, { duration: 500, queue: false });
						if (jq.support.opacity) jq('#socialmedia div.wrapper').fadeIn();
						else jq('#socialmedia div.wrapper').show();

						currentChannel = $this.data('portlet');
						if (jq.support.opacity) jq('#socialmedia div.portlet.' + currentChannel).fadeIn();
						else jq('#socialmedia div.portlet.' + currentChannel).show();
						//jq('#socialmedia div.portlet.' + currentChannel).animate({right: '-20'}, { duration: 500, queue: false });
						//jq('#socialmedia div.buttons').animate({right: '445'}, { duration: 450, queue: false });
				}
				channelFill(currentChannel);
		  }

		  function channelFill(which) {

						});
						break;
				case 'presschannel':

						});
						break;
				case 'twitterchannel':
				      jq(".tweet_first").html('');
						jq(".tweet_first").tweet({
							 twittercache: twittercache,
							 username: "UnifyCo",
							 join_text: "auto",

						});
						break;

						fbFetch();
						break;
				case 'youtubechannel':
						jq('#youtubetitle').dotdotdot();
						jq('#youtubedescription').dotdotdot();
						break;
				}
