        html +=  '<li>' + day + '</li>';
      }else{
        if(year == (current_date.getYear() + 1900) && month == (current_date.getMonth() + 1) && day == current_date.getDate()){
          html += '<li class="selected">';
        }else{
          html += '<li>';
        }
         html += '<a href="/archives/'  + year + '/' + ('00' + month).slice(-2) + '/' + ('00' + day).slice(-2) + '/">' + day  + '</a></li>';
       }
     }
     $("#calender_d2").html(html);
     $("#cal_date").text(year + "年" + month + "月");
   }
   write_cal(current_year, current_month);

  $("#cal_next").click(function(){
    current_month++;
    write_cal(current_year, current_month);
    return false;
  });

  $("#cal_prev").click(function(){
    current_month--;
    write_cal(current_year, current_month);
    return false;
  });
