<script type="text/javascript">
    $(document).ready(function(){

        var slideShow={
            currentImage:0,
            slideshowTimer:undefined,
            delay:3000,
            next:function(){

                var image=$('.slideshow-image-'+slideShow.currentImage);
                var slides=0;

                if(image.siblings()){
                    slides=image.siblings().filter('.ml-card').length+1;
                }

                if(slideShow.currentImage<(slides-1)){
                    slideShow.select(slideShow.currentImage+1);
                }else{
                    slideShow.select(0);
                }          
            },
            select:function(index){

                $('.slideshow-image-'+slideShow.currentImage).fadeOut(500);
                $('.slideshow-image-'+index).fadeIn(500);
                slideShow.currentImage=index;

                $(".slideshow-next").removeClass('selected');
                $(".slideshow-next-"+slideShow.currentImage).addClass('selected');
            },
            start:function(){

                if(!$(".ml-card")) return false;

                slideShow.slideshowTimer = setInterval(function() {
                    slideShow.next();
                }, slideShow.delay);
                
            },
            init:function(delay){

                if(!$(".ml-card-slideshow").length) return false;

                slideShow.delay=delay;

                $(".ml-card-slideshow").find(".ml-card").hide();

                //add navigation container
                var nextContainer=$('<div class="next-container"></div>');
                nextContainer.appendTo($(".ml-card-slideshow"));

                //add navigation buttons and events
                $(".ml-card-slideshow > div.ml-card").each(function(index){
                    var image=$(this);
                    image.attr('data-index',index);
                    image.addClass('slideshow-image-'+index);

                    var next=$('<div class="slideshow-next slideshow-next-'+index+'"></div>');

                    next.appendTo(nextContainer);

                    next.click(function(){
                      
                      //do not switch if image is already selected
                      if(next.hasClass('selected')) return true;

                      slideShow.select(index);
                      
                      clearInterval(slideShow.slideshowTimer);
                      slideShow.start();
                    });
                });

                slideShow.select(0);
                return true;
            }
        }

        if(slideShow.init(15000)){
            slideShow.start();
        }        
        
    });
</script>

<script>
$( document ).ready(function() {
    $(window).scroll(function(e){
      parallax();
      parallaxstartup();
      parallaxsales();
    });

    function parallax(){
      var scrolled = $(window).scrollTop();
      $('.parallax-object').css('top',-(scrolled*0.07)+'px');
    }

    function parallaxstartup(){
      var scrolled = $(window).scrollTop();
      $('.parallax-text-startup').css('margin-top',-(scrolled*0.1)+'px');
    }


    function parallaxsales(){
      var scrolled = $(window).scrollTop();
      $('.parallax-sales').css('margin-right',-(scrolled*0.01)+'px');
      $('.parallax-sales').css('margin-left',-(scrolled*0.001)+'px');
    }

});
</script>
