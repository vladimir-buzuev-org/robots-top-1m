          elem.className += ' m-side-nav__header--afternoon';

        case(hour >= 15 && hour <= 17):
          elem.className += ' m-side-nav__header--evening';
          break;
        case(hour >= 18 && hour <= 20):
          elem.className += ' m-side-nav__header--dusk';
          break;
        case(hour == 21 || hour == 22 || hour == 23 || hour == 0 || hour == 1):
          elem.className += ' m-side-nav__header--midnight';
          break;
        default:
          break;
      }
    }
    if (overlay) {
      overlay.style.display = 'none';
