<script>
    jQuery(function ($) {
        var dateFormat = "MM d yy",
        from = $("#from").datepicker({
            defaultDate: "+1w",
            changeMonth: true,
            dateFormat: "MM d yy"
//            numberOfMonths: 3
        }).on("change", function () {
                    to.datepicker("option", "minDate", getDate(this));
                    $("#from_date").val($("#from").val());
                }),
        to = $("#to").datepicker({
            defaultDate: "+1w",
            changeMonth: true,
            dateFormat: "MM d yy"
//            numberOfMonths: 3
        }).on("change", function () {
                    from.datepicker("option", "maxDate", getDate(this));
                    $("#to_date").val($("#to").val());
                });

        function getDate(element) {
            var date;
            try {
                date = $.datepicker.parseDate(dateFormat, element.value);
            } catch (error) {
                date = null;
            }

            return date;
        }
    });

</script>
