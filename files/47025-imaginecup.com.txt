            if ('localStorage' in window && window['localStorage'] !== null) {

        function isCookieSet() {
            var cookie = document.cookie.split(";");
            var flag = false;
            for (i = 0; i < cookie.length; i++) {
                if (cookie[i].substr(1, 12) == "cookieAccess") {
                    flag = true;
                }
            }
            return flag;
        }
        function doShowCookie() {
            if (CheckBrowser()) {
                var cookieResponse = isCookieSet();
                if (cookieResponse == true) {
                    $("#CookieAlert").hide();
                }
                else {
                    $("#CookieAlert").show();
                    $("#CookieAgree").click(function (e) {
                        e.preventDefault();
                        var host = window.location.host.toString().split('.');
                        var hoststring = "domain=" + host.slice(1, host.length).join('.');
                        var d = new Date();
                        d.setTime(d.getTime() + (3653 * 24 * 60 * 60 * 1000));
                        var expireTime = "expires=" + d.toUTCString();
                        document.cookie = "cookieAccess=true;" + expireTime + "; path=/;" + hoststring;
                        $("#CookieAlert").hide();
                        return false;
