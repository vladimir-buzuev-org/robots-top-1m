<script type="text/javascript" charset="utf-8">
$(function() {
    var topBtn = $('#page-top');
    topBtn.hide();
	var timerSC = false;
	
    //スクロールが100に達したらボタン表示
    $(window).scroll(function () {
		var h = window.innerHeight ? window.innerHeight: $(window).height();


		var w = $(window).innerWidth()-5;
		var x = 1200;
		if (w <= x && w > 670) {
			$("header").css("display","none");
			if (timerSC !== false) {
				clearTimeout(timerSC);
			}
			
			timerSC = setTimeout(function() {
				$("header").css("top",$(this).scrollTop() * (x / w)).fadeIn("fast");
			}, 200);
		} else {
			clearTimeout(timerSC);
			$("header").css("top",0);
		}

		var footerOffset = $("footer").offset();
		if ($(this).scrollTop()>footerOffset.top-h+10) {
			topBtn.css("position","absolute");
			topBtn.css("top",footerOffset.top-60);
		} else {
//			topBtn.css("top",$(this).scrollTop()+h-60);
			topBtn.css("position","fixed");
			topBtn.css("top","auto");
			topBtn.css("bottom",0);
		}
        if ($(this).scrollTop() > 100) {
            topBtn.fadeIn();
        } else {
            topBtn.fadeOut();
        }
    });
    //スクロールしてトップ
    topBtn.click(function () {
        $('body,html').animate({
            scrollTop: 0
        }, 500);
        return false;
    });
	
	$('meta[property=og\\:image]').attr('content', 'https://sega.jp/images/ogpimg.png');
	
	var icon_color = "";
	if (icon_color) {
		$('body').addClass("page-topics-"+icon_color);
	}

	//海外リンク
	$(".acordion").hide();
	$("div.country").click(function(){
	// $(this).next().slideToggle();
		$(this).prev().slideToggle();
		$(this).toggleClass("open");
	});
	$(".acordion .close").click(function(){
		$(this).parent().slideToggle();
	});

	$("#megaon").hover(
		function() {
		//マウスホバー
		$("#header-bottom").addClass("hover");
		},
		function() {
        //マウスアウト
		$("#header-bottom").removeClass("hover");
		}
	);


});
