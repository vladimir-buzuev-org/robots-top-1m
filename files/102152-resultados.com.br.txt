$(document).ready(function()
{
	$(window).scroll(function()
	{
  		$('#msgbox').animate({top:$(window).scrollTop()+"px" },{queue: false, duration: 350});  
	});
	
	center = function () 
	{    
	    this.css("position","absolute");    
	    this.css("left", ( $(window).width() - this.width() ) / 2+$(window).scrollLeft() + "px");    
	    return this;
	}
});

function show(text,tp) 
{    
    if (tp=='success')
    {
        $("#msgbox").toggleClass("msgbox_success");
        $("#fechaDiv").toggleClass("fechaDiv_success");
        $('#msgbox_titulo').toggleClass("msgbox_titulo_success");
        $('#msgbox_titulo').html('SUCESSO');
    }
    else
    {
        $("#msgbox").toggleClass("msgbox_error");
        $("#fechaDiv").toggleClass("fechaDiv_error");
        $('#msgbox_titulo').toggleClass("msgbox_titulo_error");
        $('#msgbox_titulo').html('ATENÇÃO');
    }
    
    $('#msgbox').toggle();
    $('#msgbox_text').html(text);
    $('#msgbox').center();
    $('#msgbox').animate({ top: $(window).scrollTop() + "px" }, { queue: false, duration: 350 });
    $('#msgbox').show();
    $('#close_message').show();
    $('#msgbox_text').show();
}

function hide() {
    $('#close_message').hide();
    $('#msgbox_text').hide();
    $('#msgbox').hide();
}
