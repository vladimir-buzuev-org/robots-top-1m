function createCookie(name,value,days){ 
var expires = ""; 
  if (days) { 
   var date = new Date(); 
   date.setTime(date.getTime()+(days*24*60*60*1000)); 
   expires = "; expires="+date.toGMTString(); 
  } 
  document.cookie = name+"="+value+expires+"; path=/"; 
  return true; 
 } 

function cookie(){ 
 var cookieVal=readCookie("newVisitor"); 
 if (cookieVal == null || cookieVal=='true') 
 { 

var userLang = navigator.language || navigator.browserLanguage; 
//console.log(userLang); 
if (userLang.substr(0, 5).toLowerCase()=='en-us' || userLang.substr(0, 5).toLowerCase()=='en-ca'){ 
showMarketLayer(); 
} 
 }else{ 
if (cookieVal!='#'){ 
window.location.href=cookieVal; 
} 
 } 
 function readCookie(name) { 
  var nameEQ = name + "="; 
  var ca = document.cookie.split(';'); 
  for(var i=0;i < ca.length;i++) { 
   var c = ca[i]; 
   while (c.charAt(0)==' ') c = c.substring(1,c.length); 
   if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length,c.length); 
  } 
  return null; 
 } 
} 
cookie();

   var c = ca[i];
