<h1 class="start" style="padding-top:25px;">Führerschein in Sicht - Fahrschule aussuchen</h1>
<p>Sie wollen <a href="/Verkehrsrecht/Fuehrerschein/Fuehrerschein_Fahrerlaubnisklassen.html" class="menu">einen Führerschein machen</a> und wollen wissen wann Sie mit der <a href="/fahren_lernen/Tipp11index.html" class="menu">Fahrausbildung beginnen</a> können? Sie wollen den <a href="/fahren_lernen/Tipp33index.html" class="menu">Pkw-Führerschein mit 17</a> machen und am begleiteten Fahren teilnehmen? Sie <a href="https://www.fahrschulen.de" class="menu">suchen eine Fahrschule</a> und wollen wissen, <a href="https://www.fahrschulen.de/PREISE/" class="menu">wieviel die Fahrausbildung kostet</a> und welche <a href="/fahren_lernen/Tipp24index.html" class="menu">Möglichkeiten der Finanzierung</a> es gibt? Sie suchen <a href="/fahren_lernen/" class="menu">Tipps zur Fahrausbildung</a> oder zur <a href="/fahren_lernen/Tipp2index.html" class="menu">Auswahl der richtigen Fahrschule</a>? Sie wollen endlich die Freiheit auf zwei Rädern erleben und einen <a href="/fahren_lernen/Tipp7index.html" class="menu">Motorradführerschein</a> machen? Hier können Sie Ihre <a href="https://www.fahrschulforum.de/" class="menu">Fragen zur Führerscheinausbildung</a> stellen.<br></p>

<h2 class="start" >Theorie lernen und praktische Führerschein Ausbildung</h2>
<p>Sie haben Ihre Fahrschule schon gefunden und suchen ein <span class="alerttext"><a href="https://www.fuehrerschein-lernsystem.de/login.asp?lliwedsf" class="menu">Online-Lernsystem</a></span> mit allen aktuellen amtlichen Prüfungsfragen und Erklärungen zu allen Fragen für alle Führerscheinklassen, eine <span class="alerttext"><a href="http://www.iführerschein.de/" class="menu">App zum Theorie Lernen</a></span> für <span class="alerttext"><a href="https://itunes.apple.com/de/app/ifuhrerschein/id413246093?mt=8" target="_parent" class="menu">iOS</a></span>, <span class="alerttext"><a href="https://play.google.com/store/apps/details?id=com.fahrschule.de.full" target="_parent" class="menu">Android</a></span> und <span class="alerttext"><a href="https://www.microsoft.com/de-de/store/p/i-f%C3%BChrerschein-2017/9wzdncrdls51" target="_parent" class="menu">Windows Phone</a></span>, eine <a href="/Fuehrerschein_Software.html" class="menu">Lernsoftware zum Offline-Üben</a>, <a href="/fahren_lernen/Tipp32index.html" class="menu">Lehrmaterial</a> in <a href="/fahren_lernen/Tipp22index.html" class="menu">Fremdsprachen</a>, <a href="/Testbogen/" class="menu"> 5000 kostenlose Testbögen</a> für jede Führerscheinklasse, den kompletten <a href="/Fragenkatalog/" class="menu">amtlichen Fragenkatalog für die Theorieprüfung</a> mit Fehlerstatistik, <a href="/fahren_lernen/" class="menu">Tipps zum Fahren lernen</a>, ein <a href="https://play.google.com/store/apps/details?id=com.fahrschule.de.signtrainer" class="menu">Spiel zum Lernen der Verkehrszeichen</a> oder einem <a href="/Verkehrsuebungsplaetze.html"  class="menu">Verkehrsübungsplatz</a>? Wollen Sie sich <a href="/Fahrstunde.html" class="menu">Lehrfilme zur praktischen Fahrausbildung ansehen</a> oder in unserem <a href="https://www.fahrschulforum.de/" class="menu">Fahrschulforum</a> mit anderen Fahrschülern oder Fahrlehrern diskutieren? Sie wollen wissen, wie die <a href="/Verkehrsrecht/Pruefungsrichtlinie/" class="menu">theoretische oder die praktische Führerscheinprüfung</a> abläuft? Hier gibt es bereits Antworten auf die meisten Fragen.<br></p>

<h2 class="start" >Geschwindigkeit, rote Ampel, Alkohol, Drogen, ..., Führerschein weg</h2>
<p>Sie haben sich (mit Ihrem <a href="fahren_lernen/Tipp3index.html" class="menu">Führerschein auf Probe</a>) nicht an die <a href="Verkehrsrecht/StVO/" class="menu">Straßenverkehrsordnung (StVO)</a> gehalten und wollen im <a href="Verkehrsrecht/Bussgeld/" class="menu">Bußgeldkatalog</a> nachsehen, was jetzt passiert? Sie haben <a href="fahren_lernen/Tipp4index.html" class="menu">zu viele Punkte im Fahreignungsregister</a> und wollen diese abbauen? Ihr Führerschein ist weg und jetzt müssen Sie zur <a href="MPU_Beratung/"  class="menu">MPU</a> und suchen einen auf <a href="Verkehrsrecht/Rechtsanwalt/frmSearch.asp" class="menu">Verkehrsrecht spezialisierten Anwalt</a>?<br></p>

<h2 class="start" >Infos und Tipps für alle, die den Führerschein schon haben</h2>
<p>Wenn Sie bereits Inhaber einer Fahrerlaubnis sind, finden Sie hier Informationen zu den Themen <a href="Verkehrsrecht/" class="menu">Verkehrsrecht</a>, <a href="Verkehrsrecht/Fuehrerschein/Fuehrerschein.html" class="menu">EU-Führerschein</a> und zur <a href="Verkehrsrecht/Fuehrerschein/Fuehrerschein_Besitzstandsregelungen.html" class="menu">Umschreibung eines alten Führerscheins</a>.<br>

const ipRemote = "194.247.31.31";
var plz,city,country	
plz = "";
city = "";
country = "";

var bValidAdsPages = true;
var bValidBLPages = false;
var getCurrentPosition = false;



function fetchGeoIPFree( callback ) {
	console.log("Fetching data from GeoIPFree");
	
	// we skip fetchGeoIPFree atm.
	if (callback) callback(plz,city,ipRemote);
	return;
	
	
	$.ajax({
			type: 'GET',
			dataType: "json",
			url: "https://geoipfree.com/json/"+ipRemote,
			data: {},
			success: function(data) {
				console.log(data);
				if ( plz == "" ) {
					plz = data['zip_code'];
				}
				if ( city == "" ) {
					city = data['city'];
				}
				if ( country == "" ) 
					country = data['country_name'];
				
				if (callback) callback(plz,city,ipRemote)
				
			},
			error: function () { console.log('error'); } 
			}
	);
};


function fetchGoogleMap(latitude, longitude, callback ) {
	
	console.log("Fetching data from GoogleMaps");

	$.ajax({
		type: 'GET',
		dataType: "json",
		url: "https://maps.googleapis.com/maps/api/geocode/json?latlng="+latitude+","+longitude+"&sensor=false",
		data: {},
		success: function(data) {

			$.each( data['results'],function(i, val) {
				$.each( val['address_components'],
						function(i, val) {
						
						if (val['types'] == "postal_code")
							if ( plz == "" ) {
								plz = val['long_name'];
							}

						if (val['types'] == "locality,political")
							if ( city == "" ) {
								city = val['long_name'];
							}
							
						if (val['types'] == "country,political")
							if ( country == "" ) 
								country = val['long_name'];

				});
			});
			console.log('Success');
			
			if (plz!='') {
				if (callback) callback(plz,city,ipRemote)
			} else
				fetchGeoIPFree(callback);
		},
		error: function () { console.log('error'); } 
	}); 
};
	
function displayGeoAd(plz,city,ipRemote) {

	if (bValidBLPages) {
		if ( $('#Ort').val() == "" && city != "" ) 
			$('#Ort').val(city);
	}
	
	if (bValidAdsPages)
		$("#werbung").load("/werb/geoad.ajax.asp?zipcode="+plz+"&city="+city+"&ip="+ipRemote);
}

function handle_geolocationerror(err) {
    fetchGeoIPFree(displayGeoAd);
}


function initGeoLoc() {
    if (bValidBLPages && navigator.geolocation) {
        navigator.geolocation.getCurrentPosition(function (position) {
            console.log("Got position!");
			getCurrentPosition = true;
			fetchGoogleMap(position.coords.latitude,position.coords.longitude, displayGeoAd );
        },handle_geolocationerror, {maximumAge: 75000});
    } else if (!getCurrentPosition)
		fetchGeoIPFree(displayGeoAd);
}

$(document).ready( function () {    
    initGeoLoc();
});
