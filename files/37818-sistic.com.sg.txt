const _TOKEN_EXPIRED = 600;
const _ERRORS = 300;
const _NUMBER_OF_RETRIES = 3;

function handleError(xhr, errorCallback = null, originalAjax){
	let statusMessage = JSON.parse(xhr.responseText).statusMessage;
	statusMessage = (!statusMessage)? LanguageUtil.getMessage("general.error.message") : statusMessage;
	if(xhr.status < _ERRORS) return true; // do not run error if the status is under 400

	if(xhr.status === _TOKEN_EXPIRED) {
		if(originalAjax.retries < _NUMBER_OF_RETRIES){
			originalAjax.retries += 1;
			$j.ajax(originalAjax);
			return false;
		}
	}

	if(errorCallback === null){
		AjaxUtil.handleError(xhr.status, xhr.responseText , statusMessage);
	}else{
		let handleError = errorCallback(xhr.status, xhr.responseText , statusMessage);
		if (handleError === undefined || handleError) {
			handleError = AjaxUtil.handleError(xhr.status, xhr.responseText , statusMessage)
		}
		return handleError;
	}
};

			var node = el.firstChild;
			do {
