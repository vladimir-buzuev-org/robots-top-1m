var wwidth;
(function($) {

    if (!$('body').hasClass('ie8')) {
        $('.hd-mobile .logo').html($('.main-header .hd-upper .logo').html());
    }

    function doOnOrientationChange()
      {
        switch(window.orientation)
        {
          case -90:
          case 90:
            //alert('landscape');
            break;
          default:
                if($(window).scrollTop() > 50){
                    //$('.header.main-header').removeClass('sticky');
                    $('.header.main-header').css({'top': '0px', 'left': '0px', 'z-index': 100, 'width': '100%','position':'fixed !important'}).addClass('sticky');
                    //$('.main-wrapper').css('left','0')
                    window.scrollTo(0,$(window).scrollTop() + 1);
                }
                $('<style></style>').appendTo($(document.body)).remove();
            break;
        }
      }
    if (window.addEventListener) {
    	window.addEventListener('orientationchange', doOnOrientationChange);
    }


    Essilor.stickyHeader = function() {
        //Generate mobile header content
        if (!$('body').hasClass('ie8')) {
            $('.hd-mobile .logo').html($('.main-header .hd-upper .logo').html());
        }
        if ($('.header.main-header').length !== 0) {
            $(function() {
                var left = 0;
                var width = '100%';
                var window_width = function() {
                    wwidth = $(window).width();
                    left = 0;
                    width = '100%';
                    if (wwidth >= 1280) {
                        left = (wwidth - 1280) / 2;
                        width = '1280px';
                    } else if (wwidth > 800 && wwidth < 1280) {
                        left = -$(window).scrollLeft();
                        width = '1024px';
                        if (wwidth > 1024) {
                            width = '100%';
                        }
                    } else if (wwidth <= 800) {
                        if ($('.main-wrapper').css('left') != 0 && $('.main-wrapper').css('left') != 'auto') {
                            left = $('.main-wrapper').css('left');
                        }
                    }
                };
                var sticky_navigation_offset_top_1 = $('.header.main-header').offset().top + 60;
                var sticky_navigation = function() {
                    var scroll_top = $(window).scrollTop();
                    if (scroll_top > sticky_navigation_offset_top_1) {
                        if( $('body').hasClass('ie8') ){
                            width = '1280px';
                            left = (wwidth - 1280) / 2;
                        }
                        if($(window).width() <= 800) {
                            $('.header.main-header').css({'top': '0px', 'left': left, 'z-index': 100, 'width': width, 'position': 'fixed'}).addClass('sticky').find('.hd-upper').slideUp();
                        } else {
                            $('.header.main-header').css({'top': '0px', 'left': left, 'z-index': 100, 'width': width}).addClass('sticky').find('.hd-upper').slideUp();
                        }

                        $('.main-container').addClass('sticky-header');
                    } else {
                        if ($(window).width() > 800) {
                            $('.header.main-header').css({'top': '0px', 'left': '0px', 'width': '100%'}).removeClass('sticky').find('.hd-upper');
                            $('.header.main-header').css({'top': '0px', 'left': left, 'z-index': 100, 'width': width, 'position': 'fixed'}).find('.hd-upper').stop(true, false).slideDown();
                        } else {
                            $('.header.main-header').css({'top': '0px', 'left': '0px', 'width': '100%', 'position': 'fixed'}).removeClass('sticky');
                            //$('.main-container').css({'padding-top': '80px'});
                        }
                        $('.main-container').removeClass('sticky-header');
                    }
                };

                window_width();
                $(window).resize(function() {
                    window_width();
                    sticky_navigation();
                });
                sticky_navigation();
                $(window).scroll(function() {
                    window_width();
                    sticky_navigation();
                });
            });
        }
    };
