// função para abrir o feedback
function abreFeed()
{
	if($("#pergunta").val()=='perg1'){$('#perg1').css('display','');$('#perg2').css('display','none');$('#perg3').css('display','none');}
	if($("#pergunta").val()=='perg2'){$('#perg2').css('display','');$('#perg3').css('display','none');$('#perg1').css('display','none');}
	if($("#pergunta").val()=='perg3'){$('#perg3').css('display','');$('#perg1').css('display','none');$('#perg2').css('display','none');}
    $('#menos').animate({'marginRight' : "+=274px" });    
	$("#mais").css('display','none');
}
//função fecha feedback
function fechaFeed()
{
	$('#imgBom').css('display','');
	$('#imgRegular').css('display','');
	$('#imgRuim').css('display','');
	
	$('#menos').animate({'marginRight' : "-=274px" }); 
	$("#mais").css('display','');

	$('#img2').css('display','none');
	$("#formDiv").css('display','none');
	$("#votoBom").css('display','none');
	$("#votoRegular").css('display','none');
	$("#votoRuim").css('display','none');
 
}
//função muda fundo apos vota e abre as boz de perguntas
function mudaImg(voto)
{
	$('#voto').val(voto);
	$('#img1').css('display','none');
	$('#img2').css('display','');
	
	$('#perg3').css('display','none');
	$('#perg1').css('display','none');
	$('#perg2').css('display','none');
	
	$('#imgBom').css('display','none');
	$('#imgRegular').css('display','none');
	$('#imgRuim').css('display','none');
	
	
	
	

	if(voto=='Bom')
	{
		$("#formDiv").css('display','block');
		$("#votoBom").css('display','block');
	}else if(voto=='Regular')
	{
		$("#formDiv").css('display','block');
		$("#votoRegular").css('display','block');
	}else
	{
		$("#formDiv").css('display','block');
		$("#votoRuim").css('display','block');
	}
}
// envia o formulário com os campos preenchidos
function enviaForm()
{

	var resp1 = $('#resp1').val();
	var resp2 = $('#resp2').val();
	var resp3 = $('#resp3').val();
	var voto = $('#voto').val();
	var mail = $('#mail').val();
	var pergunta = $('#pergunta').val();
	var mail =  $('#mail').val();
	
	if(resp1!=''){var resposta=resp1;}
	else if(resp2!=''){var resposta=resp2;}
	else if(resp3!=''){var resposta=resp3;}
	if(pergunta=='perg1'){pergunta = 'O que você achou do novo site da ABAC';}
	if(pergunta=='perg2'){pergunta = 'Você encontrou o que procurava';}
	if(pergunta=='perg3'){pergunta = 'O que você achou do conteúdo do nosso site';}
		
	location = '/votaSite&voto='+voto+'&resp='+resposta+'&pergunta='+pergunta+'&mail='+mail+'&pag=home';
}

function marcacao(id)
{
	$('#'+id).css('opacity','0.4');

function marcacaoOff(id)

	$('#'+id).css('opacity','1.0');

}
