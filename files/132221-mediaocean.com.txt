<!-- Set and Get UTM Cookies for Marketo Forms -->

<script type="text/javascript">
  jQuery(function ($) {

      function getParameterByName(name) {
          name = name.replace(/[\[]/, "\\[").replace(/[\]]/, "\\]");
          var regex = new RegExp("[\\?&]" + name + "=([^&#]*)"),
                  results = regex.exec(location.search);
          return results === null ? "" : decodeURIComponent(results[1].replace(/\+/g, " "));
      }

      var source = getParameterByName('utm_source');
      var medium = getParameterByName('utm_medium');
      var campaign = getParameterByName('utm_campaign');
      var term = getParameterByName('utm_term');

      if (source !== '' ) {
        $.cookie('utm_source', source, {expires: 200});
      }
      if (medium !== '') {
        $.cookie('utm_medium', medium, {expires: 200});
      }
      if (campaign !== '') {
        $.cookie('utm_campaign', campaign, {expires: 200});
      }
      if (term !== '') {
        $.cookie('utm_term', term, {expires: 200});
      }
      $(document).ready(function () {
          $('input[name=utmsource').val($.cookie('utm_source'));
          $('input[name=utmmedium').val($.cookie('utm_medium'));
          $('input[name=utmcampaign').val($.cookie('utm_campaign'));
          $('input[name=utmterm').val($.cookie('utm_term'));
      });

      MktoForms2.whenReady(function(form){
        form.onSubmit(function(){
          console.log('onSubmit fired');
          $.cookie('utm_source', source, {expires: 1});
          $.cookie('utm_medium', medium, {expires: 1});
          $.cookie('utm_campaign', campaign, {expires: 1});
          $.cookie('utm_term', term, {expires: 1});
        });
      });
  });
</script>

<!-- Ensure use of business email address in form submission -->

<script type="text/javascript">
(function (){
  // Include the email domains you would like to block in this list
  var invalidDomains = ["@gmail.","@yahoo.","@hotmail.","@live.","@aol.","@outlook."];

  MktoForms2.whenReady(function (form){
    form.onValidate(function(){
      var email = form.vals().Email;
      if(email){
        if(!isEmailGood(email)) {
          form.submitable(false);
          var emailElem = form.getFormElem().find("#Email");
          form.showErrorMessage("Must be a business email address.", emailElem);
          console.log("Form is not valid");
        }else{
          form.submitable(true);
        }
      }
    });
  });

  function isEmailGood(email) {
    for(var i=0; i < invalidDomains.length; i++) {
      var domain = invalidDomains[i];
      if (email.indexOf(domain) != -1) {
        return false;
      }
    }
    return true;
  }
})();
</script>
