        function position_top(gnbSize) {
            var windowHeight = document.body.clientHeight,
                cntsEl = document.getElementById("cnts"),
                cntsHeight = cntsEl.clientHeight,
                cntsMargin = (windowHeight - cntsHeight - gnbSize) / 2;
            cntsmarginMin = 40;
            if (cntsMargin >= cntsmarginMin) {
                cntsEl.style.margin = cntsMargin + "px auto";
            } else {
                cntsEl.style.margin = cntsmarginMin + "px auto";
            }
        }

        function func_onload() {
            var pos = 0;
            var strName = '';
            var strValue = '';

            var tmpErrorMessage = '';
            var tmpErrorMessageInfo = '';

            var strQueryString = document.location.href.substr(document.location.href.indexOf("?") + 1);
            var arrCollection = strQueryString.split("&");

            for (var idx = 0; idx < arrCollection.length; idx++) {
                pos = arrCollection[idx].indexOf("=");
                if (pos == -1) continue;
                strName = arrCollection[idx].substring(0, pos);
                strValue = arrCollection[idx].substring(pos + 1);

                if (strName == 'error')
                    break;
                else
                    strValue = '';
            }

            if (strValue == '5xx') {
                tmpErrorMessage = "지금은 서비스를 이용할 수 없습니다.";
                tmpErrorMessageInfo = "이용에 불편을 드려 죄송합니다.<br>잠시 후 다시 한 번 시도해주세요.";
            } else if (strValue == '4xx') {
                tmpErrorMessage = "페이지에 접근할 수 없습니다.";
                tmpErrorMessageInfo = "이용에 불편을 드려 죄송합니다.<br>잘못된 접근이거나 권한이 없는 페이지입니다.";
            } else if (strValue == '404') {
                tmpErrorMessage = "페이지를 찾을 수 없습니다.";
                tmpErrorMessageInfo = "제거되었거나 이름이 변경되었을 수 있습니다.<br>주소를 다시 한 번 확인해주세요.";
            } else if (strValue == '503') {
                tmpErrorMessage = "지금은 서비스를 이용할 수 없습니다.";
                tmpErrorMessageInfo = "사용자가 너무 많아 서비스 이용이 어렵습니다.<br>잠시 후 다시 한 번 시도해주세요.";
            } else if (strValue == '8xx') {
                tmpErrorMessage = "페이지가 일시적으로 차단되었습니다.";
                tmpErrorMessageInfo = "비정상적인 접근 시도로 인해 일시적으로 차단되었습니다.<br>잠시 후 다시 한번 시도해주세요.";
            } else {
                tmpErrorMessage = "사용할 수 없는 페이지입니다.";
                tmpErrorMessageInfo = "이용에 불편을 드려 죄송합니다.<br>주소를 다시 한 번 확인해주세요.";
            }

            document.getElementById('ErrorMessage').innerHTML = tmpErrorMessage;
            document.getElementById('ErrorMessageInfo').innerHTML = tmpErrorMessageInfo;
            position_top(document.getElementById("GNB_Wrapper").clientHeight);
        }

        });
