                              // flexslider's randomize is crappy so we use Fisher-Yates shuffle instead.
                              function shuffle(array) {
                                   var currentIndex = array.length
                                  , temporaryValue
                                  , randomIndex
                                  ;

                                  // While there remain elements to shuffle...
                                  while (0 !== currentIndex) {

                                      // Pick a remaining element...
                                      randomIndex = Math.floor(Math.random() * currentIndex);
                                      currentIndex -= 1;

                                      // And swap it with the current element.
                                      temporaryValue = array[currentIndex];
                                      array[currentIndex] = array[randomIndex];
                                      array[randomIndex] = temporaryValue;
                                 }

                                 return array;
                             }

                              function shuffle(array) {
                                   var currentIndex = array.length
                                  , temporaryValue

                                  ;

                                  // While there remain elements to shuffle...
                                  while (0 !== currentIndex) {

                                      randomIndex = Math.floor(Math.random() * currentIndex);

var rssJSONComponent = {
    parseDate: function (date) {
        var result = parseInt(date.substring(date.indexOf("(") + 1, date.indexOf(")")));
        return result;
    },
    formatDate: function (date) {
        year = "" + date.getFullYear();
        month = "" + (date.getMonth() + 1);
        if (month.length == 1) {
            month = "0" + month;
        }
        day = "" + date.getDate();
        if (day.length == 1) {
            day = "0" + day;
        }
        hour = "" + date.getHours();
        if (hour.length == 1) {
            hour = "0" + hour;
        }
        minute = "" + date.getMinutes();
        if (minute.length == 1) {
            minute = "0" + minute;
        }
        second = "" + date.getSeconds();
        if (second.length == 1) {
            second = "0" + second;
        }
        return year + "-" + month + "-" + day + " " + hour + ":" + minute + ":" + second;
    }
};

var twitterFirstPage = {
	avatar_size: 32,                        // [integer]  height and width of avatar if displayed (48px max)
	count: 10,                              // [integer]  how many tweets to display?
	list: currentLanguage,                  // [string]   optional name of list belonging to username
	page: 1,                                  // [integer]  which page of results to fetch (if count != fetch, you'll get unexpected results)
	retweets: true,                           // [boolean]  whether to fetch (official) retweets (not supported in all display modes)
	auto_join_text_default: "i said,",        // [string]   auto text for non verb: "i said" bullocks
	auto_join_text_ed: "i",                   // [string]   auto text for past tense: "i" surfed
	auto_join_text_ing: "i am",               // [string]   auto tense for present tense: "i was" surfing
	auto_join_text_reply: "i replied to",     // [string]   auto tense for replies: "i replied to" @someone "with"
	auto_join_text_url: "i was looking at",   // [string]   auto tense for urls: "i was looking at" http:...
	twitter_url: "twitter.com",               // [string]   custom twitter url, if any (apigee, etc.)
	username: "NWT6",
	twitter_api_url: "/webproxy/TwitterFeedProxy.aspx",
	twitter_search_url: "http://www.twitter.com", // [string]   custom twitter search url, if any (apigee, etc.)
	template: "{avatar}{user}{time}{join}{text}",   // [string or function] template used to construct each tweet <li> - see code for available vars
	urlRegexp: /\b((?:[a-z][\w-]+:(?:\/{1,3}|[a-z0-9%])|www\d{0,3}[.]|[a-z0-9.\-]+[.][a-z]{2,4}\/)(?:[^\s()<>]+|\(([^\s()<>]+|(\([^\s()<>]+\)))*\))+(?:\(([^\s()<>]+|(\([^\s()<>]+\)))*\)|[^\s`!()\[\]{};:'".,<>???????]))/gi,
	comparator: function(tweet1, tweet2) {    // [function] comparator used to sort tweets (see Array.sort)
		return tweet2["tweet_time"] - tweet1["tweet_time"];
	},
	days: function (dateString) {
		return Date.parse(dateString);
	},
	getTweets: function () {
		$.ajax({
			url: webCore.getWebAPIURL("prod") + "TwitterFeed/TwitterList",
			type: "GET",
			data: {
				app: "nordic",
				list: twitterFirstPage.list,
				count: twitterFirstPage.count
			},
			cache: true,
			dataType: "jsonp",
			jsonp : "callback",
			jsonpCallback: "handleTwitterResponse"
		});
	},
	    // Convert twitter API objects into data available for
    // constructing each tweet <li> using a template
	extractTemplateData: function (item) {
		var o = {};
		o.item = item;
		o.source = item.source;
		o.screen_name = item.user.identifier.screenName;
		o.name = item.user.name;
		o.avatar_size = twitterFirstPage.avatar_size;
		o.avatar_url = item.user.profileImageUrl;
		o.retweet = typeof(item.retweeted) !== 'undefined' && item.retweeted;
		o.tweet_time = twitterFirstPage.days(dateFromISO(item.createdAt));
		o.join_text = twitterFirstPage.join_text == "auto" ? twitterFirstPage.buildAutoJoinText(item.text) : twitterFirstPage.join_text;
		o.tweet_id = item.statusId;
		o.twitter_base = "http://"+twitterFirstPage.twitter_url+"/";
		o.user_url = o.twitter_base + o.screen_name;
		o.tweet_url = o.user_url + "/status/"+o.tweet_id;
		o.reply_url = o.twitter_base + "intent/tweet?in_reply_to=" + o.tweet_id;
		o.retweet_url = o.twitter_base + "intent/retweet?tweet_id=" + o.tweet_id;
		o.favorite_url = o.twitter_base + "intent/favorite?tweet_id=" + o.tweet_id;
		o.retweeted_screen_name = o.retweet && item.inReplyToScreenName;
		o.tweet_relative_time = twitterFirstPage.relativeTime(o.tweet_time);
		o.entities = item.entities ? (item.entities.urlEntities).concat(item.entities.media || []) : [];
		o.tweet_raw_text = o.retweet ? ('RT @' + o.retweeted_screen_name + " " + item.retweetedStatus.text) : item.text; // avoid '...' in long retweets
		o.tweet_text = $([twitterFirstPage.linkURLs(o.tweet_raw_text, o.entities)]).linkUser().linkHash()[0];
		o.tweet_text_fancy = $([o.tweet_text]).makeHeart().capAwesome().capEpic()[0];
	
		// Default spans, and pre-formatted blocks for common layouts
		o.user = twitterFirstPage.t('<div class="tweet_content"><span class="tweet_name">{name}</span> <a class="tweet_user" href="{user_url}" target="_blank">@{screen_name}</a>', o);
		o.join = twitterFirstPage.join_text ? twitterFirstPage.t(' <span class="tweet_join">{join_text}</span> ', o) : ' ';
		o.avatar = o.avatar_size ? twitterFirstPage.t('<div class="tweet_avatar"><a href="{user_url}" target="_blank"><img src="{avatar_url}" height="{avatar_size}" width="{avatar_size}" alt="{screen_name}\'s avatar" title="{screen_name}\'s avatar" border="0"/></a></div>', o) : '';
		o.time = twitterFirstPage.t('<span class="tweet_time">{tweet_relative_time}</span>', o);
		o.text = twitterFirstPage.t('<p><span class="tweet_text">{tweet_text_fancy}</span></p></div>', o);
		o.reply_action = twitterFirstPage.t('<a class="tweet_action tweet_reply" href="{reply_url}">reply</a>', o);
		o.retweet_action = twitterFirstPage.t('<a class="tweet_action tweet_retweet" href="{retweet_url}">retweet</a>', o);
		o.favorite_action = twitterFirstPage.t('<a class="tweet_action tweet_favorite" href="{favorite_url}">favorite</a>', o);
		return o;
	},
	buildAutoJoinText: function (text) {
		if (text.match(/^(@([A-Za-z0-9-_]+)) .*/i)) {
			return twitterFirstPage.auto_join_text_reply;
		} else if (text.match(twitterFirstPage.urlRegexp)) {
			return twitterFirstPage.auto_join_text_url;
		} else if (text.match(/^((\w+ed)|just) .*/im)) {
			return twitterFirstPage.auto_join_text_ed;
		} else if (text.match(/^(\w*ing) .*/i)) {
			return twitterFirstPage.auto_join_text_ing;
		} else {
			return twitterFirstPage.auto_join_text_default;
		}
	},
	relativeTime: function (date) {
		var relative_to = (arguments.length > 1) ? arguments[1] : new Date();
		var delta = parseInt((relative_to.getTime() - date) / 1000, 10);
		//console.log(relative_to + ' ' + dateFromISO(date));
		//console.log(relative_to.getTime() - parseInt(date));
		
		var r = "";
		if (delta < 1) {
			r = "now";
		} else if (delta < 60) {
			r = delta + " sec";
		} else if(delta < (45*60)) {
			r = (parseInt(delta / 60, 10)).toString() + " min";
		} else if(delta < (24*60*60)) {
			r = (parseInt(delta / 3600, 10)).toString() + " h";
		} else {
			r = (parseInt(delta / 86400, 10)).toString();
			if(r > 1) {
				r += " days";
			} else {
				r += " day";
			}
		}
		return r;
	},
	linkURLs: function (text, entities) {
		return text.replace(twitterFirstPage.urlRegexp, function(match) {
			var url = (/^[a-z]+:/i).test(match) ? match : "http://" + match;
			var text = match;
			for(var i = 0; i < entities.length; ++i) {
				var entity = entities[i];
				if (entity.url == url && entity.expanded_url) {
					url = entity.expanded_url;
					text = entity.display_url;
					break;
				}
			}
			return "<a href=\""+twitterFirstPage.escapeHTML(url)+"\" target=\"_blank\">"+twitterFirstPage.escapeHTML(text)+"</a>";
		});
	},
	escapeHTML: function (s) {
		return s.replace(/</g,"&lt;").replace(/>/g,"^&gt;");
	},
	replacer: function (regex, replacement) {
		return function() {
			var returning = [];
			this.each(function() {
				returning.push(this.replace(regex, replacement));
			});
			return $(returning);
		};
	},
	t: function (template, info) {
		if (typeof template === "string") {
			var result = template;
			for(var key in info) {
				var val = info[key];
				result = result.replace(new RegExp('{'+key+'}','g'), val === null ? '' : val);
			}
			return result;
		} else {
			return template(info);
		}
	}
};

$.fn.extend({

	linkHash: twitterFirstPage.replacer(/(?:^| )[\#]+([\w\u00c0-\u00d6\u00d8-\u00f6\u00f8-\u00ff\u0600-\u06ff]+)/gi,' <a href="'+twitterFirstPage.twitter_search_url+'/search?q=%23$1&lang=all'+((twitterFirstPage.username && twitterFirstPage.username.length == 1 && !twitterFirstPage.list) ? '&from='+twitterFirstPage.username.join("%2BOR%2B") : '')+'" target=\"_blank\">#$1</a>'),

function dateFromISO(s) {
  s = s.split(/\D/);
  return new Date(Date.UTC(s[0], --s[1], s[2], s[3], s[4], s[5]));
}

function handleTwitterResponse(data) {
		if(data.success) {
			var result = data.result;
			
			var twitterContainer = $("#tweets");
	 		var list = $("<ul></ul>").addClass("tweet_list");
	        
	        twitterContainer.empty().append(list);
	        
	        list.empty();
	
	        var tweets = $.map(result, twitterFirstPage.extractTemplateData);
	        
	        tweets = tweets.sort(twitterFirstPage.comparator).slice(0, twitterFirstPage.count);
	        list.append($.map(tweets, function(o) { return "<li>" + twitterFirstPage.t(twitterFirstPage.template, o) + "</li>"; }).join('')).
	            children("li:first").addClass("tweet_first").end().
	            children("li:odd").addClass("tweet_even").end().
	            children("li:even").addClass("tweet_odd");
	
	        twitterContainer.trigger("loaded").trigger((tweets.length === 0 ? "empty" : "full"));
	        window.setTimeout(twitterFirstPage.getTweets, 120000);
		} else {
			$("#tweets").append("<p></p>").text(data.message);
		}
}
