    function DispatchMultiselectClass(multiselectData, $mainSelect, $secondSelect, $urlButton, $urlInsButton) {
        var that = this;
        var data = multiselectData;
        var mainSelect = $mainSelect;
        var secondSelect = $secondSelect;
        var urlButton = $urlButton;
        var urlInsButton = $urlInsButton;
        var countrySbSelector = '';


        function init(main, second) {
            if (main) {
                mainSelect.selectbox({
                    onChange: mainChangeHendler,
                    onOpen: function () {
                        var mas = $$('.select_wrapper_country li a');
                        for(var i=0; i<mas.length; i++){
                            if(dispatchMultiselectData[mas[i].readAttribute('rel')].shipping != '1'){
                                mas[i].style.background = 'none';
                            }
                        }
                    }
                });
            }
            if (second) {
                secondSelect.selectbox('detach').selectbox({
                    onChange: secondChangeHendler
                });
            }
            countrySbSelector = jQuery('.select_wrapper_country .sbSelector');
            that.checkButtons();
        }

        this.checkButtons = function () {
            var mainSelectedValue = mainSelect.val();
            var secondSelectedValue = secondSelect.val();
            //console.log(mainSelectedValue, secondSelectedValue);

            var $data = data[mainSelectedValue]['languages'][secondSelectedValue];

            if($data != undefined) {
                if (!$data['url_inst'] || $data['url_inst'] == '') {
                    urlInsButton.hide();
                }
                else{
                    urlInsButton.show();
                }

                if (!$data['url'] || $data['url'] == '') {
                    urlButton.hide();
                } else {
                    urlButton.show();
                }
            }
        }

        init(true, true);

        function mainChangeHendler(value, inst) {
            DispatchMultiselect.rebuildSecond();
        }

        function secondChangeHendler(value, inst) {
            DispatchMultiselect.checkButtons();
            DispatchMultiselect.renameButtons();
        }

        this.renameButtons = function () {
            var item = data[mainSelect.val()]['languages'][secondSelect.val()];
            urlButton.text(item.lable_shop);
            urlInsButton.text(item.lable_website);
        }
        this.rebuildSecond = function () {
            var items = data[mainSelect.val()]['languages'];

            if(data[mainSelect.val()].shipping == '1'){
                countrySbSelector.removeAttr('style');
            } else {
                countrySbSelector.css('background', 'none');
            }

            var result = '';
            for (var key in items) {
                result += '<option value="' + key + '">' + items[key]['lang_label'] + '</option>';
            }
            urlButton.text(items[Object.keys(items)[0]].lable_shop);
            urlInsButton.text(items[Object.keys(items)[0]].lable_website);
            secondSelect.html(result);
            init(false, true);
        }

        this.urlInsButtonClick = function ($this, $event) {
            var url = data[mainSelect.val()]['languages'][secondSelect.val()]['url_inst'];
            if (url) {
                window.location.href = url;
            }
            $event.preventDefault();
        }

        this.urlButtonClick = function ($this, $event) {
            var refererUrl = jQuery('#dispatch_wrapper').attr('data-referer-url');
            var productId = jQuery('#dispatch_wrapper').attr('data-referer-product') ? jQuery('#dispatch_wrapper').attr('data-referer-product') : '';
            var chacedUrl = data[mainSelect.val()]['languages'][secondSelect.val()]['url'];
            var url = '';

            jQuery.ajax({
                async: false,
                type: "POST",
                cache:false,
                data: {'referer_url':refererUrl,'chaced_url':chacedUrl, 'product_id':productId},
                url: window.location.origin + '/dispatch/index/checkPageExist',
                dataType: "json",
                success: function (answer) {
                    if (answer.status){
                        url = answer.url;
                    } else {
                        url = chacedUrl;
                    }

                    if (url) {
                        var form = jQuery('<form action="'+url+'" method="POST">' +
                            '<input type="hidden" name="geo_redirect" value="true">' +
                            '</form>').submit();
                        jQuery('body').append(form);
                        jQuery(form).submit();
                    }
                    Mage.Cookies.clear('dispatch_product_id');
                    $event.preventDefault();
                }
            });
        }
    }

    jQuery(document).ready(function () {
        var mainSelect = jQuery("#countrySelect");
        var secondSelect = jQuery("#languageSelect");
        var urlButton = jQuery("#urlButton");
        var urlInsButton = jQuery("#urlInsButton");

        DispatchMultiselect = new DispatchMultiselectClass(
            dispatchMultiselectData, mainSelect, secondSelect, urlButton, urlInsButton
        );
        DispatchMultiselect.rebuildSecond();

    });

        function coverResize(){
            var covers = $('.pre_home_bg');

            var offset  = ($(obj).width()-$(document).width())/2;
            if(offset > 0){
                offset *= -1;
            } else {
                offset = 0;
            }
            $(obj).css('left',offset);
        }

        $(window).resize(function(){

            coverResize();
        });
