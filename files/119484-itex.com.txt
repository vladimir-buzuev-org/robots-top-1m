		function showPosition(position) {
			var xhttp = new XMLHttpRequest();
			xhttp.open("GET", "/Fb/itexbrkdir.aspx?lat=" + position.coords.latitude + "&lng=" + position.coords.longitude);
			xhttp.onloadend = handler;
			xhttp.send();
		}
