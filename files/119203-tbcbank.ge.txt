        } else {
            clearTimer();
            carouselControl.hide();

    function selectPhoto(index, currentIndex) {
        var animatedHeaderItems = $('.animatedHeaderItem'),
            animatedHeaderItem = animatedHeaderItems.eq(index),
            currentAnimatedHeaderItem = animatedHeaderItems.eq(currentIndex);

        if (index == 0 && currentIndex == 0) {
            animatedHeaderItem.css({left: '0%', display: 'block'});
        } else if (currentIndex == 0 && index == animatedHeaderItems.length - 1) {
            animatedHeaderItem.css({left: '100%', display: 'block'});
            currentAnimatedHeaderItem.animate({left: '-100%'}, 500);
        } else if (index == 0 && currentIndex == animatedHeaderItems.length - 1) {
            animatedHeaderItem.css({left: '-100%', display: 'block'});
            currentAnimatedHeaderItem.animate({left: '100%'}, 500);
        } else if (index > currentIndex) {
            animatedHeaderItem.css({left: '-100%', display: 'block'});
            currentAnimatedHeaderItem.animate({left: '100%'}, 500);
        } else if (index < currentIndex) {
            animatedHeaderItem.css({left: '100%', display: 'block'});
            currentAnimatedHeaderItem.animate({left: '-100%'}, 500);
        }

        animatedHeaderItem.animate({left: '0%'}, 500);

            .eq(index).addClass('active');
        content.find('.find-more').removeClass('active')
            .eq(index).addClass('active');

            index = currentIndex == 0 ? photos.length - 1 : currentIndex - 1;

        var currentIndex = getCurrentIndex();
