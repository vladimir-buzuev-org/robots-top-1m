			mobileSearchContainerToggle: function(event) {
				event.preventDefault();

				this.dom.mobileTypeaheadInput.val('');
				this.dom.mobileSearchContainer.toggleClass('expanded');
				this.dom.mobileSearchContainer.toggleClass('hide');
				this.dom.mobileSearchIcon.toggleClass('active');
				if(this.currentCall != null) {
					this.currentCall.abort();
				}
				if(this.dom.mobileSearchContainer.hasClass('expanded')) {
					this.dom.mobileTypeaheadInput.focus();
				}
				return false;
			},
			globalNavToggle: function() {
				this.dom.globalNav.toggleClass('expanded');
				var toggleText = this.dom.globalToggle.find('.toggle-text');

				if(this.dom.globalNav.hasClass('expanded')) {
					toggleText.text(toggleText.attr('data-opened-text'));
					this.dom.globalToggle.addClass('expanded');
					this.dom.globalNav.css({
						height: this.dom.globalNav.attr('data-expanded-height') + 'px'
					});
				}

				else {
					toggleText.text(toggleText.attr('data-closed-text'));
					this.dom.globalToggle.removeClass('expanded');
					this.dom.globalNav.css({
						height: '0px'
					});
				}
			},
			mobileNavViewSwitch: function() {
				if(this.dom.mobileNavTop.hasClass('active')) {
					this.dom.mobileNavTop.removeClass('active');
					this.dom.mobileNavBottom.addClass('active');
					this.dom.mobileNavToggle.addClass('expanded').find('.text').text(this.dom.mobileNavToggle.attr('data-expanded-text'));
				}

				else {
					this.dom.mobileNavTop.addClass('active');
					this.dom.mobileNavBottom.removeClass('active');
					this.dom.mobileNavToggle.removeClass('expanded').find('.text').text(this.dom.mobileNavToggle.attr('data-collapsed-text'));
				}
			},
			onMobileNavSiteToggleClick: function(event) {
				event.preventDefault();
				var target = $(event.target).closest('.nav-site-toggle');
				var parent = target.parent();
				parent.siblings().removeClass('active');
				parent.toggleClass('active');
				return false;
			},
			makeQuery: function(event) {

				// No special keys
				var ignoreCodes = [37, 38, 39, 16, 17, 18, 91]
				if(_.indexOf(ignoreCodes, event.keyCode) > -1) {
					return false;
				}

				if(this.currentCall != null) {
					this.currentCall.abort();
				}

				var searchTerm = this.dom.typeaheadInput.val();
				if(searchTerm.length > 2) {
					this.currentCall = $.ajax({
						url: this.api,
						type: 'get',
						data: {
							searchTerm: searchTerm,
							limit: this.dataLimit
						},
						complete: $.proxy(this.callComplete, this),
						success: $.proxy(this.handleResults, this)
					});
				} else {
					this.dom.searchResultsContainer.html('');
				}
			},
			callComplete: function() {
				this.currentCall = null;
			},
			handleResults: function(data) {
				console.log(this.label(), data.results);
				this.dom.searchResultsContainer.html('');
				for(var i in data.results) {
					var result = data.results[i];
					if ((result.contentType == 'article') || (result.contentType == 'film')) {
						this.dom.searchResultsContainer.append(this.templates.article(result));
					}
					console.log(result);
				}
//				this.dom.resultsContainer.html(this.template(data)).addClass('expanded');
			},
			onMessage: function(event, data) {
				console.log(this.label() + ' onMessage', data);
				if(_.isString(data.data)) {
					if(data.data == 'login:embedded-login-click') {
						Snag.helpers.login.modalLogin();
					} else if(data.data == 'login:announce-self') {
						console.log(this.label() + ' Found iframe');
						this.dom.iframeAuthWidget = data.source;

						// Out-of-sync login situation
						if(data['loggedIn'] === true && Snag.constants.user.loggedIn === false) {
							this.dom.iframeAuthWidget.postMessage('viewlift|login:sync', '*');
							this.div.addClass('logged-in');
							Snag.constants.user.loggedIn = true;
						} else {
							// Pull missing user properties from the masthead object to the Snag object
							$.extend(Snag.constants.user, this.setup.data.user);
						}
					}else if(data.data == 'login:success' || data.data == 'login:sync:success') {
						this.loginSuccess(data);
						this.broadcast('login:success');
					}else if(data.data == 'site-skin:parameters') {
						this.initSkinAd(data);
					}else if(data.data == 'site-skin:markup') {
						this.addSkinAdMarkup(data);
					}else if(data.data == 'Dfp:PrestialAd:Parameters') {
						this.initPrestitialAd(data);
					} else if(data.data == 'points:added') {
						this.dom.iframeAuthWidget.postMessage('viewlift|points:added', "*");
					}
				}
			},
			initPrestitialAd: function(data) {
				console.log(this.label() + 'initPrestitialAd', data);
				if(data['image']){
					require(['advertising/prestitial-ad'], function(prestitialAdModule){
							$('.viewlift-prestitial-ad').show();
							this.prestitialAd = new prestitialAdModule();
							this.prestitialAd.div = this.div;
							this.prestitialAd.setup = this.setup;
							this.prestitialAd.viewURLPrefix = data['viewURLPrefix'] || '';
							this.prestitialAd.clickThroughUrl = data['clickThroughUrl'] || '';
							this.prestitialAd.image = data['image'] || '';
							this.prestitialAd.url = data['url'] || '';
							this.prestitialAd.timeDuration = data['duration']||10;
							this.prestitialAd.overlayBackgroundColor = data['overlayBackgroundColor'] || '';
							this.prestitialAd.postProcess();
					}, function(error){
						console.log('Failed to load prestitial ad module!');
						$('.viewlift-prestitial-ad').hide();
					});
				}else{
					console.log('ad image for prestitial-ad missing from data(DFP)');
					$('.viewlift-prestitial-ad').hide();
				}
			},
			loginSuccess: function(data) {
				console.log(this.label() + 'loginSuccess()', data);
				try {
					this.dom.iframeAuthWidget.postMessage('viewlift|login:success', '*');
				} catch (e) {
					console.error(e);
				}
				if($('.modal-dialog').hasClass('start') == false){
					Snag.RemoveModal();
				}
				this.div.addClass('logged-in');

				if(_.isObject(data['user'])) {
					if(data.user.subscribed == "true"){
						var _url = $.cookie('rsvp_return_url') || null;
						if(_url !== null){
							$.removeCookie('rsvp_return_url');
							this.newTab(_url);
						}
					}
					Snag.constants.user = data.user;
					Snag.constants.user.loggedIn = true;
				}

				// Reload iframeAuthWidgets
				$('.js-masthead-auth').each(function (i, iframe) {
					iframe.contentWindow.location.reload();
				});

				// Sync login if:
				// 		1. we know the token cookie is present
				// 		2. we know this is a sub-site within the network
				if(this.setup.config['networkDomain'] && _.isString(data['token'])) {
					Snag.SetLoading();
					$.ajax({
						url: '/apis/svod/token/' + data.token,
						type: 'get',
						data: {
							site: Snag.constants.site.internalName,
							rememberMe: true
						},
						complete: function() {
							Snag.UnsetLoading();
						},
						success: $.proxy(function(data) {
							console.log('Logged In', data);
							this.setShopUrls(data);
						},this),
						error: function() {
							console.warn('Not Logged In', arguments);
						}
					});
				}

				this.setShopUrls(data);
			},

			onRegistration: function() {
				if (Snag.constants.user.newRegistration) {
					// set some flag that says that you are new registree
					localStorage.setItem('needsPersonalization', true);
				}
			},

			initialPersonalization: function () {
				localStorage.removeItem('needsPersonalization');
				sessionStorage.setItem('initPersonalization', true);
				$.when(Snag.Confirm('Personalize Your Experience', 'Would you like to add your favorite players, teams, and tags to your profile? It will allow you to get specific content and instant updates about the things you are most interested in.'))
				.then(function() {
					window.location = 'https://www.monumentalsportsnetwork.com/user/personalization';
				});
			},

			initSkinAd: function(data) {
				console.log(this.label() + 'initSkinAd', data);
				require(['advertising/site-skin'], function(siteSkinModule){
					this.siteSkinAd = new siteSkinModule();
					this.siteSkinAd.div = this.div;
					this.siteSkinAd.setup = this.setup;
					this.siteSkinAd.viewURLPrefix = data['viewURLPrefix'] || '';
					this.siteSkinAd.image = data['image'] || '';
					this.siteSkinAd.url = data['url'] || '';

					this.siteSkinAd.postProcess();
				}, function(error){
					console.log('Failed to load skin ad module!');
				});
			},
			addSkinAdMarkup: function(data) {
				if(_.isString(data['markup'])) {
					$('#site-skin').append($(data.markup));
				}
			},

			// Reflect when asked
			getPoints: function() {
				var points = this.setup.data['pointActivities'] || [];
				this.broadcast('rewards:setPoints', { points: points });
			},

			// Pass UUID to Monumental Shop url
			setShopUrls: function (data) {
				var uuid;
				var user;

				if (checkNested(data, 'user', 'id')) {
					user = data.user;
					uuid = data.user.id;
				} else if (checkNested(this.setup.data, 'user', 'id')) {
					user = this.setup.data.user;
					uuid = this.setup.data.user.id;
				} else {
					return false;
				}

				if (!user.validSubscription) {
					return false;
				}

				$.ajax({
					url: '/apis/user/svod/encryptUUID',
					type: 'get',
					data: {
						uuID: uuid
					},
					success: $.proxy(function(data) {
						// Select all anchor tags with href containing the string "shop.monumental"
						var $links = $('a[href*="shop.monumental"]');
						var token = data.data;

						$links.each(function (i, link) {
							// Get Old URL
							var oldURL = $(link).attr('href');

							// Modify URL
							var newURL = URI(oldURL)
								.domain("monumentalsportsnetwork.com") // make sure domain is monumentalsportsnetwork.com not monumentalsports.com
								.addSearch("ViewLift_Token", token); // add ViewLift_Token

							// Set Url
							$(link).attr('href', newURL);
						})
					}, this),
				});
			}
		});
	}
	return tmpModule;
});
define('noop', [], function() {
	function tmpModule(defaults) {
		Module.call(this, {
			mod: 'noop',
			postProcess: function() {
				console.log(this.label(), 'Post Process', this);
			}
		});
	}
	return tmpModule;
});
