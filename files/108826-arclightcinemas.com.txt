        function setLocation(location) {
            setTimeout(function () {
                var url = '/httphandlers/locationnavigationhandler.ashx';
                var data = {
                    option: 'desktop',
                    locationId: location,
                    contextId: '{EC74D733-D0A7-47EB-8F1D-281CB504D72F}'
                }
                var changeCallback = function (data) {
                    AL.Util.ajax.unmask();
                    if (data.Status == 'OK') {
                        if (data.ReturnData)
                            window.location = data.ReturnData;
                        else
                            window.location = window.location;
                    }
                };

                var changeCallbackError = function (data) {
                    AL.Util.ajax.unmask();
                    AL.Util.dialog.alert('msgChangeLocation', "There was an error changing your location.", 320, 200);
                }

                AL.Util.ajax.mask();
                AL.Util.createAjaxRequest(url, changeCallback, changeCallbackError, 'POST', data, 'JSON');

                AL.Util.cookie.setCookie("LocationSet", "1");
                //AL.Util.cookie.setCookie("AL_DefaultLocation", $('select.ddlLocation option:selected').val());

            }, 1000);

        function displayLocationSelector() {
            if (locationSet != null) {
                $('.main-wrapper').removeClass('blur');
                $('.location-selector-overlay').hide();
            } else {
                $('select.ddlLocation').selectmenu("close");
                $('.main-wrapper').addClass('blur');
                $('<div class="location-selector-overlay"></div>').insertAfter($('.main-wrapper'));
                $('.location-selector-overlay').show();
                $('.main-wrapper.blur, .location-selector-overlay').on('click', function (e) {
                    e.preventDefault();
                });

                    $('select.ddlLocation').selectmenu("open");

    function showSignUpInformation() {
