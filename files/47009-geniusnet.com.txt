<script type="text/javascript">
  var dom = document.getElementById('carousel_panle');
  var startTouchY = 0;
  var endTouchY = 0;
  function addEvent(elem, type, showAll) {
    type = type.split(' ');
    type.forEach(function (item) {
      if (item == "touchstart") {
        startTouchY = 0;
        endTouchY = 0;
        elem.addEventListener(item, function (ev) {
          startTouchY = ev.touches[0].clientX;
        });
      } else if (item == "touchend") {
        elem.addEventListener(item, function (ev) {
          if (endTouchY - startTouchY > 70) {//向右滑動
            $('#carousel-marketingHomeBanner').carousel('prev')
          } else if (endTouchY - startTouchY < -70) {//向z左滑動
            $('#carousel-marketingHomeBanner').carousel('next')
          }
        });
      } else if (item == "touchmove") {
        elem.addEventListener(item, function (ev) {
          if (ev.touches.length != 0) {
            endTouchY = ev.touches[0].clientX;
          }
        });
      }
    });
  }
  addEvent(dom, 'touchstart touchmove touchend', false);
</script>
