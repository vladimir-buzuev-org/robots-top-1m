<script type="text/javascript">

function checkLocationForm( $form )
{
    var returnVal = false;

    if ( $form.hasClass( "depotForm" ) )
    {
        if ( $form.find( 'input[type="text"]' ).val() != "" )
        {
            if ( $form.find( 'select[name="country"]' ).length > 0 )
            {
                var country = $form.find( 'select[name="country"]' ).val();
            }
            else
            {
                var country = $form.find( 'input[name="country"]' ).val();
            }

            var zip = "";
            var depotno = "";
            if ( $form.find( "input[name='depotNo']" ).length > 0 )
            {
                depotno = $form.find( "input[name='depotNo']" ).val();
            }
            else if ( $form.find( "input[name='depot']" ).length > 0 )
            {
                zip = $form.find( "input[name='depot']" ).val();
            }
            var url = '/nl/depotfinder/depotfinder/(scriptmode)/1';
            var postcode = "PostalCode=" + zip + "&DepotNr=" + depotno + "&Country=" + country + "&Landingpage=0&locationfinder=1";

            $.ajax( {
                type: "POST",
                url: url,
                data: postcode,
                async: false,
                success: function( msg ){
                    if ( msg === "error" )
                    {
                        $( ".localPartner").find( ".errorBox_depot" ).addClass( "active" );
                    }
                    else
                    {
                        returnVal = true;
                    }
                }
            });
        }
        else
        {
            $( ".localPartner").find( ".errorBox_depot" ).addClass( "active" );
        }
    }
    else if ( $form.hasClass( "pickupForm" ) )
    {
        if ( $form.find( 'input[type="text"]' ).val() != "" )
        {
            returnVal = true;
        }
        else
        {
            $( ".localPartner").find( ".errorBox_pickup" ).addClass( "active" );
        }
    }

    return returnVal;
}


jQuery(function($)
{
    $( ".localPartner" ).find( "form" ).bind( 'submit', function(e) {
        var $form = $( this );

        if ( !checkLocationForm( $form ) )
        {
            e.preventDefault();
        }
        var label = $form.data( "label" );
        trackingEvent( "input_field", "submit", "b2b:find_contact_partner:" + label, "", "" );
    });

    $( ".localPartner").find( 'input[type="text"]').bind( "focus", function(e) {
        $( ".localPartner").find( 'div[class^="errorBox_"]' ).removeClass( "active" );
    });

});
</script>
