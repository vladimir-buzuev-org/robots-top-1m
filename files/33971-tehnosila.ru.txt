<script language="JavaScript">
    function ShowHidePassword() {
        if($('#USER_PINCODE').val() <= 0)
        {
            $('#USER_PINCODE').hide();
            $('#USER_PINCODE_TEXT').show();
        }
        else
        {
            $('#USER_PINCODE').show();
            $('#USER_PINCODE_TEXT').hide();
        }
    }
    $(document).ready(function () {

        // IE 9 placeholders fix
        $('input[name="USER_PASSWORD"]').focus(function(event) {
          $(this).parents('.q-checkoutBlock').find('.mph + .errorBox').remove();
          $(this).parents('.q-checkoutBlock').find('.mph').removeClass('invalid');
        });
        $('input[name="USER_PASSWORD"][type="text"]').on('keyup', function(event) {
          if ($(this).val() === '') {
            $(this).siblings('.regInputMod.mph').hide();
          }
        });

        if (!$.browser.msie) {
            $('#USER_PINCODE').attr('placeholder', 'PIN');
        }
        else
        {
            ShowHidePassword();
        }
        $('#USER_PINCODE_TEXT').focus(function(){
            if ($.browser.msie && $('#USER_PINCODE').val() <= 0)
            {
                $('#USER_PINCODE_TEXT').hide();
                $('#USER_PINCODE').show().focus();
            }
        });
        $('#USER_PINCODE').blur(function(){
                if ($.browser.msie && $('#USER_PINCODE').val() <= 0)
                {
                    ShowHidePassword();
                }
            });

                        // То что НЕ вызывается на странице спасибо.
        processNewCardPopup();
        bindResetRegistrationForm();
        
        $('.regInputSubmit').click(function (event) {
            //$(document).trigger('form.submit.click', {eventContext: 'popup', eventForm: 'popupRegForm'});
            if (!$(this).hasClass('regInputSubmitDisabled')) {
                $('.regError').each(function () {
                    $(this).removeClass('enabled');
                });
                $('.regMainInfo, .regMainError').html('');
                var data = {
                    'email': $('#NEW_EMAIL').val(),
                    'name': $('#NEW_NAME').val(),
                    'lastName': $('#NEW_LAST_NAME').val(),
                    'password': $('.regPasswordWrap input:visible[name="USER_PASSWORD"]').val(),
                    'hasBonusCard': $('#bonus_card').is(':checked') ? 1 : 0,
                    'cardNumber': $('#USER_CARD').val(),
                    'cardPin': $('#USER_PINCODE').val(),
                    'issueCard': ($('[name=NEW_CARD]').is(':checked')) ? 1 : 0,
                    'organization': ($('#user_payment_type_popup').is(':checked')) ? 1 : 0,
                    'orderId': $('[name=ORDER_ID]').val(),
                    'token': $('[name=TOKEN]').val(),
                    'fromPage': $('[name=FROM_PAGE]').val(),
                    'SOURCE_REG': $('input[name="SOURCE_REG"]').val(),
                    'do_register':'Y'
                };
                $('.regInputSubmit').addClass('regInputSubmitDisabled');
                var urlAjax = 'https://' + window.location.host + '/_ajax/loyalty/registration.php';
                var crossDomain = true;
                if ($.browser.msie && $.browser.version < 10) {
                    urlAjax = '/_ajax/loyalty/registration.php';
                    crossDomain = false;
                }
                jQuery.ajax({
                    type: 'POST',
                    url: urlAjax,
                    crossDomain: crossDomain,
                    xhrFields: {
                        withCredentials: crossDomain
                    },
                    data: data,
                    success: function (result, textStatus, jqXHR) {
                        if (result.success == 1) {
                            $(document).trigger('auth.registration.success', {emailHash: md5($('#NEW_NAME').val() + md5('eld0r@d0123'))});
                            $(document).trigger('form.response.success.time', {eventContext: 'popup', eventForm: 'popupRegForm'});
                            $(document).trigger('form.submit.success_time', {eventContext: 'popup', eventForm: 'popupRegForm'});
                            $(document).trigger('form.submit.success', {eventContext: 'popup', eventForm: 'popupRegForm'});
                            var popupRegFinal = ($('.regInputSubmit').attr('rel') == 'getCard') ? $('.popupRegFinalGetCard') : $('.popupRegFinalCustom');

                            $('.popupReg').closePopup(100, function () {

                                var additionalTextContainer = popupRegFinal.find('.regFinMainB'),
                                    removeAdditionalTextContainer = true,
                                    messageToDisplay = '.regFinMessage_noCardIssued';
                                var isAssign = false;

                                if (result.notification) {
                                    removeAdditionalTextContainer = false;
                                    additionalTextContainer.html(result.notification);
                                }
                                else if(result.data.cardNumber)
                                {
                                    if(result.data.type == 1)// Регистрация с получением карты и карта создана
                                    {
                                        messageToDisplay = '.regFinMessage_cardIssued';
                                    }
                                    else if(result.data.type == 2)// Регистрация с привязкой карты
                                    {
                                        messageToDisplay = 'Бонусы за совершенный заказ будут начислены на привязанную карту № ' + formatCard(result.data.cardNumber) + '. Для входа на сайт используйте Ваш логин или номер карты, и действующий пароль.';
                                        isAssign = true;
                                        additionalTextContainer.remove();
                                    }
                                    removeAdditionalTextContainer = false;
                                }

                                if (removeAdditionalTextContainer) {
                                    additionalTextContainer.remove();
                                }
                                if(isAssign)
                                {
                                    $('.regFinMessageDisplayed').html(messageToDisplay);
                                }
                                else
                                {
                                    $('.regFinMessageDisplayed').html($(messageToDisplay).html());
                                }
                                                            });

                                                            window.location.href = "/personal/";

                                                    } else {
                            if (undefined != result.errors) {
                                RegistrationError.showAll(result.errors);
                            }
                            if (undefined != result.message) {
                                if (result.type == 'info') {
                                    var messageContainer = $('.popupReg').find('.regMainInfo');
                                    if (undefined != result.data.existImUser && result.data.existImUser) {
                                        if (result.data.existImUser) {
                                            fillForm(result.data.userProfile);
                                        }
                                        $('.regInputSubmit').css('background-image', 'url("/static/images/redesign2/registration/getCard.png")');
                                        $('.regInputSubmit').attr('rel', 'getCard');
                                    }
                                } else {
                                    var messageContainer = $('.popupReg').find('.regMainError');
                                }
                                if ($('.errorBox:visible').length < 1) {
                                    messageContainer.html('<p>' + result.message + '</p>');
                                }
                            }
                        }
                    }, error: function (http, message, exc) {
                        //console.log(message);
                        $(document).trigger('form.response.fail.time', {eventContext: 'popup', eventForm: 'popupRegForm'});
                        $(document).trigger('form.submit.fail_time', {eventContext: 'popup', eventForm: 'popupRegForm'});
                        $(document).trigger('form.submit.fail', {eventContext: 'popup', eventForm: 'popupRegForm'});
                    }
                });
            }
            return false;
        });
    })
</script>
