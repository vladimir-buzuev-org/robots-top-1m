        // Autocomplete Schnellsuche via Webservice
        // http://jquery-with-asp.net/
        $("#schnellsuche_input").autocomplete({
	        source: function (request, response) {
	            $.ajax({
	                type: "POST",
	                contentType: "application/json; charset=utf-8",
	                url: "/webservice_schnellsuche_d.asmx/GetSchnellsuche",
	                // data: "{ 'prefixText': '" + request.term + "' }",
	                data: '{\'prefixText\':' + JSON.stringify(request.term) + '}',
	                dataType: "json",
	                success: function (data) {
	                    // here I will store list of all results which
	                    // will be passed to autocomplete widget
	                    var autocompleteOutput = [];
	 
	                    // looping to get all the results out of the list of
	                    // addresses returned by the webservice
	                    $.each(data.d, function (index, AutocompleteItem) {
	                        autocompleteOutput[index] = AutocompleteItem.value;
	                    });
 
	                    // passing all the available options to autocomplete widget
	                    response(autocompleteOutput);
	 
	                    // store results in global variable, it will be needed
	                    // later to get the rest of details for selected option
	                    autocompleteData_1 = data.d;  // alle Daten -> value, suchtyp und objektnr
                      autocompleteData_2 = autocompleteOutput;  // nur value -> mit Hilfe dieses Arrays wird dann für das select-Event der Index des ausgewählten Objekts für den Zugriff auf die weiteren Daten (suchtyp und objektnr) bestimmt
	                },
	                error: function (result) {
	                    alert("wegen unerwarteter Fehler können die Daten nicht geladen werden");
	                }
	            });
	        },
	        minLength: 1,
	        select: function(event, ui) {
            // Index des ausgewählten Objekts bestimmen
            var v_index = jQuery.inArray(ui.item.value, autocompleteData_2);
            // alert('Index: ' + v_index);
            // alert(autocompleteData_1[v_index].suchtyp);

            // mit Hilfe des Index den Suchtyp bestimmen und je nach Suchtyp auf die Zielseite verzweigen
	          switch (autocompleteData_1[v_index].suchtyp) {
              case "Land":
                window.location = '/auswobj_d.aspx?typ=schnellsuche&land=' + autocompleteData_1[v_index].suchwert + '&seitennr=1';
                break;
              case "Region":
                window.location = '/auswobj_d.aspx?typ=schnellsuche&region=' + autocompleteData_1[v_index].suchwert + '&seitennr=1';
                break;
              case "Unterregion":
                window.location = '/auswobj_d.aspx?typ=schnellsuche&unterregion=' + autocompleteData_1[v_index].suchwert + '&seitennr=1';
                break;
              case "Ort":
                window.location = '/auswobj_d.aspx?typ=schnellsuche&ort=' + autocompleteData_1[v_index].suchwert + '&seitennr=1';
                break;
              case "Objektnummer":
                window.location = '/objekt_d.aspx?objnr=' + autocompleteData_1[v_index].suchwert;
                break;                
              default: ;
                break;               
            }
          }
	      });

      function get_Anzahl_Treffer() {     
          // Querystring aufbereiten
          var v_querystring = '';
          
          if ($('#DropDownList_Laender').val().length > 0 && $('#DropDownList_Laender').val().indexOf('bitte auswählen') == -1)
          {
            v_querystring == '' ? v_querystring = v_querystring + '?' : v_querystring = v_querystring + '&'; 
            v_querystring = v_querystring + 'DropDownList_Laender=' + encodeURIComponent($('#DropDownList_Laender').val());          
          }
          
          if ($('#Input_Region').val().length > 0 && $('#Input_Region').val().indexOf('nicht wichtig') == -1)
          {
            v_querystring == '' ? v_querystring = v_querystring + '?' : v_querystring = v_querystring + '&';
