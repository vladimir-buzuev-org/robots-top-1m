        var CarstoreAddress = document.getElementById("Carstore_Address");

        //Init Google Maps
        function init() {
            new google.maps.places.Autocomplete(CarstoreAddress);

            $("#Carstore_Address").change( function(){
                var geocoder = new google.maps.Geocoder();
                var address  = $('#Carstore_Address').val();
                if(address != '') {
                    geocoder.geocode({'address': address}, function (results, status) {
                        if (status == google.maps.GeocoderStatus.OK) {
                            var latitude = results[0].geometry.location.lat();
                            var longitude = results[0].geometry.location.lng();
                            $("#lat").val(latitude);
                            $("#lng").val(longitude);
                        }
                    });
                }
            });
        }
        google.maps.event.addDomListener(window, 'load', init);

        function submitFormCarStore() {
            var LCDV6    = $('#carchoice').find(":selected").val();
            var url      = 'http://www.carstore.citroen.fr';
            var address  = $('#Carstore_Address').val();

            if(address == '' && LCDV6.length == 0){
                $('#carchoice').removeClass('errorInput');
                $('#Carstore_Address').removeClass('errorInput');
                window.open(url,'_blank');
            }
            else if(address != '' && LCDV6.length >= 6){
                $('#carchoice').removeClass('errorInput');
                $('#Carstore_Address').removeClass('errorInput');
                url += '/Resultat?mod='+ LCDV6.substr(0, 4) +'&bod='+ LCDV6.substr(4, 2)+'&lat='+$("#lat").val()+'&lng='+$("#lng").val();
                window.open(url,'_blank');
            }
            else {
                $('#carchoice').addClass('errorInput');
                $('#Carstore_Address').addClass('errorInput');
            }
            return false;
        }

        function validateRegistrationNum() {
            var RegistrationNum = $('#RegistrationNum').val().trim().replace(/\-/g, '');
            var url             = 'https://www.reprise-citroen.fr';
            if(RegistrationNum.length > 0){
                var Regex1 = new RegExp('^[a-zA-Z]{2}-?[0-9]{3}-?[a-zA-Z]{2}$');
                var Regex2 = new RegExp('^[0-9]{1,4}-?[a-zA-Z]{1,3}-?[0-9]{2,3}$');
                var Regex3 = new RegExp('^[0-9]{1,4}-?[a-zA-Z]{1,3}-?2[a-bA-B]$');

                if (Regex1.test(RegistrationNum) || Regex2.test(RegistrationNum) || Regex3.test(RegistrationNum)) {
                    $('#ErrorRegistrationNum').hide();
                    $('#RegistrationNum').removeClass('errorInput');

                    url += '/confirmation/?immat=' + RegistrationNum;
                    window.open(url,'_blank');
                } else {
                    $('#ErrorRegistrationNum').show();
                    $('#RegistrationNum').addClass('errorInput');
                }
            } else {
                $('#ErrorRegistrationNum').hide();
                $('#RegistrationNum').removeClass('errorInput');
                window.open(url,'_blank');
            }

            return false;
        }
