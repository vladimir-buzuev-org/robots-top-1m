function submitForm() {

    /**
     * if a user is searching for author, we want to default sorting to releveance, however if the user decides to change the sort-by we keep it
     * this is done by checking if the author was entered fot the first time, then default to relevence, but if author name is the same as previuos submit, but the sortby has changed
     * it means the user wants a different sort-by so dont change it
     */
    if ($filters_query.val() && queryFilter != $filters_query.val()) {
        $('#filters_sortBy').val("sortByRelevance");
    }

    if (isMobile()) { // in mobile view if a user chooses particular years along side the all-years being still checked, then i
        if($.isArray($filters_years.val())) {
            if ($.inArray("", $filters_years.val()) > -1) {
                var yearsArray = $filters_years.val();
                for (var i = 0; i < yearsArray.length; i++) {
                    if (!yearsArray[i]) {
                        yearsArray.splice(i,1);
                        i--;
                    }
                }
                $filters_years.val(yearsArray);
            }
        }
    } else {
        if (checkedAll && !yearSearchFilterUsed) { // only send all years when filter is not being used, cause if list is filtered then only send the chosen years
            $filters_years.val("");
        }
    }
    $('#catalog-search-form').submit();
}

function isMobile() {
    if ( 
    !/iP(od|hone)/i.test(window.navigator.userAgent) && !/IEMobile/i.test(window.navigator.userAgent) && 
    !/Windows Phone/i.test(window.navigator.userAgent) && !/BlackBerry/i.test(window.navigator.userAgent) && 
    !/BB10/i.test(window.navigator.userAgent) && !/Android.*Mobile/i.test(window.navigator.userAgent)
    ) {
        return false;
    } else {
        return true;
    }
}
