            function GetData(IsloadMore, TabID, TabType, PageNo) {

                var tab = $('.tabbednav-up[tabcontainerid]');
                var tabContainer = $("#" + $(tab).attr("tabcontainerid"));

                var Tabs = GetTabs(TabType);

                            $('#imgloading').show();
                        }
                        else {
                            tabContainer.children("[loader |= true]").show();

    $(document).ready(function () {
        //Added Rotator variables For Featured Content Slider
        var contentTimer = null;
        var loadItem;
        var itemPointer;
        var nextItemPointer;
        var navCount = 0;

        //set first slide
        itemPointer = $('#rotator-itemHolder > .rotator-slide').first();

        slideStart();

        function slideStart() {

            //paint first slide
            loadItem = itemPointer.html();
            $(loadItem).insertBefore("#rotator-slideEnd");

            //increment to next slide
            itemPointer = $('#rotator-itemHolder > .rotator-slide').next();

            //timer to animate slides
            contentTimer = setInterval(function () {

                //paint next slide in queue   
                loadItem = itemPointer.html();
                $(loadItem).insertBefore("#rotator-slideEnd");

                //animate first slide
                $($('#rotator-slideBox > .rotator-slideItem')[0]).animate({
                    marginLeft: '-620'
                }, 1500, function () {

                    //destroy first slide that has moved from view
                    $($('#rotator-slideBox > .rotator-slideItem')[0]).remove();

                    //check for end of slides, if end start back at first slide
                    nextItemPointer = itemPointer.next('#rotator-itemHolder > .rotator-slide');
                    if (nextItemPointer.length == 0) {
                        itemPointer = $('#rotator-itemHolder > .rotator-slide').first();
                    } else {
                        itemPointer = itemPointer.next('#rotator-itemHolder > .rotator-slide');
                    }

                    //animate the nav for slides
                    navRotate();
                });
                //set slide animation interval                 
            }, 8000);
        }

        //track items, set nav for current one
        function navRotate() {
            if (navCount == 2) {
                navCount = 0;
            } else {
                navCount++;
            }
            setNavButtons(navCount);
        }

        //nav buttons click
        $(".rotator-button").click(function () {
            clearInterval(contentTimer);
            var navItem = $(this).attr('tab');
            setNavButtons(navItem);
            $('#rotator-slideBox > .rotator-slideItem').remove();

            itemPointer = $($('#rotator-itemHolder > .rotator-slide')[navItem]);
            loadItem = itemPointer.html();
            $(loadItem).insertBefore("#rotator-slideEnd");
        });

        //flop nav buttons
        function setNavButtons(navItem) {
            $("#rotator-buttonBox > .rotator-button").removeClass("rotator-buttonUp").addClass("rotator-buttonDown");
            $($("#rotator-buttonBox > .rotator-button")[navItem]).removeClass("rotator-buttonDown").addClass("rotator-buttonUp");
        }
    });
