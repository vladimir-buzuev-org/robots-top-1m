<script type="text/javascript">

defer(cookie_monitor);


function defer(method) {
    if (window.jQuery)
        method();
    else
        setTimeout(function() { defer(method) }, 50);
}

function cookie_monitor() {

    jQuery(document).ready(function() {

      trigger();

      jQuery('#close_cookie_notice').click(function() {

        Cookies.set('epa_cookies_accepted', 1);

        trigger();
      });

      jQuery('#more_information').click(function() {
        window.location.href = '/footer/privacy/';
      });

      function trigger() {
        if(Cookies.get('epa_cookies_accepted')) {
          jQuery('#cookie-container').fadeOut(250);
        } else {
          jQuery('#cookie-container').css('display', 'block');
        }
      }

  });
}
