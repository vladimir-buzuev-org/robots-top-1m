    function setMobileFooter() {
        if (is_breakpoint('xs')) {
            var show_footer = false;
            var is_homepage = true;
            var is_recipe = false;
            var is_product = false;

            //show_footer = is_homepage || is_recipe || is_product;
            show_footer = true;

            if (show_footer) {
                $('.navbar-fixed-bottom').addClass('visible-xs').show();

                if (is_recipe) {
                    $('.mobile-recipe-menu').removeClass('hidden-xs').show();
                    handleMobileTextMeForm();
                    handleMobileMailForm('recipe');
                } else if (is_product) {
                    $('.mobile-product-menu').removeClass('hidden-xs').show();
                    handleMobileTextMeForm();
                    handleMobileMailForm('product');
                } else {
                    $('.mobile-home-menu').removeClass('hidden-xs').show();
                }

            } else {
                $('.navbar-fixed-bottom').removeClass('visible-xs').hide();
            }
        }
    }

    function handleMobileTextMeForm() {

        $('#textmeform-mobile').submit(function (e) {
            e.preventDefault();
            e.stopImmediatePropagation();
            $('.mobile-text-container .form-group').removeClass('has-error');
            //$('.mobile-text-container button').hide();
            $("#mobile-text-me-loading-modal").fadeIn();

            to_number = $(".mobile-text-container #mobile-number-input").val();
            accept_terms = $(".mobile-text-container #mobile-text-agree-input").is(':checked');
            birthday = $("#mobile-month-birthday").val() + '/' + $("#mobile-day-birthday").val() + '/' + $("#mobile-year-birthday").val();

            jQuery.ajax({
                type: "get",
                dataType: "json",
                url: 'http://www.smirnoff.com/en-us/wp-admin/admin-ajax.php',
                data: {
                    action: 'sendSMS_ajax',
                    to: to_number,
                    birthday: birthday,
                    accept_terms: accept_terms,
                    post_id: '5'
                },
                success: function (response) {
                    if (response.success) {
                        $(".mobile-text-container #text-me-show").hide();
                        $("#mobile-text-me-loading-modal").delay(300).fadeOut();
                        $(".mobile-text-container #text-me-thankyou").show();
                    } else {
                        $('.mobile-text-container .mobile-birthday-select').addClass('has-error');
                        $('.mobile-text-container .form-group').addClass('has-error');
                        //alert(response.error_message);
                    }
                    // $('.mobile-text-container button').show();
                }
            });

            $("#mobile-text-me-loading-modal").delay(300).fadeOut();
            return false;
        });
    }

    function handleMobileMailForm(postType) {
        $('#mobilemailform').submit(function (e) {

            e.preventDefault();
            e.stopImmediatePropagation();
            $('#mobilemailform .form-group').removeClass('has-error');
            $("#mobile-mail-loading-modal").fadeIn();
            var valid = validateMobileMailForm();
            if (valid) {
                $('#mobilemailform .form-group').removeClass('has-error');
                to = $("#mobile-to-input").val();
                from = $("#mobile-from-input").val();
                msg = $("#mobile-msg-input").val();
                jQuery.ajax({
                    type: "get",
                    dataType: "json",
                    url: 'http://www.smirnoff.com/en-us/wp-admin/admin-ajax.php',
                    data: {action: 'sendMail_ajax', to: to, msg: msg, from: from, post_id: '5', postType: postType},
                    error: function (xhr, status) {
                        // alert(status);
                    },
                    success: function (response) {
                        $("#mobile-mail-show").hide();
                        $("#mobile-mail-loading-modal").delay(300).fadeOut();
                        $("#mobile-mail-thankyou").show();
                        // if(response.success) {}else {}
                    }
                });
            } else {
                $('#mobilemailform .form-group').addClass('has-error');
            }
            $("#mobile-mail-loading-modal").delay(300).fadeOut();
            return false;
        });
    }

    function validateMobileMailForm() {
        $(".modal-error").hide();
        to = $("#mobile-to-input").val();
        from = $("#mobile-from-input").val();
        msg = $("#mobile-msg-input").val();
        chck = $("#mobile-mail-agree-input").prop('checked');
        valid = true;

        if (!msg || 0 === msg.length) {
            valid = false
            //$("#msg-error").show();
        }
        if (!chck) {
            valid = false
            //$("#chck-error").show();
        }
        if (!validateEmail(to)) {
            valid = false;
            //$("#to-error").show();
        }
        if (!validateEmail(from)) {
            valid = false;
            //$("#from-error").show();
        }

        return valid;
    }
