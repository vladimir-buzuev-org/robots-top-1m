                        $("#sinistro_cpf_segurado").val($("#CPF").val());
                        $("#sinistro_sexo_segurado").val($("#sexo").val());
                    } else {
                        $("#sinistro_nome_segurado").val("");
                        $("#sinistro_cpf_segurado").val("");
                        $("#sinistro_sexo_segurado").val("");
                    }
                }
            });

            $("[data-modal-reference=\"SeguroVida\"]").on({
                "click": function() {

                    $("[data-modal-id=\"SeguroVida\"]").modal({
                        backdrop: false
                    });
                }
            });
