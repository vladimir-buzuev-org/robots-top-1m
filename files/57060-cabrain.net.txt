function execSetCondition(data, cnt, level, type, dataList, parentFlg) {
    var isData = false;
    for (var i in data) {
        if (cnt == level) {
            isData = checkCondition(data[i].type, data[i].id, type, dataList.id);
            if (isData) break;
        } else {
            if (cnt + 1 == level && checkCondition(data[i].type, data[i].id, dataList.ptype, dataList.pid)) {
                parentFlg = true;
            }
            isData = execSetCondition(data[i].data, cnt++, level, type, dataList, parentFlg);
        }
        if (isData)break;
    }
    if (parentFlg && !isData) {
        var obj = createConditionObject(type, targetName, dataList.id, dataList.name);
        data.push(obj);
    }
    return isData;
}


function setCondition(type, targetName, dataList, level) {
    var flg;
    var parentFlg;
    for (var i in selectCondition) {

        if (selectCondition[i].active) {
            for (var n in dataList) {
                flg = false;
                parentFlg = false;
                level1:
                for (var m in selectCondition[i].condition.data) {
                    if (level == 1) {
                        flg = checkCondition(selectCondition[i].condition.data[m].type, selectCondition[i].condition.data[m].id, type, dataList[n].id);
                        if (flg) break;
                    } else if (level > 1) {
                        if (level == 2 && checkCondition(selectCondition[i].condition.data[m].type, selectCondition[i].condition.data[m].id, dataList[n].ptype, dataList[n].pid)) {
                            parentFlg = true;
                        }
                        for (var m2 in selectCondition[i].condition.data[m].data) {
                            if (parentFlg) {
                                flg = checkCondition(selectCondition[i].condition.data[m].data[m2].type, selectCondition[i].condition.data[m].data[m2].id, type, dataList[n].id);
                                if (flg) break;
                            }
                            if (level > 2) {
                                if (level == 3 && checkCondition(selectCondition[i].condition.data[m].data[m2].type, selectCondition[i].condition.data[m].data[m2].id, dataList[n].ptype, dataList[n].pid)) {
                                    parentFlg = true;
                                }
                                for (var m3 in selectCondition[i].condition.data[m].data[m2].data) {
                                    if (parentFlg) {
                                        flg = checkCondition(selectCondition[i].condition.data[m].data[m2].data[m3].type, selectCondition[i].condition.data[m].data[m2].data[m3].id, type, dataList[n].id);
                                        if (flg) break;
                                    }
                                }
                                if (parentFlg && !flg) {
                                    var obj = createConditionObject(type, targetName, dataList[n].id, dataList[n].name);
                                    selectCondition[i].condition.data[m].data[m2].data.push(obj);
                                    flg = true;
                                }
                                if (flg) break level1;
                            }
                        }
                        if (parentFlg && !flg) {
                            var obj = createConditionObject(type, targetName, dataList[n].id, dataList[n].name);
                            selectCondition[i].condition.data[m].data.push(obj);
                            flg = true;
                        }
                        if (flg) break level1;
                    }
                }
                if (flg == false) {
                    var obj = createConditionObject(type, targetName, dataList[n].id, dataList[n].name);
                    selectCondition[i].condition.data.push(obj);
                }
            }
        }
    }
    createCondtionText();
}
function createConditionObject(type, targetName, id, name) {
    var obj = new Object();
    obj.type = type;
    obj.key = 'cond_' + conditionCount++;
    obj.name = targetName;
    obj.id = id;
    obj.label = name;
    obj.data = [];
    return obj;
}
function checkCondition(dataType, dataId, type, id) {
    return (dataType == type && dataId == id);
}

    for (var n in data) {
        if (data[n].data.length > 0) {
            execCreateConditonText(tagName, data[n].data, label + data[n].label);

        }
    }

    $(document).on('opening', '#modal_department', function () {

    $('#btn_submit_hd').on('click', function() {
        var formItemName = 'hospital_department';
        var data = [];
        deleteFormValue(formItemName, false);
        $('input[name="hd_check"]').each(function() {
            if ($(this).is(':checked')) {
                addFormValue(formItemName, $(this));
                data.push({'id' : $(this).val(), 'name' : $('label[for="' + $(this).prop('id') + '"]').text()});
            }
        });
        delConditionFromType('hd', data);
        setCondition('hd', formItemName, data, 1);
    });
});

function delCondition(key) {
    var obj;
    for (var i in selectCondition) {
        if (selectCondition[i].active) {
            obj = getDeleteConditonObject(key, selectCondition[i].condition.data);
            break;
        }
    }
    return obj;
}
function getDeleteConditonObject(key, data) {
    var result;
    for (var n in data) {
        if (data[n].data.length > 0) {
            result = getDeleteConditonObject(key, data[n].data);
        }
        if (data[n].key == key) {
           result = data[n];
           data.splice(n, 1);
        }
    }
    return result;
}
function delConditionFromType(type, dataList) {
    var obj;
    for (var i in selectCondition) {
        if (selectCondition[i].active) {
            obj = getDeleteConditonObjectFromType(type, selectCondition[i].condition.data, dataList);
            break;
        }
    }
    return obj;
}
function delAllConditionFromType(type, dataList) {
    var obj;
    for (var i in selectCondition) {
        obj = getDeleteConditonObjectFromType(type, selectCondition[i].condition.data, dataList);
    }
    return obj;
}
function getDeleteConditonObjectFromType(type, data, dataList) {
    var isSet;
    for (var i in data) {
        if (data[i].data.length > 0) {
            getDeleteConditonObjectFromType(type, data[i].data, dataList);
        }
        if (data[i].type == type) {
            isSet = false;
            for (var n in dataList) {
                if (dataList[n].id == data[i].id) {
                    isSet = true;
                }
            }
            if (!isSet) {
                deleteConditionFormItem(data[i]);
                data.splice(i, 1);
                getDeleteConditonObjectFromType(type, data, dataList);
            }
        }
    }
}
