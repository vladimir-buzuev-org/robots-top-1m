/**
 * See if there are any CJK ideographs is a string 
 */

function isCJKChar(ch) {
	if (ch < 0x3300) return false;
	if (ch <= 0x9FFF) return true;
	if ((ch >= 0xFE30 && ch <= 0xFE4F) || (ch >= 0xF900 && ch <= 0xFAFF) ||
		(ch >= 0x20000 && ch <= 0x2A6DF) || (ch >= 0x2A700 && ch <= 0x2B81F) ||
		(ch >= 0x2F800 && ch <= 0x2FA1F)) {
		return true;
	}
	return false;
}

function containsCJKIdeograph(str) {
	var codepoints = punycode.ucs2.decode(str);
	for (var i=codepoints.length-1; i >= 0; --i) {
		if (isCJKChar(codepoints[i])) {
			return true;
		}
		return false;
	}
}

</script>

(function ($) {

/**
* @function
* @property {object} jQuery plugin which runs handler function once specified element is inserted into the DOM
* @param {function} handler A function to execute at the time when the element is inserted
* @param {bool} shouldRunHandlerOnce Optional: if true, handler is unbound after its first invocation
* @example $(selector).waitUntilExists(function);
*/

$.fn.waitUntilExists    = function (handler, shouldRunHandlerOnce, isChild) {
    var found       = 'found';
    var $this       = $(this.selector);
    var $elements   = $this.not(function () { return $(this).data(found); }).each(handler).data(found, true);

    if (!isChild)
    {
        (window.waitUntilExists_Intervals = window.waitUntilExists_Intervals || {})[this.selector] =
            window.setInterval(function () { $this.waitUntilExists(handler, shouldRunHandlerOnce, true); }, 200)
        ;
    }
    else if (shouldRunHandlerOnce && $elements.length)
    {
        window.clearInterval(window.waitUntilExists_Intervals[this.selector]);
    }

    return $this;
}

}(jQuery));

<script type="text/javascript">
function translate_placeholder() {
    $('.placeholder').each(function() {
        var trid = $(this).attr('id') + '_tr';

        if ($.support.placeholder) {
            // non IE, just straight replacement of the translated version
            $(this).attr('placeholder', $('#' + trid).text());
        }
        else {
            //take care IE
            var _placeholder = $(this).attr('placeholder'),
            _val = $(this).val();
            // only replace if the content is the same as place holder
            if (_placeholder && (_val == _placeholder)) {
                $(this).attr('placeholder', $('#' + trid).text());
                $(this).val($('#' + trid).text());
            }
        }
    });
}

function triggerTranslation(preferredLang) {
    var checkLang = ($(".goog-te-combo").find("[value='"+preferredLang+"']").length);
    if (checkLang > 0) {
        var nodes = document.getElementById(':0.targetLanguage').childNodes;
        for(var i=0; i<nodes.length; i++) {
            if (nodes[i].nodeName.toLowerCase() == 'select') {
                // setting the preferred language in the google translation widget
                nodes[i].value = preferredLang;
                // create an change event so google  translation can do its translation
                var evt = document.createEvent("HTMLEvents");
                evt.initEvent("change", false, true);
                nodes[i].dispatchEvent(evt);
                break;
             }
        }
    }
}

function checkForTranslation() {
    var preferredLang = "";
    //console.log('preferredLang ' + preferredLang);
    if (preferredLang) {
        var preferredLangArr = preferredLang.split(",");
        preferredLang = preferredLangArr[0];
        if ($(".goog-te-combo").find("[value='"+preferredLang+"']").length == 0) {
            if (preferredLang.indexOf("-") > 0) {
                preferredLang = preferredLang.substring(0, preferredLang.indexOf("-"));
            }
        }
    }
    else {
        preferredLang = 'en';
    }

    var googTrans = $.cookie('googtrans');
    if (googTrans) {
        googTrans = googTrans.substring(googTrans.lastIndexOf("/") + 1);
    }
    else {
        googTrans = 'en';
    }

    if (preferredLang != googTrans) {
        var translateToPrefLang = 'no';
        var translateConfirmCookie = $.cookie('merlot_translate_confirm');
        if (translateConfirmCookie != 'shown') {
            $('#translateConfirmLightboxDiv').lightbox_me({
                centered: true,
                overlaySpeed: 100,
                onLoad: function() {
                    $('#translateNoButton').click(function(e) {
                        if ($('#hideTranslateMsg').prop('checked')) {
                            $.cookie('merlot_translate_confirm', 'shown', { path: '/' });
                            $.cookie('translateToPrefLang', translateToPrefLang, { path: '/' });
                        }
                        $('#translateConfirmLightboxDiv').trigger('close');
                    });
                    $('#translateYesButton').click(function(e) {
                        translateToPrefLang = 'yes';
                        if ($('#hideTranslateMsg').prop('checked')) {
                            $.cookie('merlot_translate_confirm', 'shown', { path: '/' });
                            $.cookie('translateToPrefLang', translateToPrefLang, { path: '/' });
                        }
                        $('#translateConfirmLightboxDiv').trigger('close');
                        triggerTranslation(preferredLang);
                    });
                    // If lightbox is present, set the close button to get the
                    // focus
                    $('#translateYesButton').focus();
                    },
            });
        } else {
