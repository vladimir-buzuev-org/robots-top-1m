function getga_has_universal_event_tracking()
{
	var scripts = document.getElementsByTagName('script');

	for (var i = scripts.length; i--;)
	{
	    if (scripts[i].src.indexOf(String.fromCharCode(103,111,111,103,108,101,45,97,110,97,108,121,116,105,99,115,46,99,111,109,47,97,110,97,108,121,116,105,99,115,46,106,115)) > 0)
	    {
	    	return true;
	    }
	}

	return false;
}

function getga_add_event_tracking(has_universal_event_tracking, getga_selector, getga_event, getga_category, getga_action_label, getga_label)
{
	jQuery(function($){

		$(getga_selector).on(getga_event, function(e){

			var NA = 'undefined';

			
			if(getga_label.indexOf('{ITEM_TITLE}') > -1)
			{
				getga_label = getga_label.replace("{ITEM_TITLE}", ($(this).attr('title') !== undefined ? $(this).attr('title') : NA));
			}

			if(getga_label.indexOf('{PAGE_URL}') > -1)
			{
				getga_label = getga_label.replace("{PAGE_URL}", location.href);
			}

			if(getga_label.indexOf('{PAGE_RELATIVE_URL}') > -1)
			{
				getga_label = getga_label.replace("{PAGE_RELATIVE_URL}", location.pathname);
			}

			if(getga_label.indexOf('{LINK_URL}') > -1)
			{
				getga_label = getga_label.replace("{LINK_URL}", ($(this).attr('href') !== undefined ? $(this).attr('href') : NA));
			}

			if(getga_label.indexOf('{LINK_RELATIVE_URL}') > -1)
			{
				getga_label = getga_label.replace("{LINK_RELATIVE_URL}", ($(this).attr('href') !== undefined ? $(this).attr('href').replace(/^(?:\/\/|[^\/]+)*\//, '/') : NA));
			}

			if(getga_label.indexOf('{IMAGE_SRC}') > -1)
			{
				getga_label = getga_label.replace("{IMAGE_SRC}", ($(this).attr('src') !== undefined ? $(this).attr('src') : NA));
			}

			if(getga_label.indexOf('{IMAGE_ALT}') > -1)
			{
				getga_label = getga_label.replace("{IMAGE_ALT}", ($(this).attr('alt') !== undefined ? $(this).attr('alt') : NA));
			}

			if(getga_label.indexOf('{TAG_HTML}') > -1)
			{
				getga_label = getga_label.replace("{TAG_HTML}", $(this).html().replace(/(<([^>]+)>)/ig,'').replace(/(\r\n|\n|\r)/gm,'').replace(/\s{2,}/g, ' '));
			}

			if(has_universal_event_tracking)
			{
				try
				{
					ga('send', 'event', getga_category, getga_action_label, getga_label);
									}
				catch(err)
				{
									}
			}
			else
			{
				try
				{
					_gaq.push(['_trackEvent', getga_category, getga_action_label, getga_label]);
									}
				catch(err)
				{
									}
			}
		});

	});
}
