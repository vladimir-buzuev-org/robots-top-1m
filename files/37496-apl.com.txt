function Pager(tableName, itemsPerPage) {
    	
    this.tableName = tableName;
    this.itemsPerPage = itemsPerPage;
    this.currentPage = 1;
    this.pages = 0;
	this.inited = false;
    this.pagerName = "";
    this.positionIds = new Array();
    
    this.addPositionId = function(positionId) {
                var idx = this.positionIds.length;
                this.positionIds[idx] = positionId;
    }
    
    this.checkExistingPositionId = function(positionId) {
                for(i=0;i<this.positionIds.length;i++)
                                if(this.positionIds[i]==positionId) {
                                                return true;
                                }
                return false;
    }    
    
    this.showRecords = function(from, to) {        
        var rows = document.getElementById(tableName).rows;
        // i starts from 1 to skip table header row
        for (var i = 1; i < rows.length; i++) {
            if (i < from || i > to)  
                rows[i].style.display = 'none';
            else
                rows[i].style.display = '';
        }
    }
    
    this.showPage = function(pageNumber) {
		      
                if (! this.inited) {
                                alert("not inited");
                                return;
                }


                for(i=0;i<this.positionIds.length;i++) {
                        var oldPageAnchor = document.getElementById('pg_' + this.positionIds[i] + this.currentPage);
                        oldPageAnchor.className = 'pg-normal';
                }

                this.currentPage = pageNumber;
				for(i=0;i<this.positionIds.length;i++) {
                        this.showPageNav(this.pagerName, this.positionIds[i]);
                        var newPageAnchor = document.getElementById('pg_' + this.positionIds[i] + this.currentPage);
                        newPageAnchor.className = 'pg-selected';
						newPageAnchor.style.color = '#3D3D3D';
						newPageAnchor.style.cursor = 'none';
                }
        
	//alert("pageNumber => " + pageNumber + " itemsPerPage => " + parseInt(itemsPerPage));
        var from = (pageNumber - 1) * itemsPerPage + 1;
        var to = parseInt(from) + parseInt(itemsPerPage) - 1;
	 //alert("from => " + from + "   To => " + to);
        this.showRecords(from, to);
    }   
    
    this.prev = function() {
        if (this.currentPage > 1)
            this.showPage(this.currentPage - 1);
    }
    
    this.next = function() {
        if (this.currentPage < this.pages) {
            this.showPage(this.currentPage + 1);
        }
    }                        
    
    this.init = function() {

    	
    	this.currentPage = 1;
        this.pages = 0;
        var rows = document.getElementById(tableName).rows;
        var records = (rows.length - 1); 
        this.pages = Math.ceil(records / itemsPerPage);
        this.inited = true;
    }

    this.showPageNav = function(pagerName, positionId) {

                if (! this.inited) {
                                alert("not inited");
                                return;
                }
                var element = document.getElementById(positionId);


                var numOfPagination = 5;
                this.pagerName = pagerName;
                if(this.checkExistingPositionId(positionId)==false)
                                this.addPositionId(positionId);

				var pagerHtml='';
                    
                    if(this.currentPage==1){

                    //pagerHtml = '<p class="medium"><span  class="leftarrowdisable">First</span> ';
                    //pagerHtml+= '<span  class="leftarrowdisable">Previous</span>';
					pagerHtml = '<p class="medium">';
					pagerHtml+= '<span  class="leftarrowdisable"><img src="images/left.jpg"/></span>';

					}else{

                    pagerHtml = '<p class="medium">';
                    //pagerHtml+= '<a  class="leftarrow" href="#navhome" onclick="' + pagerName + '.prev();" >Previous</a>';					
					pagerHtml+= '<a class="leftarrow" href="#navhome" onclick="' + pagerName + '.prev();" ><img src="images/left.jpg"/></a>';					
                    }
                                
        var begin = this.currentPage;
        if(this.pages<(this.currentPage+numOfPagination)) {

                begin = this.pages - numOfPagination + 1;
                if(begin<=0)
                                begin = 1;
        }      
        var i = 0;
        for (var page = begin; i < numOfPagination && page <= this.pages; page++) { 

                pagerHtml += '<a id="pg_'+ positionId + page + '" class="pg-normal-numbers"  href="#navhome" onclick="' + pagerName + '.showPage(' + page + ');">' + page + '</a> ';
                i = i + 1;
        }
        
        if(this.pages==this.currentPage){

        //pagerHtml += '<span class="rightarrowdisable" >Next</span> ';
		pagerHtml += '<span class="rightarrowdisable" ><img src="images/right.jpg"/></span> ';
        pagerHtml += '</p> ';            
        }else{

        //pagerHtml += '<a class="rightarrow" href="#navhome" onclick="'+pagerName+'.next();" >Next</a> ';
		//pagerHtml += '<a class="rightarrow" href="#navhome" onclick="'+pagerName+'.showPage('+this.pages+');" >Last</a></p> ';            
		pagerHtml += '<a class="rightarrow" href="#navhome" onclick="'+pagerName+'.next();" ><img src="images/right.jpg"/></a> ';
        pagerHtml += '</p> ';            
        
		}
        if(this.pages==1){  
        element.innerHTML = '';
		}else{
			
        	element.innerHTML = '<div id="underline" style="margin-bottom:15px;"></div>'+pagerHtml;
			
        }
        
         
    }
    
    
}

				function putErr(ele,msg){

				$("[data-err-msg-for='"+eId+"']").show();
