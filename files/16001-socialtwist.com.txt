			paramName = paramName.replace(/[\[]/,"\\\[").replace(/[\]]/,"\\\]");
			var regexS = "[\\?&]"+paramName+"=([^&#]*)";
			var regex = new RegExp( regexS );
			var results = regex.exec(queryParams);
			if( results == null )
			  return "";
			else
			  return results[1];
      }
