  var w = window;
  var M = Math;
  var ls = w.localStorage;
  var ss = w.sessionStorage;

  if(w.MetaTracker !== void 0) {
    throw new Error('MetaTracker is already set on the global object.');
  }

  // Generates a UUID-like string
  function uuid() {
    function s4() {
      return M.floor((1 + M.random()) * 0x10000)
        .toString(16)
        .substring(1);
    }

    return s4() + s4() + '-' + s4() + '-' + s4() + '-' +
      s4() + '-' + s4() + s4() + s4();
  }

  // Returns a random integer in the [min, max] interval
  function randomInInterval(min, max) {
    return M.floor(M.random() * (max - min + 1)) + min;
  }

  // Returns if a Google Analytics object is present (it might be spoofed, but it is there).
  function hasGoogleAnalyticsObject() {
    return typeof w === 'object' &&
      typeof w.ga == 'function' &&
      typeof w.ga.create === 'function' &&
      typeof w.ga.create.toString == 'function';
  }

  // Returns if a Google Analytics object is present but is spoofed; uBlock Origin and possibly other blockers do that.
  // @see {https://github.com/gorhill/uBlock/blob/de1ed89f62bf041416d2a721ec00741667bf3fa8/assets/ublock/resources.txt#L201-L243}
  function isGoogleAnalyticsObjectSpoofed() {
    return hasGoogleAnalyticsObject() &&
      Function.prototype.toString.call(w.ga.create).match('Tracker') !== null;
  }

  // Map of available measurements; each key maps to a function that is passed the current timing as first parameter,
  // and the MetaTracker instance as this context.
  var measurements = {
    // User ID (from localStorage, if available)
    uid: function uid() {
      if(typeof ls !== 'undefined') {
        return ls.getItem(this.storagePrefix + '.uid');
      }
      return 'unsupported';
    },

    // Session ID (from sessionStorage, if available)
    sid: function sid() {
      if(typeof ss !== 'undefined') {
        return ss.getItem(this.storagePrefix + '.sid');
      }
      return 'unsupported';
    },

    // Post date (in Date.toJSON format)
    // @see {https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Date/toJSON}
    // Its name is not camelCase because it has a special meaning for ElasticSearch
    post_date: function postDate() {
      return (new Date()).toJSON();
    },

    // Timing after start
    timing: function timing(_timing) {
      return _timing;
    },

    // Site (eg. 'allocine', 'jeuxvideo')
    site: function site() {
      return this.site;
    },

    // Current page location
    href: function href(state) {
      return w.location.href;
    },

    // Presence of the Google Analytics Object, but not spoofed
    ga: function ga() {
      return hasGoogleAnalyticsObject() && !isGoogleAnalyticsObjectSpoofed();
    },

    // Presence of the Google Analytics Object, but spoofed
    spoofedGa: function spoofedGa() {
      return hasGoogleAnalyticsObject() && isGoogleAnalyticsObjectSpoofed();
    },

    // Presence of the Google Tag Manager Object
    gtm: function gtm() {
      return typeof w.google_tag_manager === 'object';
    },

    // Presence of the eStat (Médiamétrie) Object
    eStat: function eStat() {
      return typeof w.eStat_Whap === 'function';
    },

    // Presence of the eStat tracker function (should match eStat() in all known cases)
    eStatSend: function eStatSend() {
      return typeof w.eStatWhap === 'object' && typeof w.eStatWhap.send === 'function';
    },

    // Is Adblock detected according to the Webediads detection script?
    wadsBlocking: function wadsBlocking() {
      return !!w.wadsBlocking;
    },

    // Is the wads global object present?
    wads: function wads() {
      return typeof w.wads === 'object';
    },

    // Status of the "Do Not Track" setting an the browser level
    // @see {https://developer.mozilla.org/en-US/docs/Web/API/Navigator/doNotTrack}
    doNotTrack: function doNotTrack() {
      return w.navigator.doNotTrack;
    },

    // Navigator UA String
    // Deprecated, but still widely implemented
    // @see {https://developer.mozilla.org/en-US/docs/Web/API/NavigatorID/userAgent}
    userAgent: function userAgent() {
      return w.navigator.userAgent;
    },

    // Window inner height
    height: function height() {
      return w.innerHeight;
    },

    // Window inner width
    width: function width() {
      return w.innerWidth;
    },

    // Metadata associated with the MetaTracker
    meta: function meta(timing) {
      if(typeof this.meta === 'function') {
        return this.meta.call(null, timing);
      }
      return this.meta;
    },
  };

  // Core MetaTracker class, exposed on the global window object.
  function MetaTracker(opts) {
    this.timings = opts.timings || [0, 1000, 5000, 10000, 30000, 60000];
    this.es = {};
    var es = opts.es || {};
    this.es.domain = es.domain || '193.36.45.180:80';
    this.es.index = es.index || 'meta-tracker';
    this.es.type = es.type || 'probe';
    this.endpoint = 'http://' + this.es.domain + '/' + this.es.index + '/' + this.es.type;
    this.sampling = opts.sampling || 100;
    this.site = opts.site;
    this.storagePrefix = opts.storagePrefix || 'MetaTracker';
    this.meta = opts.meta;

    this.started = false;

    this.start = this.start.bind(this);
    this.scheduleMeasurements = this.scheduleMeasurements.bind(this);
    this.performMeasurements = this.performMeasurements.bind(this);
    this.sendProbe = this.sendProbe.bind(this);

    if(opts.autostart !== false) {
      this.start();
    }
  }

  // Factory function
  MetaTracker.create = function create(opts) {
    return new MetaTracker(opts);
  }

  MetaTracker.sample = function sample(sampling) {
    return randomInInterval(0, sampling - 1) === 0;
  };

  // Start the MetaTracker instance by issuing measurements scheduled timeouts
  MetaTracker.prototype.start = function start() {
    if(this.started) {
      throw new Error('MetaTracker already started');
    }
    this.started = true;
    if(MetaTracker.sample(this.sampling)) {
      if(typeof ls !== 'undefined') {
        if(ls.getItem(this.storagePrefix + '.uid') === null) {
          ls.setItem(this.storagePrefix + '.uid', uuid());
        }
      }

      if(typeof ss !== 'undefined') {
        if(ss.getItem(this.storagePrefix + '.sid') === null) {
          ss.setItem(this.storagePrefix + '.sid', uuid());
        }
      }

      this.timings.forEach(this.scheduleMeasurements);
    }
  };

  // Schedules a single run of measurements at the given timing
  MetaTracker.prototype.scheduleMeasurements = function scheduleMeasurements(timing) {
    setTimeout(this.performMeasurements, timing, timing);
  };

  // Performs a scheduled measurement at the given timing
  MetaTracker.prototype.performMeasurements = function performMeasurements(timing) {
    var probe = Object.create(null);
    Object.keys(measurements).forEach(function performMeasurement(key) {
      try {
        probe[key] = measurements[key].call(this, timing);
      }
      catch(err) {
        probe[key] = 'error: ' + err.toString();
      }
    }.bind(this));
    this.sendProbe(probe);
  };

  // Send a probe to the ElasticSearch endpoint using an XHR
  MetaTracker.prototype.sendProbe = function sendProbe(probe) {
    var id = uuid();
    var xhr = new XMLHttpRequest();
    xhr.open('PUT', this.endpoint + '/' + id + '/_create');
    xhr.setRequestHeader('Content-type', 'application/x-www-form-urlencoded');
    xhr.send(JSON.stringify(probe));
  };

  w.MetaTracker = MetaTracker;
})()
</script>
<script>window.MetaTracker.create({ site: 'millenium/v7' });</script>
