    function updateLanguage(select) {
        var selOpt = select.options[select.selectedIndex].value;
        if (selOpt === '0') {
            jQuery('#bul-popin-languages').find('option').remove();
            jQuery('#bul-popin-languages').attr('disabled', 'disabled');
        } else {
            $j.ajax({
                url: 'https://www.bulgari.com/en-de/alpenregion/countries/getLanguagesOfWebsite/',
                data: {website_group_id: select.options[select.selectedIndex].value},
                success: function (data) {
                    if (data.length > 0) {
                        jQuery('#bul-popin-languages').html('').append('<option value="0" selected>Select Language</option>');
                        for (var i = 0; i < data.length; i++) {
                            jQuery('#bul-popin-languages').append('<option value="' + data[i].code + '">' + data[i].label + '</option>');
                        }
                        jQuery('.bul-popin-languages').each(function () {
                            jQuery(this).data('customselect').refresh();
                        });
                    }
                }
            });
        }
    }

    function getCookieCart() {
        var cookies = document.cookie.split(';');
        for (var i = 0; i < cookies.length; i++) {
            var name = cookies[i].split('=')[0];
            if (name.trim() == 'cartQty') {
                return parseInt(cookies[i].split('=')[1]);
            }
        }
        return 0;
    }

                        var cookies = document.cookie.split(';');

                            var name = cookies[i].split('=')[0];
                            if(name.trim() == 'customer_privacy_policy') {
