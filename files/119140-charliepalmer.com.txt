<script>
    var LightBox = (function(jQuery) {

        var open = false;
        var modal = document.querySelector("#phone-lightbox")

        function init() {
            addEventListeners();
            initScroll();
        }

        function addEventListeners() {
            var catBtn = document.querySelector('.call-us--btn')
            if(catBtn) {
                catBtn.addEventListener('click', toggleModal);
            }
            var close = document.querySelector('#phone-lightbox .close-lightbox')
            if(close) {
                close.addEventListener('click', toggleModal);
            }
            if(modal){
                modal.addEventListener('click', handleModalClick);
            }
        }

        function initScroll() {
            jQuery('#phone-lightbox .phone-lightbox-inner').mCustomScrollbar();
        }

        function toggleModal() {
            if(modal){
                open = !open;
                toggleClass(modal, "show")
            }
        }

        function handleModalClick (e) {
            if(e.target && e.target.id === "phone-lightbox") {
                toggleModal();
            }
        }

        function toggleClass(el, className) {
            if (el.classList) {
              el.classList.toggle(className);
            } else {
              var classes = el.className.split(' ');
              var existingIndex = classes.indexOf(className);

              if (existingIndex >= 0)
                classes.splice(existingIndex, 1);
              else
                classes.push(className);

              el.className = classes.join(' ');
            }
        }

        return {
            init: init
        }
    })(jQuery);

    LightBox.init();
