		$("body").on("submit","#searchForm", function(){
			
			var inputValue = $("#textWrite").val();

			inputValue = inputValue.replace(/\&/g, '&amp;');
			inputValue = inputValue.replace(/\"/g, '&quot;');
			inputValue = inputValue.replace(/\</g, '&lt;');
			inputValue = inputValue.replace(/\>/g, "&gt;");
			inputValue = inputValue.replace(/\'/g, "&#39;"); 
			inputValue = inputValue.replace(/\\/g, ""); 
			
			$("#textToFind").val(inputValue);
		});

	
	$(window).ready(function(){
 
	    $( "#textWrite" ).autocomplete({
	      source: function( request, response ) {
	        $.ajax({
	          url: 'https://www.inail.it/rest-public-enterprisesearch-gw/SuggestionService',
	          dataType: "jsonp",
	        jsonp: "callback",
	        jsonpCallback: 'jsonpCallback',
	        contentType: "application/json",		  
			  data: {
	            textToFind: request.term,
				searchScope: 'internet'
	          },
	          success: function( data ) {
	            response( data.slice(0,10) );
	          },
	          error: function(){
	              response([]);
	              }
	        });
	      },
	      minLength: 1,
		  select: function(event, ui) { 
			$(event.target).val(ui.item.value); 
	        $("form[name='searchForm']").submit(); }
	    });

  });
