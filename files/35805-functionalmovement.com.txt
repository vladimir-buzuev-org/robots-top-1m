    function getcart() {
        $.ajax({
            url: '/store/getcart',
            success: function (cart) {
                if (cart) {
                    appCart.$data.cart = cart;
                    appCart.$data.showcart = true;
                    appCartCount.$data.cartCount = cart.Rows.length + cart.EventRows.length;
                    appCartCount.$data.shownumber = true;
                    if (appCartCount.$data.cartCount > 0) {
                        $("#checkoutBtn").show();
                        $("#orderSummary").show();
                        $("#noRecords").hide();
                    }
                    else {
                        $("#checkoutBtn").hide();
                        $("#orderSummary").hide();
                        $("#noRecords").show();
                    }
                }
            }
        });
    };



    // vue mod for Cart Count
    var appCartCount = new Vue({
        el: '#eCartCount',
        data: {
            cartCount: '',
            shownumber: false
        }
    });

    var appCart = new Vue({
        el: '#eCart',
        data: {
            cart: '',

        },
        methods:
            {

                        url: '/store/removefromcart' + '/' + itemId,

                        type: "GET",
                        contentType: 'application/json; charset=utf-8',
                        cache: false,
                        data: {},

                            if (data === 'True') {

                            }

                    var itemId = event.currentTarget.dataset.value;
                    var url = '/events/event' + '/' + itemId;

                }
            }

<script>
    function sendEmailContact() {
        var recipient = $('#Recipient').val();
        var email = $('#EmailAddress').val();
        var name = $('#Name').val();
        var topic = $('#Topic').val();
        var message = $('#Message').val();

        $.ajax({
            type: "POST",
            url: '/home/SendContact',
            data: { 'recipient': recipient, 'email': email, 'name': name, 'topic': topic, 'message': message },
            success: function (msg) {
                if (msg.result === 'ok')
                { $("#message-contact").text('Contact Email Successfully sent to ' + recipient); }
                else
                { $("#message-contact").text('Contact Email Failed To Send'); }
                $("#contact-modal").modal('hide');        
                $("#emailAlert-sendContact").modal('show');
                return false;
            }
        });
    }
