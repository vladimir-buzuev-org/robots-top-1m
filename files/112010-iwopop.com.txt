    function GetRequest() {
        var url = location.search;
        var theRequest = new Object();
        if (url.indexOf("?") != -1) {
            var str = url.substr(1);
            strs = str.split("&");
            for (var i = 0; i < strs.length; i++) {
                theRequest[strs[i].split("=")[0]] = unescape(strs[i].split("=")[1]);
            }
        }
        return theRequest;

    function FilterList() {
        var tpl_category_id = $.trim($("#templateCategories a[class='choose']").attr("tpl_category_id"));
        var tpl_belong_color = $.trim($("#templateColorList a[class*='choose']").attr("tpl_belong_color"));
        var datalist = $("#dataList");

        datalist.children("li").attr("isavailable", "false");

        if (tpl_category_id != "0" && tpl_belong_color != "") {
            datalist.children("li[tpl_category_id='" + tpl_category_id + "'][tpl_belong_color*='" + tpl_belong_color + "']").attr("isavailable", "true");
        } else if (tpl_belong_color != "") {
            datalist.children("li[tpl_belong_color*='" + tpl_belong_color + "']").attr("isavailable", "true");
        } else if (tpl_category_id != "0") {
            datalist.children("li[tpl_category_id='" + tpl_category_id + "']").attr("isavailable", "true");
        } else {
            datalist.children("li").attr("isavailable", "true");
        }
        initClass();
        changePageManage(datalist.children("li[isavailable='true']").size());
    }

    /****************************************
    * 函数名称：ChangePageManage 查询后分页回调方法 
    *****************************************/
    function changePageManage(ishowcount) {
        if (ishowcount > 0) {
            $("#noData").hide();
            $("#dataList").show();
            var pagesize = $("#showpagehtml").attr("pagesize");
            var fpage = ishowcount % pagesize != 0 ? parseInt(ishowcount / pagesize) + 1 : parseInt(ishowcount / pagesize);
            pageManage(1, fpage, pagesize);
        } else {
            $("#dataList").hide();
            $("#noData").show();
        }
    }
    /****************************************
    * 函数名称：SearchTemplateId 查询模板id
    *****************************************/
    function SearchTemplateId() {
        var id = $("#txtSearchTemplates").val();
        var categoryId = $("#hidCategorys").val();
        var colorType = $("#hidColor").val();
        if (/^\d+$/.test(id) && id != "") {
            $("#dataList li").each(function () {//所有类
                if (parseInt(categoryId) == 0) {
                    if ($(this).attr("tpl_id") == id) {
                        $(this).removeClass("dn");
                    } else {
                        $(this).addClass("dn");
                    }
                } else {//选择分类
                    if (parseInt(colorType) == 0) {//颜色分类
                        if ($(this).attr("tpl_id") == id && $(this).attr("tpl_category_id") == categoryId) {
                            $(this).removeClass("dn");
                        } else {
                            $(this).addClass("dn");
                        }
                    } else {
                        if ($(this).attr("tpl_id") == id && $(this).attr("tpl_category_id") == categoryId && $(this).attr("tpl_belong_color").indexOf(colorType) != -1) {
                            $(this).removeClass("dn");
                        } else {
                            $(this).addClass("dn");
                        }
                    }
                }
            });
            var data_url = $("li[tpl_id='" + id + "']").children().children().attr("data-url");
            $("li[tpl_id='" + id + "']").children().children().attr("src", data_url);
            $(".viewpage").show().siblings().hide();
        }
    }
