  function customizeFlexSlider(){
    var activeWidth = document.body.parentElement.clientWidth;
    var resolutionClass;
    if(activeWidth < 640){
      resolutionClass = 'mobile';
      if(!jQuery('.flexslider').hasClass('mobileVersion')){
        jQuery('.flexslider').addClass('mobileVersion');
      }
      
      jQuery('.flexslider').removeClass('tabletVersion desktopVersion');
    }
    else if(activeWidth >= 640 && activeWidth < 1220){
      resolutionClass = 'tablet';
      if(!jQuery('.flexslider').hasClass('tabletVersion')){
        jQuery('.flexslider').addClass('tabletVersion');
      }
      
      jQuery('.flexslider').removeClass('mobileVersion desktopVersion');
    }
    else {
      resolutionClass = 'desktop';
      if(!jQuery('.flexslider').hasClass('desktopVersion')){
        jQuery('.flexslider').addClass('desktopVersion');
      }
      
      jQuery('.flexslider').removeClass('mobileVersion tabletVersion');
    }

    // Rewrite data attributes
    jQuery('.flexslider .slideWrapper video').each(function(i, element) {
      element.pause();
    });
    jQuery('.flexslider .slideWrapper.'+resolutionClass).each(function(key, element){
        // image handling
        var $img = jQuery(element).find('.imageContainer img');
        var dataSrc = $img.attr('data-src');
        var $imgContainer = jQuery(element).find('.imageContainer');
        if(dataSrc != undefined){
          $img.attr('src', dataSrc);
          $imgContainer.css('background-image', 'url('+ dataSrc + ')');
        }
        
        jQuery(element).find('.imageContainer video').each(function() {
          this.play();
        })
      });

    // Manipulate selector class
    jQuery.each(jQuery('.visibleSlide'), function(key, element){
      jQuery(element).removeClass('visibleSlide');
    });

    jQuery('.flexslider .slides li').each(function(key, element){
      var visibleDataAttr = jQuery(element).attr('data-visible-'+resolutionClass);
      if(visibleDataAttr == 1){
        if(!jQuery(element).hasClass('visibleSlide')){
          jQuery(element).addClass('visibleSlide');
        }
      }
    });
  }

    var XtxDataTipped = function(tag) {
      var jqTag = jQuery(tag);
      var data = jqTag.data('tipped-querystring');
      var skin = jqTag.data('tipped-skin') || 'w4yBorder';
      var options = jqTag.data('tipped-options') || {};
      var value = jqTag.data('tipped-value') || '';

      if(value == '' && jqTag.data('tipped-tag')) {
        value = jQuery(jqTag.data('tipped-tag')).get(0);
      }

      options = jQuery.extend(true, {
        skin: skin
      }, options);

      if(jqTag.data('tipped-afterUpdate')) {
        options.afterUpdate = jqTag.data('tipped-afterUpdate');
      }

      if(value != '') {
        return Tipped.create(tag, value, options);
      } else {
        options = jQuery.extend(true, {
          ajax: {
            cache: false,
            data: data
          }
        }, options);

        if(data) {
          return Tipped.create(tag, jqTag.data('tipped-url') + '?' + data, options);
        } else {
          return Tipped.create(tag, jqTag.data('tipped-url'), options);
        }
      }
    }
