        $(document).ready(function() {
	    MAK.init();

    	    var _images = [];
	    var _images_current = 0
	    var _timer_run = 1;

	    function show_image(_id, _speed) {
		$(".news-image").stop();
	        $(".news-line-selected").removeClass("news-line-selected");
	        $("#news-line-" + _id).addClass("news-line-selected");
	        $("#news-image-" + _id).fadeTo(_speed, 1, function() {
	    	    for (var i in _images) {
			if (_images[i] != _id) {
			    $("#news-image-" + _images[i]).fadeTo(_speed, 0);
			} else {
		    	    _images_current = i;
		        }
		    }
		});
	    }

	    $(".news-image").each(function() { _images.push($(this).data("id")); });
	
	    show_image(_images[0], 1);

	    $(".news-line").hover(
		function() { _timer_run = 0; show_image($(this).children("div").children("a").data("id"), 200); },
		function() { _timer_run = 1; }
	    );
	    setInterval(function(){
		if (_timer_run == 0) { return; }
		_images_current++;
		if (_images_current >= _images.length) { _images_current = 0; }
		show_image(_images[_images_current], 1000);
	    }, 5000);	
	});
