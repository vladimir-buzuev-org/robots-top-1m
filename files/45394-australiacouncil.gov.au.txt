				// wraps a fancybox link around all content images 26/02/2015
				$('#main-content .content-body p img').wrap(function() {
					if ($(this).parent().attr('href').indexOf('vimeo') > 0) {
						 $(this).parent().attr('target', '_blank');
						 $(this).css('cursor', 'pointer');
						 return "";
					}
					if ($(this).attr('alt') == undefined) {
					    $(this).attr('alt') = $('h1').eq(0).text() + ' Content Image';
					}
					// Don't wrap if the image parent is already an anchor tag  27/06/2017
					if ($(this).parent().is('a')) return "";
					
					return "<a href='" + $(this).attr('src') + "' class='fancybox-img'></a>";
				});

				// loads content images into a fancybox
				$('#main-content .content-body p a.fancybox-img').fancybox({
					'transitionIn'	:	'elastic',
					'transitionOut'	:	'elastic',
					'speedIn'		:	600,
					'speedOut'		:	200,
					'padding'		:	5
				});

                    var ext = $(this).attr('href').split('.').pop().toUpperCase();
                    ga('send', 'event', ext, 'Download', $(this).attr('href'));
                });

                $('li.print > a').on('click', function() {
                    var ext = $(this).attr('href').split('.').pop().toUpperCase();
                    ga('send', 'event', ext, 'Print', $(this).attr('href'));
                });

				$('#main-content img').each(function() {
					var alt = $(this).attr('alt');

					if (alt == undefined) {
					    alt = 'Content Image';
					} else if (!alt.endsWith('Image')) {
						alt += ' Image';
					}

					$(this).attr('alt', alt);
				});

				$('iframe').each(function() {
					var title = $(this).attr('title');

					if (title == undefined) {
					    title = 'Inline Frame';
					}

					$(this).attr('title', title);
				});

				$('h1,h2,h3,h4,h5,h6,#main-content a').each(function() {
					if ($(this).attr('href') && $(this).attr('href').indexOf('vimeo') == -1 && $(this).attr('href').indexOf('youtube') == -1 && $(this).text().replace(' ','').replace(/(\r\n|\n|\r)/gm,"") == '') {
						$(this).remove();
					}
				});

				$('button').attr('title', 'button');

                $('input, select, textarea').each(function() {
                	$(this).attr('title', $(this).attr('name'));
                });

                $('table').each(function(i) {
                	$(this).attr('id', $(this).attr('id') + '-' + i);
                });
