function validaSenha(){
	s = document.FormLogin;
	if (s.email.value == ""){
		alert("E-mail deve ser preenchido!");
		s.email.focus();
		return false;
	}
	document.getElementById('link_senha').innerHTML = "aguarde...";
	var dados = jQuery( this ).serialize();
	jQuery.ajax({
	type: "GET",
	url: "http://www.web5.com.br/formularios.php?tipo=senha&email="+s.email.value,
	data: dados,
	success: function( data )
	{
	if (data==1) {
		swal({   
			title: "Verifique seu e-mail!",
			text: "sua nova senha de acesso foi enviada para o seu e-mail...",
			type: "success",
			timer: 3500,
			showConfirmButton: false
		});			
	} else {
		swal({   
			title: "E-mail nao encontrado!",
			text: "Tente novamente...",
			type: "error",
			timer: 3500,
			showConfirmButton: false
		});			
	}
	document.getElementById('link_senha').innerHTML = "clique aqui";
	s.email.focus();
	}
	});
	return false;
}
