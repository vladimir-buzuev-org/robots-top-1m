		function getStatusCountfromGAE(socialMedias, socialHitTextString){

			$.ajax({
		        url: 'https://go-prod-social-media-dot-go-social-media.appspot.com/_ah/api/socialmediaservice/v1/statuscount',
		        type: 'GET',
				data: { json:socialMedias },
		        async: true,
		        success: function (data) {
		            var result = data;

	            	var res = socialHitTextString.substring(1, socialHitTextString.length - 1);
	            	res = res.split(", ");

		            for(i = 0; i < result.items.length; i++){		            	
		            	var item = result.items[i].socialName + '_' + result.items[i].socialKey;
		            	var hitText = "";
		         		            	
		            	for(j = 0; j < res.length; j++){
		            		var hitTextArray = res[j].split("=");
		            		if(hitTextArray[0] == item){
		            			hitText = hitTextArray[1];
		            		}
		            	}
		       		            	
		            	var statsCount = result.items[i].statsCount ? numberWithCommas(result.items[i].statsCount) : "";		            	
		            	hitText = hitText != "null" ? hitText : "";
		            	var content = statsCount + "<span>" + hitText + "</span>";
		            	$("[id="+ escapeStr(item) +"]").empty();
		            	$("[id="+ escapeStr(item) +"]").append(content);	
		            }
		        }
		    });
		}
		
		function escapeStr(str) {
		    if (str) {
		        return str.replace(/[!"#$%&'()*+,.\/:;<=>?@[\\\]^`{|}~]/g, "\\\$&")
		    }

		    return str;
		}
		
		function numberWithCommas(x) {
