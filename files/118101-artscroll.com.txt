    function checkEmail(email) {
        var lastAtPos = email.lastIndexOf('@');
        var lastDotPos = email.lastIndexOf('.');
        return (lastAtPos < lastDotPos && lastAtPos > 0 && email.indexOf('@@') == -1 && lastDotPos > 2 && (email.length - lastDotPos) > 2);
