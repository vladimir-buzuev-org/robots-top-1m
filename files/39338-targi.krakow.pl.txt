        function setPagesHtml() {

        function setPagesVisibility(page, pagesCount) {
            if (page == 1) {
                $container.find('.article-pager .prev-page').css('visibility', 'hidden');
            } else {
                $container.find('.article-pager .prev-page').css('visibility', 'visible');
            }
            if (page == pagesCount || !pagesCount) {
                $container.find('.article-pager .next-page').css('visibility', 'hidden');
            } else {
                $container.find('.article-pager .next-page').css('visibility', 'visible');
            }

            $container.find('.article-pager').find('.dots').remove();
            if (pagesCount > 7) {
                var pagesBeforeCurrentPage = 4 - Math.min(pagesCount - page, 2),
                    pagesAfterCurrentPage = 4 - Math.min(page - 1, 2),
                    removeSecondPageBefore = (page == pagesCount - 3 ? 1 : 0),
                    removeSecondPageAfter = (page == 4 ? 1 : 0),
                    $dots = $('<span class="dots">...</span>');
                for (var i = 1; i <= pagesCount; i++) {
                    if (
                        (
                            i < page - pagesBeforeCurrentPage + removeSecondPageBefore ||
                            i > page + pagesAfterCurrentPage - removeSecondPageAfter
                        ) &&
                        i != 1 &&
                        i != pagesCount
                    ) {
                        $container.find('.article-pager .page[data-page="' + i + '"]').hide();
                    }
                    if (
                        i == page + pagesAfterCurrentPage &&
                        page + pagesAfterCurrentPage + 1 < pagesCount
                    ) {
                        $container.find('.article-pager .page[data-page="' + i + '"]').after($dots.clone());
                    }
                    if (
                        i == page - pagesBeforeCurrentPage &&
                        page - pagesAfterCurrentPage - 1 > 1
                    ) {
                        $container.find('.article-pager .page[data-page="' + i + '"]').before($dots.clone());
                    }
                }
            }
        }
