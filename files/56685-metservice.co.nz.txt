var REDIRECT_TO_MOBILE_COOKIE = "redirect_to_mobile";
var MOBILE_URL = 'http://m.metservice.com';
function getCookie(cookieName)
{
var i, x, y, ARRcookies = document.cookie.split(";");
for (i = 0; i < ARRcookies.length; i++) {
try {
x = ARRcookies[i].substr(0, ARRcookies[i].indexOf("="));
y = ARRcookies[i].substr(ARRcookies[i].indexOf("=") + 1);
x = x.replace(/^\s+|\s+$/g, "");
if (x == cookieName) {
return unescape(y);
}
} catch (err) {
log("There was a problem reading the cookies:- " + err.message);
}
}
}
function setCookie(cookieName, value, numDaysBeforeExpire) {
try {
var exdate = new Date();
exdate.setDate(exdate.getDate() + numDaysBeforeExpire);
var expiryDateStr = exdate.toGMTString();
document.cookie = cookieName + '=' + value + '; expires=' + expiryDateStr + '; path=/';
} catch (err) {
log("There was a problem setting cookies:- " + err.message);
}
}
function cookiesAreEnabled() {
var testValue = new Date().toGMTString().replace(/[ ,]/g, '-'); // Guarantees a new value each time, requires dash
var TEST_COOKIE_NAME = 'testCookies';
setCookie(TEST_COOKIE_NAME, testValue, 1);
var retrievedCookie = getCookie(TEST_COOKIE_NAME);
var enabled = (retrievedCookie == testValue);
return enabled;
}

var isMobile = isMobileBrowser;
if (isMobile) {
