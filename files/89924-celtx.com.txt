	function checkFormSectionValidity($form, $section){
		var valid = true;
		$req = $section.find('input[required], select[required], textarea[required]');
		$req.each(function(){
			if(!this.validity.valid){
				valid = false;
				$(this).addClass('input-error');
				$(this).one('change', function(){
					$(this).removeClass('input-error');
				});
			}
		});

		if(!valid){
			//trigger validation popups

			$form.one('submit', function(e){
				e.preventDefault();
			});

			$inputSubmit = $form.find('input[type="submit"]');

			if($inputSubmit.length>0){
				$inputSubmit.click();
			} else {
				$form.submit();
			}
		}

		return valid;
	}

		validate = function(user, pass, messageEl){
			
			var hasError = false;
			
			if (pass.val().length <= 0){
				pass.addClass('error');
				pass.focus();
				showError(messageEl, "Please enter your password.");
				hasError= true;

		setupFormLogin = function(userTb, passTb, submitBtn, errorSpanId){

			userTb.bind('keypress', function(){
				$(this).removeClass('error');
				hideError(errorSpanId);
			});
			
			passTb.bind('keypress', function(){
				$(this).removeClass('error');
				hideError(errorSpanId);
			});

			var inputList = [userTb, passTb, submitBtn];

			submitBtn.click(function(e){

				// stop the submission of the form.
				e.preventDefault();
				
				hideError(errorSpanId);
				
				if (validate(userTb, passTb, errorSpanId)){
					return;
				}
				
				updateInputs(inputList, true);
			
				cx.auth.Session.login(userTb.val(), passTb.val())
				.addCallback(function(){
					goToApplication();
				})
				.addErrback(function(e){
					showError(errorSpanId, e.message, userTb);
					updateInputs(inputList, false);
				});
			});

		setupForm = function(form, userTb, passTb, fName, lName, city, country, jobTitle, companyName, teamSize, phone, submitBtn, errorSpanId, spinnerId, isLogin, opt_promoInput){

			//special case for invtied people
			if(!isLogin && $.cookie('inviteduser') === 'yes' && $.cookie('cx_session')){
				var cxm = cx.module.Manager.getInstance();
        cxm.load("cxdocs", function() {

          cx.auth.Session.getInstance().getLogon(cx.docs.Docs).addCallback(function(logon){
            userTb.val(logon.getName());

            userTb.attr('disabled', 'true');

            if(opt_promoInput && opt_promoInput.length>0){
              opt_promoInput.addClass('disabled');
              opt_promoInput.attr('disabled', 'true');

            userTb.siblings('label.placeholder-input').hide();

          });
        });

			userTb.bind('keypress', function(){
				$(this).removeClass('error');
				hideError(errorSpanId);

			passTb.bind('keypress', function(){
				$(this).removeClass('error');
				hideError(errorSpanId);
			});
			
			var inputList = [userTb, passTb, submitBtn, fName, lName, city, country, jobTitle, companyName, teamSize, phone];

			if(opt_promoInput && opt_promoInput.length > 0){
				inputList.push(opt_promoInput);

				opt_promoInput.bind('keypress', function(){
					$(this).removeClass('error');
					hideError(errorSpanId);
				});
			}
				
			form.submit(function(e){
				//prevent duplicate submissions, by not allowing more than one every 500 milliseconds
				if(window.formSubmitted)
					return;
				window.formSubmitted = true;
				window.setTimeout(function() {window.formSubmitted = false}, 500);

				// stop the submission of the form.
				e.preventDefault();

				// userTb.val( userTb.val().toLowerCase() );
				
				hideError(errorSpanId);
				

				var valid = checkFormSectionValidity(form, form);

				if(!valid){
					return;
				}
				
				updateInputs(inputList, true);
			
				if (isLogin){
					cx.auth.Session.login(userTb.val(), passTb.val())
					.addCallback(function(){
						goToApplication();
					})
					.addErrback(function(e){
						showError(errorSpanId, e.message, userTb);
						updateInputs(inputList, false);
					});
				}
				else{
					spinnerId.css('visibility', 'visible');

					//special case for invtied people
					if($.cookie('inviteduser') === 'yes' && $.cookie('cx_session')){
						cx.auth.Session.updatePassword(passTb.val())
						.addErrback(function(e){
							showError(errorSpanId, e.message, userTb);
						})
						.addCallback(function(){
							$.removeCookie("inviteduser", 'yes');
							window.location.href = '/a/ux/index#getting-started';

							userTb.val(),
							passTb.val(),

							lName.val(), 
							city.val(), 
							country.val(), 
							jobTitle.val(), 
							companyName.val(), 
							teamSize.val(), 
							phone.val(),
							promocode)
						.addCallback(function(){
							spinnerId.css('visibility', 'hidden');
							signupSuccess(userTb.val());		
						})
						.addErrback(function(e){
							showError(errorSpanId, e.message, userTb, opt_promoInput);
							updateInputs(inputList, false);
							spinnerId.css('visibility', 'hidden');
						});

					}
				}

// Place the signup button in the header directly above the signup button in the dropdown or else according to window size
function placeSignUpBtn() {

	var signupPos = document.getElementById("btn-signup").offsetLeft - $(window).scrollLeft()
	if ($(window).width() > 804) {
		$("#signup-btn-header").css("left", signupPos+100);
		$("#signup-btn-header").css("right", "auto");
	}
	else {
		$("#signup-btn-header").css("left", "auto");
		$("#signup-btn-header").css("right", "15px");
	}
}

// When the sign up fields have disappeared but the nav bar is not yet fixed, clicking the sign up button will scroll to top
function scrollUp() {
	window.scrollTo(0, 0);
	var banner = document.getElementById("signup-banner");
	banner.getElementsByClassName("field-to-focus")[0].focus();
}

// When sign up button in the nav bar is clicked, do a drop down and fade in sign up fields
function dropHeader() {

	if ($(window).scrollTop() >= ($('.banner').height() - $('#topBar').height())) {
      if ($(window).width() < 400) {
			$("#signup-header").css("position", "absolute");
			$("#signup-header").css("width", "auto");
		}

		$("#signup-btn-header").hide();
      $("#loginLink").show();

		if ($(window).width() < 804) {
			$(".topBar-fixed").css("height", "475px");
		}
		else {
			$(".topBar-fixed").css("height", "347px");
		}

	 	$("#signup-header").removeClass("signup-header-hidden");
      $("#loginLink").show();

		// At the start of a scroll, ensure the sign up button in the nav bar is once again shown
		$(window).scroll(function () {
			$("#signup-btn-header").show();
		});
	}
	else {
    	scrollUp();
	}

}

<script>

var fadeDistance = 50;

// Fade out header text on scroll
function fadeHeaderText() {

  	var fadeElements = document.getElementsByClassName('fade-header');

  	for (var i = 0; i < fadeElements.length; i++) {

  		var x = fadeElements[i].offsetTop - $(window).scrollTop();

	  	if (x < fadeDistance) {
	  		fadeElements[i].style.opacity = "0";
	  	}
	  	else {
	  		fadeElements[i].style.opacity = "1";
	  	}
  	}
}

// Fade out signup fields on scroll and show sign up button in fixed header instead
function fadeSignup() {

	var signup = document.getElementById("signup-banner");
  	var fadeElements = signup.getElementsByClassName('fade-signup');

  	for (var i = 0; i < fadeElements.length; i++) {

	  	if ((signup.offsetTop - $(window).scrollTop()) < fadeDistance) {
	  		fadeElements[i].style.opacity = "0";
	  	}
	  	else {
	  		fadeElements[i].style.opacity = "1";
	  	}
  	}

	var fadeButton = signup.getElementsByClassName('btn-signup');

	if ((signup.offsetTop - $(window).scrollTop()) < fadeDistance) {
		$("#signup-btn-header").show();
		fadeButton[0].style.opacity = "0";
	} 
	else {
		$("#signup-btn-header").hide();	
		fadeButton[0].style.opacity = "1";
	}
}

			$activeQuote.removeClass('quotecarousel-item-active');

			$activeQuote.removeClass('quotecarousel-item-active');

			if($prevQuote.prev('.quotecarousel-item').length <= 0){
