                    }

                });

                    var sync2visible = slideA_thumb.data("owlCarousel").owl.visibleItems;
                    var num = number;
                    var found = false;
                    for(var i in sync2visible){
                        if(num === sync2visible[i]){
                            found = true;
                        }
                    }
                    if(found===false){
                        if(num>sync2visible[sync2visible.length-1]){
                            slideA_thumb.trigger("owl.goTo", num - sync2visible.length+2)
                        }else{
                            if(num - 1 === -1){
                                num = 0;
                            }
                            slideA_thumb.trigger("owl.goTo", num);
                        }
                    } else if(num === sync2visible[sync2visible.length-1]){
                        slideA_thumb.trigger("owl.goTo", sync2visible[1])
                    } else if(num === sync2visible[0]){
                        slideA_thumb.trigger("owl.goTo", num-1)
                    }
                }
            });

            function convertToRGBA(rgb,opacity){
               var digits = /(.*?)rgb\((\d+), (\d+), (\d+)\)/.exec(rgb),
                   red = parseInt(digits[2]),
                   green = parseInt(digits[3]),
                   blue = parseInt(digits[4]),
                   result = 'rgba('+red+','+green+','+blue+','+opacity+')';
              return result;
