    <script type="text/javascript">
      (function() {
          var timeout = 30;
          var second = 1000;
          var message;
          var button = document.getElementById("continue");

          function messageText(value) {
              var m = "Automatically redirecting ";

              if (value == 0) {
                  m += "now.";
              } else if (value == 1) {
                  m += "in 1 second.";
              } else {
                  m += "in " + value + " seconds.";
              }

              return document.createTextNode(m);
          }

          function tick() {
              --timeout;
              message.replaceChild(messageText(timeout), message.firstChild);

              if (timeout == 0) {
                  window.location.href = button.href;
              } else {
                  setTimeout(tick, second);
              }
          }

          if (button) {
              message = document.createElement("p");
              message.className = "redirect-message";
              message.appendChild(messageText(timeout));
              button.parentNode.appendChild(message);

              setTimeout(tick, second);
          }
      })();
    </script>
