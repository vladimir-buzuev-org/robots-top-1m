                    document.cookie = "CookieSessionAuthentication=" + result;
                    //$(".nav-ebit").load(location.href + " .nav-ebit>*", "");

                    if (sessionAuthentication.ResponseDescription.toLowerCase() == "access denied") {

                }
            }
        });
