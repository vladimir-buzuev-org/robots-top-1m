function getPropertyValueFromCss(className, propertyName)
{
    var toReturn = null;
    
	if (document.styleSheets) {	
    for (var i = 0; i < document.styleSheets.length; i++)
    {
        var cssRules = null;
        // IE
        if (document.styleSheets[i].rules) {
            cssRules = document.styleSheets[i].rules;
        }
        // Mozilla and others
        else if (document.styleSheets[i].cssRules) {
            cssRules = document.styleSheets[i].cssRules;
        }
        if (cssRules != null) {
            for (var j = 0; j < cssRules.length; j++) {
                if (cssRules[j].selectorText == ("." + className) && cssRules[j].style[propertyName] != null)
                {
                    toReturn = cssRules[j].style[propertyName];
                }
            }
        }
    }
    }
    return toReturn;
}

function skm_closeSubMenus(parent){
	if (skm_OpenMenuItems == "undefined") return;
	for (var i=skm_OpenMenuItems.length-1; i>-1;i--) {
		if (parent.id.indexOf(skm_OpenMenuItems[i]) != 0) {
			if (document.getElementById(skm_OpenMenuItems[i]) != null) {
				document.getElementById(skm_OpenMenuItems[i]).style.display = 'none';
				skm_shimSetVisibility(false, skm_OpenMenuItems[i]);			
				skm_OpenMenuItems = new Array().concat(skm_OpenMenuItems.slice(0,i), skm_OpenMenuItems.slice(i+1));
			}
  		} 
	}
}
function skm_shimSetVisibility(makevisible, tableid){
	var tblRef=document.getElementById(tableid);
	var IfrRef=document.getElementById('shim'+tableid);
	if (tblRef!=null && IfrRef!=null){
		if(makevisible){
			IfrRef.style.width=tblRef.offsetWidth;
			IfrRef.style.height=tblRef.offsetHeight;
			IfrRef.style.top=tblRef.style.top;
			IfrRef.style.left=tblRef.style.left;
			IfrRef.style.zIndex=tblRef.style.zIndex-1;
			IfrRef.style.display="block";
		}else{
			IfrRef.style.display="none";
		}
	}
}
function skm_IsSubMenu(id){
	if (skm_subMenuIDs == "undefined") return false;
	for (var i=0;i<skm_subMenuIDs.length;i++)

function skm_getAscendingLefts(elem){
	if (elem==null)
		return 0;
	else
	{
		var elemPosition = getPropertyValueFromCss(elem.className,'position');
		if (!elemPosition) {
			elemPosition = elem.style.position;
		}		
		if ((elemPosition=='absolute' || elemPosition=='relative') && !skm_IsSubMenu(elem.id)) return 0;
		var x = elem.offsetLeft+skm_getAscendingLefts(elem.offsetParent);
		return x;
	}
}
function skm_getElementLeft(elem)
{
	if (elem==null)
	{
		return 0;
	} else {
		return elem.offsetLeft + skm_getElementLeft(elem.offsetParent);
	}	
}
function skm_getAscendingTops(elem){
	if (elem==null)
		return 0;
	else {	
		var elemPosition = getPropertyValueFromCss(elem.className,'position');
		if (!elemPosition) {
			elemPosition = elem.style.position;
		}
		var parent = skm_getAscendingTops(elem.offsetParent);		
		//alert(elem.tagName +' '+ elem.offsetTop + ' from '+ elem.offsetParent +' '+ parent + ' elempos: '+ elemPosition);		
		if ((elemPosition=='absolute' || elemPosition=='relative') && !skm_IsSubMenu(elem.id)) return 0;
		return elem.offsetTop + parent;
	}
}

function skm_doTick(menuID){
	if (skm_clockValue>=skm_MenuFadeDelays[menuID]){
		skm_stopTick();
		skm_closeSubMenus(document.getElementById(menuID));
	} else {
		skm_clockValue++;
		skm_ticker=setTimeout("skm_doTick('"+menuID+"');", 500);
	}
}
function skm_stopTick(){
	skm_clockValue=0;
	clearTimeout(skm_ticker);
}

        <div class='footer-text'><div class='container'><div class='row'><div class='col-lg-12 col-md-12 col-sm-12 col-xs-12'><h1>Affiliate marketing als strategie en verdienmodel</h1>

<p>Beheer je een website, forum of schrijf je een blog, en wil je daarmee geld verdienen? Of heb je een online shop en wil je de omzet vergroten? Dan is het goed dat je de weg naar affilinet hebt gevonden. affilinet is één van de meest vooraanstaande affiliate netwerken van Nederland en kan jou helpen met succesvolle affiliate marketing. Affiliate marketing is een vorm van performance marketing, wat inhoudt dat online marketing-diensten worden gemeten aan de geleverde prestaties (performance) en op basis daarvan worden vergoed. Het is een overeenkomst tussen twee partijen: de adverteerder, de eigenaar van een online shop of ander bedrijf, en de publisher, de beheerder van een informatie-website, forum of blog. De publisher plaatst op zijn pagina een link naar de website van de adverteerder, die daarvoor per bezoeker of verkocht product betaalt.</p>

<h2>Bij affiliate marketing zijn er verschillende verdienmodellen:</h2>

<p>Bij &#39;pay per click&#39; (PPC) betaalt de adverteerder per klik een vantevoren vastgelegd bedrag aan de publisher, bij &#39;pay per sale&#39; (PPS) per verkocht product en bij &#39;pay per lead&#39; (PPL) per aanmelding of afgesloten abonnement. Voor publishers heeft affiliate marketing veel voordelen: ze hoeven alleen een link op hun website te plaatsen en krijgen vanzelf geld binnen als hun bezoekers die link volgen. Maar ook voor adverteerders is affiliate marketing een voordelig model, omdat ze alleen betalen als er ook daadwerkelijk iets is verkocht of als ze meer bezoekers krijgen. Ze kunnen zo precies bepalen hoeveel een verkocht product of klik waard is, en daarmee hun marketing-uitgaven sturen.</p></div></div></div></div><div class='container'><div class='accordion-service'><h3><span class='ui-accordion-header-icon'></span>Wat doen affiliate netwerken precies?</h3><div><p>Om met een website genoeg geld te verdienen heeft een publisher een groot aantal adverteerders nodig. Omgekeerd geldt voor adverteerders dat het te veel werk zou zijn om zelf alle publishers te benaderen. Affiliate netwerken bieden hier de oplossing door de contacten tussen adverteerders en publishers tot stand te brengen. Als publisher kun je aan affiliate programma&#39;s in de branche van je keuze deelnemen, en heb je zo toegang tot adverteerders uit het topsegment. Als adverteerder kun je een partnerprogramma opbouwen, waar publishers vervolgens aan deel kunnen nemen. Daarbij maakt het voor affilinet niet uit of je je webshop nog aan het opbouwen bent, of een al gevestigd merk vertegenwoordigt.</p>

<p>Een affiliate netwerk biedt daarnaast veel services die het vinden en beheren van contacten tussen publishers en adverteerders makkelijker maken. Onze consultants gaan voor je op zoek naar creatieve oplossingen voor het integreren van links op jouw website, van banners tot video ads. Ze gaan in op individuele wensen en doelen, maar verliezen daarbij de cijfers en statistieken niet uit het oog. Bovendien zijn ze op de hoogte van de technische eisen, bijvoorbeeld voor affiliate marketing op mobiele websites. Wij regelen de facturatie, stellen de nodige statistieken ter beschikking en hebben handige tools, waarmee je bijvoorbeeld actiecode-programma&acute;s kunt vinden, of als adverteer <a href="/nl/adverteerder/tools/match-master">publishers kunt zoeken</a> die bij jouw programma passen. Bovendien stellen affiliate netwerken zoals affilinet tracking software ter beschikking, die meet hoeveel mensen via de website van de publisher op de website van de adverteerder belanden, of daar iets kopen.Meer informatie over tracking en cookies vind je in de <a href="/nl/kenniszone/begrippenlijst">begrippenlijst</a> en in onze <a href="/nl/kenniszone/cookies">kenniszone</a>.</p></div><h3><span class='ui-accordion-header-icon'></span>Waarom kiezen voor affiliate marketing via affilinet?</h3><div><p>Beheer je een blog, website, prijsvergelijkingssite, forum of verstuur je regelmatig een nieuwsbrief per e-mail? Wij begeleiden je gedurende het hele proces van affiliate marketing. Wij werken internationaal samen met meer dan 2.500 adverteerders, waardoor we ook voor niche-websites de juiste aanspreekpartner zijn. Van inductieplaten tot duikmateriaal: wij vinden voor elk onderwerp de juiste oplossing. affilinet neemt je het tracken van je websitebezoekers uit handen en stelt statistieken en rapporten ter beschikking, die je helpen om je activiteiten te analyseren en optimaliseren. Ten slotte is affilinet veilig en transparant: een eigen team controleert de kwaliteit van het netwerk en de bij ons aangesloten publishers regelmatig. Meer weten? Kijk dan <a href="/nl/publisher">hier</a>.</p>

<p>Van kleine online shops tot grote bedrijven kan ook iedere adverteerder bij ons terecht. Er zijn een aantal branches waar we in uitblinken: als je een bedrijf in travel, parking, ticketing, retail in fashion of retail in living hebt, is affilinet het beste affiliate netwerk voor jou. Verkoop je je producten niet via internet? Affiliate marketing kan dan toch een oplossing voor je zijn, bijvoorbeeld door het creëren van leads naar een website met informatie over jouw product. We hebben internationaal meer dan 700.000 publishers bij wie we jouw ads kunnen plaatsen. Naast Duitsland, waar het hoofdkantoor zit, zijn we met onze dochterondernemingen voor affiliate marketing in Nederland, België, Engeland, Frankrijk, Spanje, Oostenrijk en Zwitserland actief. Meer informatie vind je <a href="/nl/adverteerder">hier</a>.</p></div></div></div>
