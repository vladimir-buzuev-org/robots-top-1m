		var collapseIt = function(collapse) {
			if (collapse) {
				var currLi;
					
				setTimeout(function(){ 
					listItms[currCount].style.display = "none";
					if (currCount > 1) {
						currCount--;
						collapseIt(collapse);
					} else {
						currCount = 1;
					}
				}, 11);
				
			} else {
				setTimeout(function(){ 
					listItms[currCount].style.display = "block"; 
					if (currCount < listItms.length - 1) {
						currCount++;
						collapseIt(collapse);
					} else {
						currCount = listItms.length - 1;
					}
				}, 11);
			}
		}

			if (menuBtn.style.display != "none") {

			}
