var editPrice = function(options) {
    if (isEmpty(options) || !$.isNumeric(options.price)) {
        return "0";
    }
    var price = options.price.toString();
    return price.replace(/(\d)(?=(\d{3})+(?!\d))/g, "$1,");
};
var displayCustomerStatus = function(options) {

        },

            if (res.status == "0") {
                var render = {};
                if (!isEmpty(res.results.cust_name)) {
                    render.ssl_domain_url = "https://www.dhc.co.jp";
                    render.my_url = "https://www.dhc.co.jp/main/main.jsp";
                    render.cust_name = res.results.cust_name;
                    render.login_status = res.login_status;
                    render.skin_rank_name = res.results.skin_rank_name;
                    render.skin_rank_dc_rate = res.results.skin_rank_dc_rate;
                    render.net_point = res.results.net_point;
                    if (!$.isEmptyObject(res.results.vip_customer_list)
                            || res.results.exists_cust_discount_dhc_card == 1
                            || res.results.exists_cust_discount_other == 1
                            || !$.isEmptyObject(res.results.individual_goods_dhc_card_list)
                            || !$.isEmptyObject(res.results.individual_goods_aparel_list)
                            || !$.isEmptyObject(res.results.individual_goods_other_list)

                            || res.results.exists_thanks_coupon == 1
                            || res.results.exists_customer_coupon == 1) {
                        render.exists_privilege = true;
                    } else {
                        render.exists_privilege = false;
                    }
                    render.vip_customer_list = res.results.vip_customer_list;
                    render.exists_cust_discount_dhc_card = res.results.exists_cust_discount_dhc_card;
                    render.exists_cust_discount_other = res.results.exists_cust_discount_other;
                    render.individual_goods_dhc_card_list = res.results.individual_goods_dhc_card_list;
                    render.individual_goods_aparel_list = res.results.individual_goods_aparel_list;
                    render.individual_goods_other_list = res.results.individual_goods_other_list;
                    render.exists_birthday_discount = res.results.exists_birthday_discount;
                    render.exists_thanks_coupon = res.results.exists_thanks_coupon;
                    render.exists_customer_coupon = res.results.exists_customer_coupon;
                    if (!$.isEmptyObject(res.results.information_list)) {
                        render.exists_information = true;
                    } else {
                        render.exists_information = false;
                    }
                    render.information_list = res.results.information_list;
                    render.usable_thanks_coupon_list = new Array();
                    $.each(res.results.usable_thanks_coupon_list, function(key, value) {
                        var tmp = {};
                        tmp.coupon_mng_name = value.coupon_mng_name;
                        if (value.discount_type == 0) {
                            tmp.discount_type_name = editPrice({price: value.price}) + "円OFF";
                        } else {
                            tmp.discount_type_name = value.rate + "％OFF";
                        }
                        render.usable_thanks_coupon_list.push(tmp);
                    });
                    render.usable_customer_coupon_list = new Array();
