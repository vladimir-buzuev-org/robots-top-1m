function createRightBanner(ad)
{
    var img = document.createElement("img");
    var link = document.createElement("a");
    link.href = ad.clickUrl;

    if (ad.creative.images && ad.creative.images[0]) {
        //コンボクリエイティブの場合
        if (ad.creative.content) {
            for (contentIndex in ad.creative.content) {
                //オプションが設定されている場合
                switch (ad.creative.content[contentIndex].key) {
                case "target":
                    //管理画面でtargetの値を登録しないとcontent属性値が空配列になることに注意！
                    link.target = ad.creative.content[contentIndex].value;
                    break;
                case "説明":
                    //説明が登録されていたら代替テキストをセット
                    img.alt = ad.creative.content[contentIndex].value;
                    break;
                default:
                }
            }
        }

        img.src = ad.creative.images[0].source;
    } else {
        //イメージクリエイティブの場合
        img.src = ad.creative.source;

        if (ad.creative.description) {
            //説明が登録されていたら代替テキストをセット
            img.alt = ad.creative.description;
        }
    }

    link.style.margin = "0px 3px 12px 3px";
    link.style.display = "block";
    link.appendChild(img);
    return link;
}

function renderRightAd(data, context) {
    if (isRightAd) return;
    if (data && data.searchResult && data.searchResult.spaces && data.searchResult.spaces[0] && data.searchResult.spaces[0].ads) {
        //広告スペースが存在する場合は広告オブジェクトを取得
        var ads = data.searchResult.spaces[0].ads;
        if (ads[0]) {
            //上バナー
            document.querySelector("#right-ad-container").appendChild(createRightBanner(ads[0]));
            isRightAd = true;
        }
        if (ads[1]) {
            //下バナー
            document.querySelector("#right-ad-container").appendChild(createRightBanner(ads[1]));
            isRightAd = true;
        }
    } else {
        //登録バナーが足りないときやアドサーバ側のバグによって意図する広告を取得できなかった場合
