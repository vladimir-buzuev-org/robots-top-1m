<script>
    jQuery.noConflict();
    jQuery(document).ready(function ($) {
        var termTemplate = "<span style='font-weight:bold;'>%s</span>";
        var urlauto = "https://search.csiro.au/suggest?q=_query_&max=10&site=Main&client=data61_frontend&access=p&format=rich";
        var urlsearch = "https://search.csiro.au/search?q=_query_&btnG=Google%2BSearch&client=data61_frontend&output=xml_no_dtd&proxystylesheet=data61_frontend&proxyreload=0&sort=date%253AD%253AL%253Ad1&wc=200&wc_mc=1&oe=UTF-8&ie=UTF-8&ud=1&exclude_apps=1&site=Data61&filter=0&getfields=*&sourcepage={6EDF11D8-8EED-4B67-94F2-FBDFD65766B2}";        
        $(".searchbtn").click(function () {
            dosearch($(this).closest(".fn-search").find(".search-term").val());
        });
        $(".search-term").keypress(function (e) {
            if (e.which == 13) {
                dosearch($(this).val());
            }
        });
        function dosearch(query) {
            if (query == "") { return false;}
            window.location.href = urlsearch.replace("_query_", query).replace("_meta_", query);
        }
        //**** autocomplete on the search box
        $(".search-term").autocomplete({

            source: function (request, response) {
                $.ajax({
                    dataType: "json",
                    type: 'get',
                    url: "/api/sitecore/AutoComplete/Get",
                    data: { url: urlauto.replace("_query_", request.term) },
                    success: function (data) {
                        response(data.terms);
                    },
                    error: function (data) {
                    }
                });
            },
            open: function (e, ui) {
                var
                    acData = $(this).data('ui-autocomplete'),
                    styledTerm = termTemplate.replace('%s', acData.term);

                acData
                    .menu
                    .element
                    .find('li')
                    .each(function () {
                        var me = $(this);
                        me.html(me.text().replace(acData.term, styledTerm));
                    });
            },
            select: function (event, ui) {
                $(this).closest(".fn-search").find(".search-term").val(ui.item.value);
                dosearch(ui.item.value);
            },
            minLength: 2,
            autoFocus: false
        });

    });

</script>

<script>
    jQuery.noConflict();
    jQuery(document).ready(function ($) {
        function validateEmailSubscribe(email) {
            var re = /\S+@\S+\.\S+/;
            return re.test(email);
        }
        // subscribe methods
        $("#sn_email").keyup(function (e) {
            if (e.keyCode == 13) {
                subscribe();
            }
        });
        window.subscribe = function () {
            var email = $("#sn_email").val();
            if (!validateEmailSubscribe(email)) {
                $("#sn_emailvalidate").show();
                $("#sn_form").addClass("error");
                return;
            }
            $("#sn_emailvalidate").hide();
            $("#sn_form").removeClass("error");
            $("#sn_form").hide();
            $("#sn_btn").disabled = true;
            $("#sn_sending").show();
            $.ajax({
                cache: false,
                type: "GET",
                url: "//www.csiro.au/api/sitecore/Subscribe/Snapshot",
                data: { fromemail: email },
                success: function (data) {
                    if (data == "OK") {
                        $("#sn_sending").hide();
                        $("#sn_success").show();
                        if (typeof ga === 'function') {
                            ga('send', 'event', 'Snapshot Data61 footer subscribe', 'click', 'Snapshot Data61 footer subscribe');
                        }
                    }
                    else {
                        $("#sn_sending").hide();
                        $("#sn_success").hide();
                        $("#sn_form").show();
                        $("#sn_form").addClass("error");
                        $("#sn_btn").disabled = false;
                        if (data == "email") {
                            $("#sn_emailvalidate").show();
                        }
                        else {
                            $("#sn_error").show();
                        }
                    }
                },
                error: function (xhr, ajaxOptions, thrownError) {
                    $("#sn_form").addClass("error");
                    $("#sn_form").show();
                    $("#sn_error").show();
                    $("#sn_btn").disabled = false;
                }
            });
            return false;
        }
    });
</script>

<script>
    jQuery.noConflict();
    jQuery(document).ready(function ($) {
        
        function enableBtn() {
            $("#enquirybtn").removeAttr("disabled");
            $("#enquirybtn").attr("value", "Send enquiry");
        }
        function validateEmail(email) {
            var re = /\S+@\S+\.\S+/;
            return re.test(email);
        }
        
        window.send_form = function ()
        {
            $("#enquirybtn").attr("disabled", "disabled");
            $("#enquirybtn").attr("value", "Sending...");
            var fname; var sname; var email; var phone; var enquiry; var heardabout; var url; var role; var sub;
            var valid = true;
            email = $("#enquiry-email").val();
            if (!validateEmail(email)) {
                
                $("#email-div").addClass("has-error");
                $("#email-error").attr("aria-hidden", "false");
                $("#email-error").removeClass("sr-only");
                $("#email-error").show();
                valid = false;
            }
            else {
                $("#email-div").removeClass("has-error");
                $("#email-error").attr("aria-hidden", "true");
                $("#email-error").hide();
            }
            fname = $("#enquiry-name").val();
            if (fname == "") {
                $("#name-div").addClass("has-error");
                $("#name-error").attr("aria-hidden", "false");
                $("#name-error").removeClass("sr-only");
                $("#name-error").show();
                valid = false;
            }
            else {
                $("#name-div").removeClass("has-error");
                $("#name-error").attr("aria-hidden", "true");
                $("#name-error").hide();
            }
            enquiry = $("#enquiry").val();
            if (enquiry == "") {
                $("#enquiry-div").addClass("has-error");
                $("#enquiry-error").attr("aria-hidden", "false");
                $("#enquiry-error").removeClass("sr-only");
                $("#enquiry-error").show();
                valid = false;
            }
            else {
                $("#enquiry-div").removeClass("has-error");
                $("#enquiry-error").attr("aria-hidden", "true");
                $("#enquiry-error").hide();
            }
            if (!valid) {
                enableBtn();
                return false;
            }
            else
            {
                sub = false;
                if ($("#chkSnapshot").is(":checked"))
                {
                    sub = true;
                }
                role = $("#enquiry-role").val();
                heardabout = $("#enquiry-heard").val();
                sname = $("#enquiry-surname").val();
                phone = $("#enquiry-phone").val();
                url = "http://data61.csiro.au/";
                sendtocontroller(fname, sname, email, phone, enquiry, heardabout, url, role, sub);
                return false;
            }
        }
        function sendtocontroller(fname, sname, email, phone, enquiry, heardabout, url, role, sub) {
            $.ajax({
                cache: false,
                type: "GET",
                url: "/api/sitecore/ContactForm/SendContact",
                data: { txtFirstName: fname, txtSurname: sname, txtEmail: email, txtPhone: phone, txtEnquiry: enquiry, HeardAbout: heardabout, pageurl: url, Role: role, subscribe: sub, sitename: "data61: "},
                success: function (data) {
                    if (data == "OK") {
                        $("#contact-error").hide();
                        $("#contact-form").hide();
                        $("#contact-success").removeClass("sr-only");
                        $("#contact-success").show();
                    }
                    else {
                        $("#contact-error").removeClass("sr-only");
                        $("#contact-error").show();
                        enableBtn();
                    }
                },
                error: function (xhr, ajaxOptions, thrownError) {
                    $("#contact-error").removeClass("sr-only");
                    $("#contact-error").show();
                    enableBtn();
                }
            });
        }
    });
</script>
