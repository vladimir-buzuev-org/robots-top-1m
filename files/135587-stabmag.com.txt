<script type="text/javascript">

  $(document).ready(function(){

    // Cookies
    function createCookie(name, value, hours) {
        if (hours) {
            var date = new Date();
            date.setTime(date.getTime() + (1 * hours * 60 * 60 * 1000));
            var expires = "; expires=" + date.toGMTString();
        }
        else var expires = "";

        document.cookie = name + "=" + value + expires + "; path=/";
    }

    function readCookie(name) {
        var nameEQ = name + "=";
        var ca = document.cookie.split(';');
        for (var i = 0; i < ca.length; i++) {
            var c = ca[i];
            while (c.charAt(0) == ' ') c = c.substring(1, c.length);
            if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length, c.length);
        }
        return null;
    }

    function eraseCookie(name) {
        createCookie(name, "", -1);
    }

    $(".readerPollClose").click(function() {
        // Set the Cookie so that we can open it back up once the timeout has expired
        if(!readCookie("STAB_POPUP_2147483647")) {
          createCookie('STAB_POPUP_2147483647', "STAB2147483647", 24);
        } else {
          createCookie('STAB_SLIDER_2147483647', "STAB2147483647", 24);
        }

        // Hide the popup for a set period
        hidePopup();
    });

    $(".readerPollDismiss").click(function() {
        // Set the cookie so the popup doesn't return
        createCookie('STAB_POPUP_2147483647', "STAB2147483647", 8760);
        createCookie('STAB_SLIDER_2147483647', "STAB2147483647", 8760);
    });

    function showPopup() {
      if (0) {
        $("#readerPollPopupShroud").fadeIn().css("display","inline-block");
        $("#readerPollPopup").fadeIn().css("display","inline-block");
      }
    }

    function hidePopup() {
      $("#readerPollPopup").fadeOut().css("display","none");
      $("#readerPollPopupShroud").fadeOut().css("display","none");
      hideSlider();

      // Wait for the timeout to expire then show the slider
      setTimeout(showSlider, parseInt(80) * 1000);
    }

    function showSlider() {
      if (0) {
        if (!readCookie("STAB_SLIDER_2147483647")) {
          $("#readerPollSlider").fadeIn().css("display","inline-block");
        }
      }
    }

    function hideSlider() {
      $("#readerPollSlider").fadeOut().css("display","none");
    }

    // Check for an existing non-exipred cookie
    if(0 && !readCookie("STAB_POPUP_2147483647")) {
        setTimeout(showPopup, parseInt(3) * 1000);
    } else if (!readCookie("STAB_SLIDER_2147483647")) {
        setTimeout(showSlider, parseInt(80) * 1000);
    }

  });

</script>
