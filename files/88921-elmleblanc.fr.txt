	 * Bypass the Google Geocoder for addresses which already contain latitude and longitude
	 *
	 * @autor CI/WBE1-EU-Weisser
	 * @date  2013-10-09
	 *
	 * @param dealerSearchAddress		Address containing latitude and longitude in following format: 48.0°;9.0°
	 */
	var bypassGoogleGeocoder = function(dealerSearchAddress) 
	{
		/*** Remove all ° and blanks from address, and replace all commas by dots ***/
		var dealerSearchAddressCleaned = dealerSearchAddress;
		dealerSearchAddressCleaned = dealerSearchAddressCleaned.replace(/ /g, '');
		dealerSearchAddressCleaned = dealerSearchAddressCleaned.replace(/°/g, '');
		dealerSearchAddressCleaned = dealerSearchAddressCleaned.replace(/,/g, '.');
		
		var currentAdressComponents = dealerSearchAddressCleaned.split(';');
		
		if (currentAdressComponents.length==2)
		{
			/*** Get latitude and longitude from address ***/
			var latComponent = currentAdressComponents[0];
			var longComponent = currentAdressComponents[1];
			/* console.log('Latitude: ' + latComponent); */
			/* console.log('Longitude: ' + longComponent); */

			/*** Put latitude, longitude, and complete lat/long address to search form and submit it ***/
			jQuery('#dealerSearchLatitude').val(latComponent);
			jQuery('#dealerSearchLongitude').val(longComponent);
			jQuery("#dealerSearchAdress").val(dealerSearchAddress);
			jQuery("#dealerSearchForm").submit();		
		}
	}
