    function buildAdButlerBanner(data,id){
        if (data == undefined){
            return false;
        }
        $('#'+id).html("<a><img id=\"bannerImage\" /></a>");
        $('#' + id).css({
            width: data.width,
            height: parseInt(data.height)
        });
        if (data.body !== ""){
            $('#'+id).html(data.body);
            if (data.accupixel_url){
                $('#'+id).append('<img src="'+data.accupixel_url+'" />');
            }
        }
        else if (data.image_url !== ""){
            if (data.redirect_url){
                $('#'+id+' a').attr({
                    href: data.redirect_url,
                    target: data.target
                });
            }
            $('#'+id+' #bannerImage').attr({
                'alt': data.alt_text,
                'width': data.width,
                'height': data.height,
                'src': data.image_url
            });

                function() {
                    if (data.accupixel_url) {
                        $('#' + id).append('<img src="' + data.accupixel_url + '" />');
                    }
                });
        }
        if (data.tracking_pixel){
            $('#'+id).append('<img src="'+data.tracking_pixel+'" />');
