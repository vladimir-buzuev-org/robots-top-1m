    <script type="text/javascript">
        var locs;
        $(function () {

            $(".test-type").val("-1");
            $(".country").val("-1");
            $(".location").val("-1");

            $(".dependent").prop('disabled', true);
            $(".find-button").addClass('disabled');

            $("select.test-type")
                .change(function(e) {
                    $(".dependent").prop('disabled', true);
                    $(".find-button").addClass('disabled');
                    $("select.country").val("-1");
                    $("select.location").val("-1");

                    var val = $("select.test-type").val();
                    if (val != -1) {
                        $.getJSON("/global/anonymousors/ielts/GetTestLocationAsync",
                            { testType: val },
                            function(response) {

                                locs = JSON.parse(response);

                                var country = $.Enumerable.From(locs.TestSessionLocations)
                                    .Select(function(x) { return { Id: x.CountryId, Name: x.CountryName } })
                                    .Distinct(function(y) { return y.Name })
                                    .OrderBy(function(o) { return o.Name })
                                    .Select(function(c) { return $("<option>").text(c.Name).val(c.Id) })
                                    .ToArray();

                                $("select.country").empty();
                                $("select.country").append($("<option value='-1' selected>Select a country*</option>"));
                                $("select.country").append(country);
                                $("select.country").prop("disabled", false);
                            });
                    }
                });

            $("select.country")
                .change(function(e) {
                    $("select.location").empty();
                    $("select.location").append($("<option value='-1' selected>Select a city*</option>"));
                    $("select.location").prop('disabled', true);
                    $(".find-button").addClass('disabled');

                    var locations = $.Enumerable.From(locs.TestSessionLocations)
                       .Where(function (x) { return x.CountryId == $("select.country").val() })
                       .Distinct(function (y) { return y.LocationName })
                       .OrderBy(function (o) { return o.LocationName })
                       .Select(function(c) { return $("<option>").text(c.LocationName).val(c.Url) })
                       .ToArray();

                    $("select.location").append(locations);
                    
                    if ($("select.country").val() != -1) {
                        $("select.location").prop('disabled', false);
                    }
                });

            $("select.location")
                .change(function(e) {
                    if ($("select.location").val() != -1) {
                        $('.find-button').removeClass('disabled');
                    } else {
                        $('.find-button').addClass('disabled');
                    }
                });

            $(".find-button")
                .click(function (event) {
                    event.preventDefault();
                    if (!$(".find-button").hasClass("disabled")) {
                        var value = $("select.location").val();
                        window.location.href = value;
                    }
                });
            });
    </script>
