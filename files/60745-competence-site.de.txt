     require([ 'jquery','js/misc' ], function($,misc) {

       $(window).on('viewportChange.patchNavigation',$.proxy(function(e){

           $('.nav.navbar-nav li.divider','.navbar-main').each($.proxy(function(index,value){

             var $divider 				= $(value);
             var $navigationItem	=	$divider.prev();
             var $parentItem			=	$navigationItem.parent().closest('li');

             if( this.getBootstrapDisplayPort($) != 'xs' )
             {
               $divider.hide();
               $navigationItem.hide();
               $('>a',$parentItem).attr('href',$('>a',$navigationItem).attr('href'));   

               $('>a',$parentItem).unbind('click.callUrl');
               $('>a',$parentItem).bind('click.callUrl',{url: $('>a',$navigationItem).attr('href')},function(evnt){
                 $(location).attr('href', evnt.data.url);
                 evnt.preventDefault();
               });
 
             }
             else
             {
               $divider.show();
               $navigationItem.show();
               $('>a',$parentItem).attr('href',null);    
               $('>a',$parentItem).unbind('click.callUrl');   
             }
             
           },this));

           //handle hover-effect
           if(misc.getBootstrapDisplayPort($) != 'xs' ){          
             $('.dropdown','.navbar-main').off('.showSubItems'); 
             $('.dropdown-submenu>a','.navbar-main').off('.nestedNavigation');
             $('.dropdown','.navbar-main').on('mouseenter.showSubItems',function(event){
							if(!$(this).hasClass('open'))
							{
								//open
								//console.log('open menu of ' + $('>a.dropdown-toggle',$(this)).text() );
                $(this).addClass('open');
		  
							}              
             });
             $('.dropdown','.navbar-main').on('mouseleave.showSubItems',function(event){

              if($(this).hasClass('open'))
 							{
								//close
								//console.log('close menu of ' +$('>a.dropdown-toggle',$(this)).text() );
								$(this).removeClass('open');		  
 							}  
              });

           }
           else
           {
             $('.dropdown','.navbar-main').off('.showSubItems'); 
             $('.dropdown-submenu>a, .dropdown-leave>a','.navbar-main').off('.nestedNavigation');

             $('.dropdown-submenu>a','.navbar-main').on('click.nestedNavigation',function(event){
                if($(this).next('.dropdown-menu').length == 1 ){
                  $(this).next('.dropdown-menu').toggle();
                  event.stopPropagation();
                  event.preventDefault();
                }
             });
           }
           
       },misc));


       $(window).trigger('viewportChange.patchNavigation'); //initial run
	   });

         <div class="list_content"> Wer sich mit dem Thema SEO auseinandersetzt, kennt die Sache mit der robots.txt-Datei, die im Rootverzeichnis einer Site abgelegt wird und dem Crawler sagt, was er crawlen darf und was nicht. Nun kam von Google eine etwas konkretere Aussage, ob und warum eine solche Datei verwendet werden sollte. Kurz zum eigentlichen Sinn der robots.txt. Sie enthält Anweisungen, wie Crawler sich auf der Website bewegen dürfen. Es ist möglich, bestimmten Crawlern gezielte Anweisungen zu geben und auch einzelne Seiten (URLs), Verzeichnisse oder auch die gesamte Website für den Crawler zu sperren.  Als „normale“ Textdatei kann die robots.txt mit einem üblichen Editor erstellt und angepasst werden.
