        function SuccessSaveSearch(data) {
            var $form = $("#SpacesFilter");
            var serializedData = $form.serializeArray();
            var query_str = '';
            $.each(serializedData, function (i, data) {
                if ($.trim(data['value'])) {
                    query_str += (query_str == '') ? '?' + data['name'] + '=' + encodeURI(data['value']) : '&' + data['name'] + '=' + encodeURI(data['value']);
                }
            });
            if (query_str.length) {
                history.replaceState(null, null, query_str);
            } else {
                history.replaceState(null, null, "/es/spaces");
            }
            FillList(data);
        }
        function FillList(data) {
            
            if (typeof data.pag != "undefined" && data.pag > 1) {
                if (typeof data.htmlGridResult != "undefined") $("#divGridResult").append(data.htmlGridResult);
            } else {
                if (typeof data.htmlGridResult != "undefined") $("#divGridResult").html(data.htmlGridResult);
            }
            if (typeof data.Count != "undefined") {
                if (data.Count > 0) {
                    $(".countAll").html('Tenemos <span id="countAll">{0}</span> espacios disponibles para ti'.replace("{0}", data.Count));
                    $(".countAllMobile").html('<span id="countAll">{0}</span> espacios disponibles'.replace("{0}", data.Count));
                } else {
                    $(".countAll").html('');
                    $(".countAllMobile").html('');
                }
            }
            if (typeof data.htmlFilterFavorites != "undefined") $("#divFilterFavorites").html(data.htmlFilterFavorites);
            if (typeof data.htmlBubbles != "undefined") $("#divBubbles").html(data.htmlBubbles);
            if (typeof data.htmlButtonActions != "undefined") $("#divButtonActions").html(data.htmlButtonActions);
            if (typeof data.htmlfaceadv != "undefined") $("#divFaceAdv").html(data.htmlfaceadv);
            if (typeof data.Search == "undefined") data.Search = null;
            $("#SearchMobile").val(data.Search);
            $("#SearchDesktop").val(data.Search);
            $("#Search").val(data.Search);
        }
