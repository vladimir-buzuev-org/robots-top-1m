    var DaikinInternet = DaikinInternet || {};
    DaikinInternet.countryselection = function() {

        if ($("[data-behaviour='redirect']").length) {
            if ($.cookie('daikinlocale')) {
                window.location.href = $.cookie('daikinlocale');
            }
        } else {
            $.getJSON("//freegeoip.net/json/", function() {}).done(function(data) {
                $('[data-country="' + data.country_code.toLowerCase() + '"]').addClass('is-active');
            }).fail(function() {
                $('[data-country="EU"]').addClass('is-active');
            })
        };



        $('.country-item select').on('change', function(event) {
            var localepath = $(this).val();
            var selectedOption = $($(this)[0].selectedOptions[0]);
            setCookiesForElement(selectedOption);

            $(this).children(':first').prop('selected', true);
            window.location.href = localepath;
        });

        $('a[data-country]').on('click', function() {
            setCookiesForElement($(this));
        });

    };
    DaikinInternet.countryselection();
