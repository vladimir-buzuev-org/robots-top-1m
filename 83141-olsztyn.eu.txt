                        this.addCloseButtonToContainer(this.container);
                        this.createContainerContent(this.container);
                        this.appendToBody(this.container);
                    },
                    createContainerContent: function (container) {
                        this.addImage(container);
                        if (this.displayWithNewsletter()) {
                            this.addNewsletterBlock(container);
                        }
                    },
                    addImage: function (container) {
                        var image = $("<img/>")
                                .attr({
                                    src: this.config.imgSrc,
                                    alt: this.config.imgAlt,
                                    title: this.config.imgTitle
                                })
                                .addClass("img-responsive");
                        container.append(image);
                    },

                        var newsletter = $("<div />").addClass("newsletter");
                        var self = this;
                        $.ajax({
                            url: "o-olsztynie/404.html?tx_nettompopup_popupwindow%5Baction%5D=newsletter&tx_nettompopup_popupwindow%5Bcontroller%5D=Popup&cHash=16e75ef8e8936b2f64a512cc0b5404f6",

                                    trigger: 'manual'

                                    trigger: 'manual'

                                    var filterName = /^[A-Za-z0-9-]/;
                                    if (!filterEmail.test(email.val())) {
                                        email.tooltip('show');
                                        email.click(function () {
                                            email.tooltip('hide');
                                        });
                                        isValid = false;
                                    } else {
                                        email.tooltip('hide');
                                    }
                                    if (!filterName.test(name.val())) {
                                        name.tooltip('show');
                                        name.click(function () {
                                            name.tooltip('hide');
                                        });
                                        isValid = false;
                                    } else {
                                        name.tooltip('hide');
                                    }
                                    if (!rule.is( ':checked' )) {
                                        rule.tooltip('show');
                                        rule.click(function () {
                                            rule.tooltip('hide');
                                        });
                                        isValid = false;
                                    } else {
                                        rule.tooltip('hide');
                                    }

                                    if (isValid) {
