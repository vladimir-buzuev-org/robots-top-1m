   function onReady(method) {
      if (window.jQuery) {
         method();
      } else {
         setTimeout(function() { onReady(method) }, 50);
      }
   }

   // Desktop only
   onReady(function() {
      var isDesktop = window.innerWidth >= 1024;
      if (!isDesktop) {
           return;
      }

      var $nav = $('.tt-main-navigation-module');
      var $cat = $('.tt-homepage__cat');
      var $cards = $('.tt-homepage__cards');
      var $panels = $('.tt-panels');
      var $logoImg = $nav.find('.tt-logo img');

      $logoImg.get(0).onload = null; // Can be removed when TRIDUP-5616 is implemented.

      $nav.hover(makeNavWhite, makeNavTransparent);
      $(window).on('scroll', onScroll);

      equalizeHeights($cards.find('.tt-homepage__card__body'));
      makeNavTransparent();
      onScroll();

      function onScroll() {
         var scroll = window.scrollY;

         makeMenuSticky(scroll);
         makeCardsSticky(scroll)
      }

      function makeMenuSticky(scroll) {
         if (scroll > 80) {
            makeNavWhite();
         } else if (scroll <= 80) {
            makeNavTransparent();
         }
      }

      function makeCardsSticky(scroll) {
         var cardTitleHeight = 80;
         var cardDefaultTop = -20;
         var imgBottom = $cat.height();
         var cardDefaultPos = imgBottom + cardDefaultTop;
         var winHeight = window.innerHeight;
         var theFold = scroll + winHeight;
         var theCutoff = theFold - cardTitleHeight;

         if (cardDefaultPos > theCutoff) {
            $cards.css({top: theCutoff - imgBottom});
         } else {
            $cards.attr('style', '');
         }
      }

      function makeNavWhite() {
         $nav.removeClass('tt-main-navigation-module--transparent');
      }

      function makeNavTransparent() {
          var isPanelOpen = $panels.hasClass('in');
         var isScrolledDown = window.scrollY > 80;
         if (isPanelOpen || isScrolledDown) { return; }

         $nav.addClass('tt-main-navigation-module--transparent');
      }

      function equalizeHeights($elements) {
         var maxHeight = 0;
         $elements.each(function(i, el) { maxHeight = Math.max(el.clientHeight, maxHeight) } );

         $elements.height(maxHeight);
      }
   });

   // Mobile only
   onReady(function() {
      var isDesktop = window.innerWidth >= 1024;

      var $sports = $('.tt-homepage__cat#sports');

      function onClickSports() {

             var subcatOffset = '-50';
