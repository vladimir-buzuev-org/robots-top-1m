	var links = $(".banner").find("a"),
	pages = $("#nav span"),

	intervalTimer = setInterval(function(){
		doSlideChange();

	pages.click(function(){
		var clicked = $(this),
		index = $.trim(clicked.text());

		clearInterval(intervalTimer);

		pages.removeClass("selected");
		pages.eq(index).addClass("selected");

		links.fadeOut();
		links.eq(index).fadeIn();

		intervalTimer = setInterval(function(){
			doSlideChange();
		}, 5000)	;
	});

	function doSlideChange(){
		var visible = links.filter(":visible").next();
		if(visible.length === 0){
			visible = links.eq(0);
		}

		pages.removeClass("selected");
		pages.eq(links.index(visible)).addClass("selected");

		links.fadeOut();
		visible.fadeIn()
	}
