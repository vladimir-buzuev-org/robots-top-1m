        function cycleImages() {
            var $active = $('#portfolio_cycler a img.active');
            var $next = ($('#portfolio_cycler a img.active').closest('a').next().find('img').length > 0) ? $('#portfolio_cycler .active').closest('a').next().find('img') : $('#portfolio_cycler a:first img');

    $(function() {
        var ele = $('#scroll-images');
        var speed = 25, scroll = 5, scrolling;

        $('#scroll-up').mouseenter(function() {
            // Scroll the element up
            scrolling = window.setInterval(function() {
                ele.scrollTop(ele.scrollTop() - scroll);
            }, speed);

            $('#scroll-up').css({ 'cursor': 'pointer' });
        });

        $('#scroll-down').mouseenter(function() {
            // Scroll the element down
            scrolling = window.setInterval(function() {
                ele.scrollTop(ele.scrollTop() + scroll);
            }, speed);

            $('#scroll-down').css({ 'cursor': 'pointer' });
        });

        $('#scroll-up, #scroll-down').bind({
            click: function(e) {
                // Prevent the default click action
                e.preventDefault();
            },
            mouseleave: function() {
                if (scrolling) {
                    window.clearInterval(scrolling);
                    scrolling = false;
                }
                $('#scroll-up').css({ 'cursor': 'default' });
                $('#scroll-down').css({ 'cursor': 'default' });
            }
        });
    });
