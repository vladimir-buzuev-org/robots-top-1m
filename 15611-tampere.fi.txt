			var imageObj   = params.image;

			var height     = params.data.h;
			var x          = params.data.x;
			var y 		   = params.data.y;
			canvas.width   = width;

			// peips
			if (scalefactor) {
				//x = x * scalefactor;

			try{
				ctx = canvas.getContext("2d");

				//console.log(e);
