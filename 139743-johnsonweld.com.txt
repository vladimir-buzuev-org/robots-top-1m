    /**
     * Module which tacks utm codes and ad campaign metrics in a cookie,
     * persisting across sessions so that traffic from landing pages can be
     * analyized and stored outside of Google Analytics.
     * 
     * Currently exports a single class, CampaignMetrics, which when
     * instantiated persists ad campaign metrics in a cookie and provides an api
     * for querying those metrics.
     */
    var campaignMetrics = (function(URI, Cookies) {
        var CampaignMetrics = (function () {
            function CampaignMetrics(expiresIn) {
                this._currentUri = new URI();
                var utmUrlParams = {};
                (function(that) {
                    CampaignMetrics._campaignMetricNames.forEach(function (paramName) {
                        var urlParamValue = that._getUrlParam(paramName);
                        if (undefined !== urlParamValue) {
                            utmUrlParams[paramName] = urlParamValue;
                        }
                    });
                })(this);
                if (Object.keys(utmUrlParams).length > 0) {
                    this._setCookie(utmUrlParams, expiresIn);
                }
            }
            CampaignMetrics.prototype.name = function () {
                return this.getMetric('name');

            CampaignMetrics.prototype.content = function () {
                return this.getMetric('content');
            };

                return this.getMetric('source');

                return this.getMetric('medium');
            };
            CampaignMetrics.prototype.term = function () {
                return this.getMetric('term');
            };
            CampaignMetrics.prototype.getMetric = function (metricName) {
                var cookie = this._getCookie();
                return cookie[metricName];
            };
            CampaignMetrics.prototype.getMetricByUrlParam = function (urlParamName) {
                var metricName = CampaignMetrics._utmToMetricName[urlParamName];
                return this.getMetric(metricName);
            };
            CampaignMetrics.prototype._getUrlParam = function (metricName) {
                var urlParamName = CampaignMetrics._utmParamNames[metricName];
                return this._currentUri.search(true)[urlParamName];
            };
            CampaignMetrics.prototype._getCookie = function () {
                var cookie = Cookies.get(CampaignMetrics._cookieName);

                    return {};

                // and returning an empty object if the cookie is malformed.

                    Cookies.remove(CampaignMetrics._cookieName);
                    return {};

                return parsedCookie;

    var fqlGetShareCount = (function($, URI) {
        return function(url, cb) {
            var query = "SELECT total_count FROM link_stat WHERE url='" +
                url + "'";
            var baseUri = new URI({
                protocol: 'https',

                path: '/fql',
            });

                data: {q: query},
                method: 'GET',
                dataType: 'json',
                jsonp: false,
