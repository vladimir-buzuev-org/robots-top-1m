function chk_cookie( pop_id ){
	if(getCookie( pop_id ) == "no"){
		document.getElementById(pop_id).style.visibility = "hidden";
	}else{
		document.getElementById(pop_id).style.visibility = "";
	}	
}

function today_close( pop_id ){
	setCookie( pop_id, "no", 1);
	document.getElementById(pop_id).style.visibility = "hidden";
}
