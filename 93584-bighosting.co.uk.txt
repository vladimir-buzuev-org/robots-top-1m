function clearDefault(fieldName,defaultVal){

		document.getElementById(fieldName).value = '';
		document.getElementById(fieldName).style.color = '#000000';
	}

function showContacts(str,type,searchfield)
{
if (str.length<3)
  { 
  document.getElementById('contactSuggest').style.display='none';
  document.getElementById("contactSuggest").innerHTML="";
  return;
  }
 else{
    document.getElementById('contactSuggest').style.display='';
	xmlHttp=GetXmlHttpObject()
	if (xmlHttp==null)
	  {
	  alert ("Your browser does not support AJAX!");
	  return;
	  } 
	  var url="/contact_manager/contacts/contactsearch.asp";
	  url=url+"?q="+str;
	  url=url+"&t="+type;
	  url=url+"&f="+searchfield;
	  url=url+"&sid="+Math.random();
	  xmlHttp.onreadystatechange=stateChanged;
	  xmlHttp.open("GET",url,true);
	  xmlHttp.send(null);
	} 
}
function stateChanged() 
{ 
if (xmlHttp.readyState==4)
	{ 
	document.getElementById("contactSuggest").innerHTML=xmlHttp.responseText;
	}
}

function GetXmlHttpObject()
{
var xmlHttp=null;
try
  {
  // Firefox, Opera 8.0+, Safari
  xmlHttp=new XMLHttpRequest();
  }
catch (e)
  {
  // Internet Explorer
  try
    {
    xmlHttp=new ActiveXObject("Msxml2.XMLHTTP");
    }
  catch (e)
    {
    xmlHttp=new ActiveXObject("Microsoft.XMLHTTP");
    }
  }
return xmlHttp;
}

function setContact(contact_name, contact_id, send_to, search_type){
	if(search_type==1){
		document.getElementById('contactSearch').value=contact_name;
		document.getElementById('assign_to').value=contact_id;
		document.getElementById('contactSuggest').style.display='none';
		if(contact_id==-2){
			document.getElementById('send_to').value='';
		}
		else{
			document.getElementById('send_to').value=send_to;
		}	
	}
	else if(search_type==3){
		document.getElementById('contactSearch').value=contact_name;
		document.getElementById('contact_id').value=contact_id;
		document.getElementById('contactSuggest').style.display='none';
	}
	else if (search_type==2){
		location.href='/contact_manager/contacts/view.asp?ci='+contact_id+'&qf=1';
	}
}
function changeStyle(resultnum,direction){
	if(direction=='in'){
		document.getElementById('result'+resultnum).style.backgroundColor='#FFFCCC';
		document.body.style.cursor='pointer';
	}
	else{
		if(resultnum!=-1){
			document.getElementById('result'+resultnum).style.backgroundColor='#FFFFFF';
		}
		document.body.style.cursor='default';
	}
}
function validateForm(){
	var errormessage = '';
	if(document.getElementById('contactSearch').value=='' || document.getElementById('contactSearch').value=='Type Team Member / Colleague Name'){
		errormessage = errormessage+'\nYou have not entered a name.';
	}
	if(document.getElementById('send_to').value==''){
		errormessage = errormessage+'\nYou have not entered an email address.';
	}
	if(errormessage != ''){
		errormessage = 'There was an error:\n'+errormessage
		alert(errormessage);
		return false;
	}
}
