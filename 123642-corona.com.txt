            _setlanguageArray: function() {
                this.langSite = this.langSite.replace(/\s+/g, '');
                this.langSite = this.langSite.slice(1, -1);
                this.langSite = this.langSite.split(',');

              /**
               * Use this method for manage the amount of languages.
               * TODO: Add conditional for more languages

            _filterLanguage: function() {
                this._setlanguageArray();

                var enableLanguageSelection = false;

                
                    
                

                if(this.site === 'budlight' && enableLanguageSelection || this.site === 'michelobultra' && enableLanguageSelection){
                    this.langSite = this.langSite.slice(1);

                    if(ourBeersIndex >= 0){
                        this.langSite.splice(ourBeersIndex, 1);

                if(this.langSite.length === 2) {
                    this._chooseLanguage();
