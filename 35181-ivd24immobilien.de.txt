    function toggleRooms(element)
    {
        element = $(element);

        var roomRange = element.closest('form').find('.room-range');

        if(parseInt(element.children(':selected').data('rooms')))
        {
            roomRange.show();
        }
        else
        {
            roomRange.hide();
            roomRange.find('.btn.active').removeClass('active');
            roomRange.find('input[type="radio"]').prop('checked', false);
            setRoomRange(element, 0, 0);
        }
    }

    function setRoomRange(element, from, to)
    {
        element = $(element);
        element.closest('form').find('input[name="zimmer_ab"]').val(from);
        element.closest('form').find('input[name="zimmer_bis"]').val(to);
    }
