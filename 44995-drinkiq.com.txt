    function getMarqueeInfo() {

        $.ajax({
            type: "GET",
            cache: false,
            data: { currentPageID: 1052 },
            url: '/umbraco/Surface/Marquee/GetMarqueeData',
            dataType: "json",

    function TogleMarquee(lastMarquee) {

        if (lastMarquee.Text === null) {
            $('.marquee-class').attr("style", "display: none !important");
            return;
        }

        if (lastMarquee.Text !== localStorage.getItem("Text")) {
            localStorage.setItem("Link", lastMarquee.Link);
            localStorage.setItem("Text", lastMarquee.Text);
            localStorage.setItem("OpenInNewWindow", lastMarquee.OpenInNewWindow);
            localStorage.setItem("UserClose", false);
            localStorage.setItem("Date", new Date());

            SetMarqueeInfo(lastMarquee);

        } else {

            var milisecondsPerDay = 86400000;
            var currentDateTime = new Date();
            var dateDiff = Math.abs(currentDateTime - new Date(localStorage.getItem("Date")));

            if (!StringToBool(localStorage.getItem("UserClose")))

                if (dateDiff < milisecondsPerDay) {

                    SetMarqueeInfo(lastMarquee);

                } else {

                    $('.marquee-class').attr("style", "display: none !important");

                }
        }

    }

    function SetMarqueeInfo(lastMarquee) {

        $('#marquee').text(lastMarquee.Text);
        $('.marquee-class').attr("style", "display: block !important");

        var openInNewWindow = StringToBool(localStorage.getItem("OpenInNewWindow")) ? "_blank" : "_self";
        var newLink = localStorage.getItem("Link");

        $('#marquee').click(function() {
            window.open(
                newLink,
                openInNewWindow
            );
        });

    }

    function StringToBool(string) {
        return string === "true";
    }
