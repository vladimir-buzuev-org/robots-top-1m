function drawGraph(id, score)
{
	var width = $(".cart_title").outerWidth(), color = d3.scale.ordinal().range(["#4a90e2", "#ccc"]);
	console.log( id+": visible ("+score+")" );

    var arc = d3.svg.arc()
                .outerRadius(width * 0.48)
                .innerRadius(width * 0.41)
                .padAngle(0.01);

    var pie = d3.layout.pie()
        .value(function(d) { return d.orig; })
        .sort(null);

   	$(id+" .card_graph").html("");
    
    var m_title = $(id+" .cart_title");

    var svg = d3.select(id + " .card_graph").append("svg")
              .attr("width", width)
              .attr("height", width)
              .attr("style", "transform: rotate(180deg)")
            .append("g").attr("class", "g").attr("transform", "translate(" + width / 2 + "," + width / 2 + ")");

    var data = [ { label: 'a', orig: 0, value: score }, { label: 'b', orig: 10, value: 10-score } ],
        g = svg.datum(data).selectAll("path")
                  .data(pie)
                .enter().append("path")
                  .attr("class", "arc")
                  .attr("fill", function(d) { return color(d.data.value); })
                  .attr("d", arc)
                  .each(function(d) { this._current = d; });              

    pie.value(function(d) { return d.value; }); // change the value function
    g = g.data(pie); // compute the new angles
    g.transition().duration(750).attrTween("d", arcTween); 

    function arcTween(a) {
      var i = d3.interpolate(this._current, a);
      this._current = i(0);
      return function(t) {
        var m_val = Math.round(score*t);

      	if (m_val >= 10) m_val = "A+"; else
      	if (m_val >=  9) m_val = "A"; else
      	if (m_val >=  8) m_val = "B+"; else
      	if (m_val >=  7) m_val = "B"; else
      	if (m_val >=  6) m_val = "C+"; else
                         m_val = "C";

        m_title.html( m_val );
        return arc(i(t));
      };
    }
}
