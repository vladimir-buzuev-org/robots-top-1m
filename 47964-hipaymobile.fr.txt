var CookieJar = {
  getAll: function() {
    return document.cookie.split(/;\s/)
    .map(function(segment) {
      return segment.split(/=/);
    })
    .reduce(function(memo, pair) {
      memo[pair[0]] = pair[1]
      return memo;
    }, {});
  },
  get: function(key) {
    return this.getAll()[key];
  },
  // ttl is a value in milliseconds indicating for how long the cookie is valid
  set: function(key, value, ttl) {
    var keyValue = [key, value].join("=")
    var now = new Date()
    // Add 395 days as milliseconds to the current time (expires in 13 months)
    now.setTime(now.getTime() +  ttl)
    var expiry = ['expires', now.toGMTString()].join("=")
    var path = ['path', "/"].join("=")
    document.cookie = [keyValue, expiry, path].join("; ")
    return value;
  }
};
