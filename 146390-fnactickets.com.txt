	function getTimeoutPanier(timeoutDiv, timeoutMinSpan, timeoutMaxSpan, labelTimemoutMinutesSpan, timeoutDivBasketPage, timeoutMinSpanBasketPage, timeoutMaxSpanBasketPage, labelTimemoutMinutesSpanBasketPage) {
		// Calculer la difference entre la date de chargement, et la date actuelle.
		var elapsedTime = new Date().getTime() - loadTimePanier;
	
		// Retrancher cela au temps restant au chargement.
		var timeLeftNow = timeLeftOnLoadPanier - elapsedTime;
		if (timeLeftNow > 0) {
			timeLeftNow = (timeLeftNow - timeLeftNow % 1000) / 1000; // secondes completes
			var secondsLeft = timeLeftNow % 60;
			timeLeftNow = (timeLeftNow - secondsLeft ) / 60; // minutes completes
			var minutesLeft = timeLeftNow % 60;
	
			// Afficher.
			if (document.getElementById(timeoutDiv)) {
				if (minutesLeft < 2) {
					if (secondsLeft % 2 == 0) {
						$("#"+timeoutDiv).addClass("even").removeClass("odd");
						$("#"+timeoutDivBasketPage).addClass("even").removeClass("odd");
					} else {
						$("#"+timeoutDiv).addClass("odd").removeClass("even");
						$("#"+timeoutDivBasketPage).addClass("odd").removeClass("even");
					}
				}
				$("#"+timeoutMinSpan).html(minutesLeft);
				$("#"+timeoutMinSpanBasketPage).html(minutesLeft);
				$("#"+timeoutMaxSpan).html(secondsLeft+""); // On rajoute les "" pour bien ajouter le 0
				$("#"+timeoutMaxSpanBasketPage).html(secondsLeft+""); // On rajoute les "" pour bien ajouter le 0
				if (minutesLeft == 1) {
					if (document.getElementById(labelTimemoutMinutesSpan)) $("#"+labelTimemoutMinutesSpan).html($("#"+labelTimemoutMinutesSpan).html().replace('s', ''));
					if (document.getElementById(labelTimemoutMinutesSpanBasketPage)) $("#"+labelTimemoutMinutesSpanBasketPage).html($("#"+labelTimemoutMinutesSpanBasketPage).html().replace('s', ''));
				}
				if (minutesLeft < 1) {
					if (document.getElementById(labelTimemoutMinutesSpan)) {
						$("#"+timeoutMinSpan).remove();
						$("#"+labelTimemoutMinutesSpan).remove();
					}
					if (document.getElementById(labelTimemoutMinutesSpanBasketPage)) {
						$("#"+timeoutMinSpanBasketPage).remove();
						$("#"+labelTimemoutMinutesSpanBasketPage).remove();
					}
				}
				
			}
	
			// Cet id est celui present dans la page du panier !
			if (document.getElementById("timeoutDivBasketPage") && document.getElementById(timeoutDivBasketPage)) {
				var htmlElement = $("#"+timeoutDivBasketPage).html();
				$("#timeoutDivBasketPage").html(htmlElement);
			}
			if (minutesLeft > 0) {
				if (document.getElementById("popinWarningPanierMinNumber")) {
					$("#popinWarningPanierMinNumber").html(minutesLeft);
				}
			} else {
				if (document.getElementById("popinWarningPanierMin")) {
					$("#popinWarningPanierMin").remove();
				}
			}
			if (secondsLeft > 0) {
				if (document.getElementById("popinWarningPanierSecNumber")) {
					if (document.getElementById("popinWarningPanierSec")) {
						$("#popinWarningPanierSec").show();
					}
					$("#popinWarningPanierSecNumber").html(secondsLeft);
				}
			} else {
				if (document.getElementById("popinWarningPanierSec")) {
					$("#popinWarningPanierSec").hide();
				}
			}
			try {
				if (warningTimeEXRP > 0 && minutesLeft == warningTimeEXRP && secondsLeft == 0) {
					openWarningPopinPanier();
				}
			} catch (ex) {
			}
			
			// On se relance dans 1 seconde.

		}
		else {
			$("#"+timeoutDiv).hide();

					openFinPopinPanier();
