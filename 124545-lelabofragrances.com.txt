		$(document).ready(function() {
			$win = $(window);
			$win.on('resize', function() {
				setBackgroundImage()
			}).resize();
			if(getCookieWithHelper(false, 'ACDC_LOCALE') != undefined) {
				$('#userSetting #userSettingLanguage').val(getCookieWithHelper(false, 'ACDC_LOCALE'));
				// remove select language option since a language is now selected
				$("#userSetting #userSettingLanguage option[value='']").remove();								
				$('#userSetting #userSettingLanguage').dropdown('update');
			}
			function setBackgroundImage() {
				var image;
				if ( $win.width() < 767 ) {
					image = $('.mobile-image').attr('src');
					$('.section-full').css('background-image', 'url(' + image + ')');
				} else if ( $win.width() < 1024 ) {
					image = $('.tablet-image').attr('src');
					$('.section-full').css('background-image', 'url(' + image + ')');
				} else {
					$('.section-full').removeAttr('style')
				}
			}
		});
		$(function() {
			/* Pre-populate the country based on query param */
			var parsableURI = new URI(location.href);
			var dataMap = parsableURI.search(true);
			var redirectURI = "/";
			if ("" != dataMap['path']) {
				redirectURI = new URI(dataMap['path']);
			}
			var defaultRegion = dataMap['defaultRegion'];
			if (defaultRegion == "CA") {
				$('#userSettingRegion').selectize()[0].selectize.setValue("CA");
			} else {		
				/* Pre-populate the country based on the IP */
				$.getJSON(js_site_var['context_path'] + '/app/account/info.json', function(data) {
					if (typeof data.requestorIp != 'undefined') {
						$.getJSON(js_site_var['context_path'] + '/app/proxy/geoip/' + data.requestorIp, function(geoipData) {
							if  (typeof geoipData.country_code != 'undefined') {
								if (typeof $('#userSettingRegion').selectize()[0].selectize.options[geoipData.country_code] != 'undefined') {
									console.log("Pre-selecting IP address country::" + geoipData.country_code);
									$('#userSettingRegion').selectize()[0].selectize.setValue(geoipData.country_code);
								}
							}
						});
					}
				});
			}
			$('#submit-link').click(function(event) {
				var parsableURI = new URI(location.href);
				var dataMap = parsableURI.search(true);
				var redirectURI = "/";
				if ("" != dataMap['path']) {
					redirectURI = new URI(dataMap['path']);
				}
				var defaultRegion = dataMap['defaultRegion'];
				$('.country-required').hide();
				$('.language-required').hide();
				var selectedUserLanguage = $('#userSetting #userSettingLanguage').val();
				// require a language selection
				if (selectedUserLanguage == '') {
					$('.language-required').show();
					return false;
				}
				// assume anything not French will be English.
				if (selectedUserLanguage != 'FR') {
					selectedUserLanguage = 'EN';
				}
				var selectedUserRegion = $('#userSetting #userSettingRegion').val();
				if (selectedUserRegion  == '') {
					$('.country-required').show();
					return false;
				}
				if (selectedUserRegion == "JP") {
					window.location.href = "http://www.lelabofragrances.jp/";
					return false;
				}
				if (selectedUserRegion == "CA") {
					for(var key in dataMap) {
						if (dataMap.hasOwnProperty(key)) {
							if (key != 'path') {
								redirectURI.addSearch(key, dataMap[key]);
							}
						}
					}
					if (redirectURI.query() != "") {
						redirectURI = redirectURI + "&region=CA&locale=" + selectedUserLanguage;
					} else {
						redirectURI = redirectURI + "?region=CA&locale=" + selectedUserLanguage;
					}
					window.location.replace("https://www.lelabofragrances.ca" + redirectURI);
					return false;
				}
				setCookieWithHelper(false, 'ACDC_LOCALE', selectedUserLanguage, { expires: 60, path: '/', domain: js_site_var['cookie_domain']  });
				setCookieWithHelper(false, 'acdc_region', selectedUserRegion, { expires: 60, path: '/', domain: js_site_var['cookie_domain']   });
				setCurrencyBasedOnRegion(selectedUserRegion);
				setVATBasedOnRegion(selectedUserRegion);
				// redirect to the desired page with the cookie set
				//window.location = '/';
				for(var key in dataMap) {
					if (dataMap.hasOwnProperty(key)) {
						if (key != 'path') {
							redirectURI.addSearch(key, dataMap[key]);
						}
					}
				}
				window.location = redirectURI;
				return false;
			});
		});
	</script>
