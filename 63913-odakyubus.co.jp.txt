$(function(){
	$('ul.slides').slider({
		sliderType: 2,
		animationTypeMv: 'crossfade',
		fadeTime: 800,
		moveFlag: 1,
		moveTime: 5000
	});

	$('#trouble').heightLine();

	$('#tabNav').tabControl();
	
	var slideHeight = function(){
		var imgHeight = $('.slides img').height();
		if(imgHeight > 544){
			$('.slides').css({height:544});
		}else{
			$('.slides').css({height:imgHeight});
		}
	}
	$(window).on('resize load', function(){
		slideHeight();
	});

	// サジェスト
	new Suggest.Local(
		"departure", 
		"departure_suggest", 
		odakyu_bus_list, 
		{dispMax: 100, highlight: true}
		);
	new Suggest.Local(
		"arrival",
		"arrival_suggest",
		odakyu_bus_list,
		{dispMax: 100, highlight: true}
		);

	// 日付計算 (今月・来月・再来月。12月以降は翌年を表示する)
	var target_date = new Date();
	var month = ('00'+(target_date.getMonth()+1)).slice(-2);
	var month2 = ('00'+(target_date.getMonth()+2)).slice(-2);
	var month3 = ('00'+(target_date.getMonth()+3)).slice(-2);

	var day = target_date.getDate();
	var year = target_date.getFullYear();
	var hour = target_date.getHours();
	var minute = target_date.getMinutes();
	if (month == '11'){ // 11, 12, 1 の場合
		$('select[name=month]')
		.append('<option value="11">11</option>')
		.append('<option value="12">12</option>')
		.append('<option value="01">01</option>');
	} else if (month == '12'){ // 12, 1, 2 の場合
		$('select[name=month]')
		.append('<option value="12">12</option>')
		.append('<option value="01">01</option>')
		.append('<option value="02">02</option>');
	} else { // それ以外
		$('select[name=month]')
		.append('<option value="' + month + '">' + month + '</option>')
		.append('<option value="' + month2 + '">' + month2 + '</option>')
		.append('<option value="' + month3 + '">' + month3 + '</option>');
	}

	// 日付と時間の初期値を現在日時にする
	$('select[name=month]').val(month);
	checkSelect($('#bus_form select[name=day]')[0], day);
	checkSelect($('#bus_form select[name=hour]')[0], hour);
	checkSelect($('#bus_form select[name=min]')[0], minute);

	// 送信時処理
	$('p.search').click(function(){
		// バリデーション
		// 出発地必須
		var orvName = $('input[name=orvName]').val();
		if(orvName == ''){
			$('#error_message').html('出発バス停を入力してください。');
			return false;
		}
		// 出発地存在チェック
		var orv_res = odakyu_bus_list.indexOf(orvName);
		if(orv_res == -1){
			$('#error_message').html('「' + orvName + '」に該当するバス停はありません。');
			return false;
		}
		// 目的地必須
		var dnvName = $('input[name=dnvName]').val();
		if(dnvName == ''){
			$('#error_message').html('到着バス停を入力してください。');
			return false;
		}
		// 目的地存在チェック
		var dnv_res = odakyu_bus_list.indexOf(dnvName);
		if(dnv_res == -1){
			$('#error_message').html('「' + dnvName + '」に該当するバス停はありません。');
			return false;
		}
		// 出発地・目的地が同じ
		if(orv_res == dnv_res){
			$('#error_message').html('出発バス停と到着バス停が同じです。');
			return false;
		}

		// 今月が11月もしくは12月の場合。1月と2月は翌年を表示
		if((month == '11' || month == '12') && ($('select[name=month]').val() == '01' || $('select[name=month]').val() == '02')){
			year = (year*1+1);
		}

		// 日付の妥当性
		var date_res = checkDate(year, $('select[name=month]').val(), $('select[name=day]').val());
		if(date_res == false){
			$('#error_message').html('日時の指定が不正です。');
			return false;
		}

		// 今日よりも前だった場合
		var day_int = parseInt($('select[name=day]').val());
		if($('select[name=month]').val() == month  && day_int < day ){
			$('#error_message').html('日付は現在より2カ月以内の範囲で指定してください。');
			return false;
		}

		// 日付を結合
		$('input[name=date]').val(
			year 
			+ $('select[name=month]').val()
			+ $('select[name=day]').val()
			+ $('select[name=hour]').val()
			+ $('select[name=min]').val()
			);
	});

	// 現在日時をデフォルト選択にする
	function checkSelect(obj, value){
		for(var i = 0; i < obj.length; i++){
			if(obj[i].value==value){
				obj[i].selected = true;
				break;
			}
		}
	}

	// 日付の妥当性
	function checkDate(year, month, day) {
		var target_date = new Date(year, (month-1), day);
		if(target_date == null || target_date.getFullYear() != year || target_date.getMonth() != (month-1) || target_date.getDate() != day) {
			return false;
		}
		return true;
	}
});
