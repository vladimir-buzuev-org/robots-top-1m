<script type="text/javascript">
  $( document ).ready(function() {
    
    $("#changeFont, #changeScheme, #changeSize").click(function(){
      var id = $(this).attr("data-open");
      $(".viMenu").hide();
      $("#" + id).show();
    });
    
    $(".viMenu span").click(function(){
      
      var addCssClass = $(this).attr("id");
      var indentifier = $(this).parent().attr("id");
      
      $(this).parent().children("span").each(function(){
        $("body").removeClass($(this).attr("id"));
      });
      
      if(typeof(addCssClass) !== 'undefined'){
        $("body").addClass(addCssClass);
        Cookies.set(indentifier, addCssClass, { expires: Infinity });
        if(indentifier == 'viScheme') $("body").addClass("vic");
        if(indentifier == 'viSize') $("body").addClass("vit");
      }else{
        Cookies.expire(indentifier);
        if(indentifier == 'viScheme') $("body").removeClass("vic");
        if(indentifier == 'viSize') $("body").removeClass("vit");
      }
      
      $(".viMenu").hide();
    });    
    
    if(Cookies.get('viFont')) $("body").addClass(Cookies.get('viFont'));
    if(Cookies.get('viScheme')) $("body").addClass(Cookies.get('viScheme')).addClass("vic");
    if(Cookies.get('viSize')) $("body").addClass(Cookies.get('viSize')).addClass("vit");
    if(Cookies.get('viButtons')) $(".viButtons").addClass("hide");
    
    if(Cookies.get('floatMenu')){ 
      $(".ancor-menu").addClass("noFloat");
      var data = $("#fixMenu").attr("data-text");
      var text = $("#fixMenu").find('span').text();      
      $("#fixMenu").attr("data-text",text).find("span").text(data);
    }
    
    $("#vi-close").click(function(){
      
      if($(".viButtons").hasClass("hide")){
        $(".viButtons").removeClass("hide");
        Cookies.expire('viButtons');
      }else{
        Cookies.set('viButtons', 1, { expires: Infinity });
        $(".viButtons").addClass("hide");
      }
      
    });
    
    $("#fixMenu").click(function(){
      
      var data = $(this).attr("data-text");
      var text = $(this).find('span').text();
      
      if($(".ancor-menu").hasClass("noFloat")){
        $(".ancor-menu").removeClass("noFloat");
        Cookies.expire('floatMenu');
        $(this).attr("data-text",text).find("span").text(data);
      }else{
        Cookies.set('floatMenu', 1, { expires: Infinity });
        $(".ancor-menu").addClass("noFloat");        
        $(this).attr("data-text",text).find("span").text(data);
      }
      
    });
    
  });
</script>
