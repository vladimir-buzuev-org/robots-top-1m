	function resize_burger() {
		if (jQuery(window).width() < 1023){

			jQuery(".content-scroll-js").mCustomScrollbar();

			jQuery(".full-menu-js .search_form").css("margin-top", "auto");
			jQuery(".full-menu-js .burger-menu").css("margin-top", "auto");

			jQuery(".full-menu-js .burger-menu #menu-menu-burger li a").css("padding", "auto");

			if (jQuery(window).height() < 700){
				jQuery(".full-menu-js .search_form").css("margin-top", "0");
			} else {
				jQuery(".full-menu-js .search_form").css("margin-top", "86px");
			}
			if (jQuery(window).height() < 611){
				jQuery(".full-menu-js .burger-menu").css("margin-top", "20px");
				jQuery(".full-menu-js .burger-menu-bottom").css("margin-top", "30px");

	jQuery(function() {
		jQuery(document).delegate('a[href*=#]:not([href=#])','click','touchstart',function(){
			if (location.pathname.replace(/^\//,'') == this.pathname.replace(/^\//,'') && location.hostname == this.hostname) {
				var target = jQuery(this.hash);
				target = target.length ? target : $('[name=' + this.hash.slice(1) +']');
				if (target.length) {
					jQuery('html,body').animate({
					scrollTop: target.offset().top
					}, 1000);

				}
			}
