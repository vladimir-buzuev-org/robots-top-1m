(function($) {
    // Cookie制御（2度め以降の訪問の場合にリンク表示）

    'use strict';

    var cookie = {
        expirationDate: '', // クッキーの期限を「年/月/日」形式で指定 (例: expirationDate: '2017/12/31',)
        cookieName: 'visited-top=1',
        myCookie: function() {
            if (cookie.expirationDate.length === 0) {
                return cookie.cookieName;
            } else {
                return cookie.cookieName + ';expires=' + new Date(cookie.expirationDate);
            }
        }
    };
    var cookies = document.cookie;
    if (!cookieExists(cookie.cookieName)) {
        $('.js-visited-top').hide();
        document.cookie = cookie.myCookie();
    }

    function cookieExists(cookieName) {
        if (cookies && cookies.length > 0) {
            return cookies.indexOf(cookieName + ';') >= 0;
        } else {
            return false;
        }
    }
})(jQuery);

(function($) {
    // タブ

    'use strict';

    var $container, $tab, $contents;

    $('.js-tabs-container').each(function() {
        $container = $(this);
        $tab = $('.js-tabs', this);
        $contents = $('.js-tab-content', $container);

        $('.js-tab-content', $container).not(':first').addClass('tabHide');

        $('a', $tab).on('click', function() {
            var targetID = $(this).attr('href');
            var $group = $('a', $tab);

            $contents.removeClass('tabActive').addClass('tabHide');
            $(targetID).removeClass('tabHide').addClass('tabActive');
            $group.removeClass('current');
            for (var i = 0; i < $group.length; i++) {
                if ($group.eq(i).attr('href') === targetID) {
                    $group.eq(i).addClass('current');
                }
            }

            return false;
        });
    });
})(jQuery);

(function($) {
    // スマホ幅のみでの開閉

    'use strict';
    var $win = $(window);

    var BREAK_POINT = 640;
    var toggleButtonClass = '.js-toggle-button';
    var $toggleButton = $(toggleButtonClass);
    var toggleContentClass = '.js-toggle-content';
    var toggleShowMoreClass = 'js-spToggle-showMore';  // もっと見るの開閉
    var toggleExpandClass = 'js-spToggle-expand';  // 見出しの開閉
    var $toggleContent = $(toggleContentClass);
    var openText = '閉じる';
    var closedText = 'もっと見る';


    $toggleButton.on('click', function(e) {
        e.preventDefault();

        var $self = $(this);
        var $selfParent = $self.parent();

        if (isSpWidth() === true) {
            if ($self.siblings(toggleContentClass).is(':visible')) {
                if ($selfParent.hasClass(toggleShowMoreClass)) {
                    $self.find('span').text(closedText);
                    $selfParent.removeClass('toggle-open').addClass('toggle-close');
                } else if ($selfParent.hasClass(toggleExpandClass)) {
                    $selfParent.removeClass('expand_open').addClass('expand_close');
                }
                $self.siblings(toggleContentClass).stop().slideUp();
            } else {
                if ($selfParent.hasClass(toggleShowMoreClass)) {
                    $self.find('span').text(openText);
                    $selfParent.removeClass('toggle-close').addClass('toggle-open');
                } else if ($selfParent.hasClass(toggleExpandClass)) {
                    $selfParent.removeClass('expand_close').addClass('expand_open');
                }
                $self.siblings(toggleContentClass).stop().slideDown();
            }
        }
    });

    $win.on('load.sp_expand resize.sp_expand', function() {
        if (isSpWidth() === false) {
            $(toggleShowMoreClass).find(toggleButtonClass).hide();
            $toggleContent.show();
        } else {
            $toggleButton.show();
            $toggleContent.each(function() {
                var $self = $(this);
                var $selfParent = $self.parent();
                if ($selfParent.hasClass('toggle-open') || $selfParent.hasClass('expand_open')) {
                    $self.show();
                } else {
                    $self.hide();
                }
            });
        }
    });

    $(document).ready(function() {
        if (!$('.' + toggleShowMoreClass).hasClass('toggle-open')) {
            $('.' + toggleShowMoreClass).addClass('toggle-close');
        }
        if (!$('.' + toggleExpandClass).hasClass('expand-open')) {
            $('.' + toggleExpandClass).addClass('expand_close');
        }
    });

    function isSpWidth() {
        return (window.innerWidth || document.documentElement.clientWidth || 0) <= BREAK_POINT;
    }
})(jQuery);
