    function fetchFlights(authToken) {
        $.ajax({
            type: "GET",
            beforeSend: function(request) {
                request.setRequestHeader("authorization", "Bearer " + authToken);
            },
            url: "https://restapi.avinor.no/followflightapi/v1/flights",
            processData: false,
            data: 'json',
            success: function (res) {
                var upcommingFlight = getUpcomingFlight(res);
                if (upcommingFlight !== null) {
                    updateText(upcommingFlight.departureAirportIata + ' &#9992; ' + upcommingFlight.arrivalAirportIata, false, true, false, upcommingFlight.flightLegId);
                } else {
                    updateText(null, false, false, true, null);
                }
            },
            error: function (err) {
                updateText(null, false, false, true, null);
            }
        });
    }

    function getUpcomingFlight(flights) {
        if (flights === null || flights === undefined) {
            return null;
        }

        var currentDateTime = new Date();
        var yesterdayDateTime = new Date();
        yesterdayDateTime.setDate(yesterdayDateTime.getDate() - 1);

        var upcommingFlightDateTime = null;
        var upcommingFlight = null;
        for (var i = 0; i < flights.length; i++) {
            var flightDateTime = new Date(flights[i].scheduleTime.split("+")[0]);

            if ((flightDateTime - yesterdayDateTime) >= 0) {
                if (upcommingFlightDateTime === null ||
                (Math.abs(flightDateTime - currentDateTime) < Math.abs(upcommingFlightDateTime - currentDateTime))) {
                    upcommingFlightDateTime = flightDateTime;
                    upcommingFlight = flights[i];
                }
            }
        }
        return upcommingFlight;
    }
