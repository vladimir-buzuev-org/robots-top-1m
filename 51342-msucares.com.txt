  <div class="views-field views-field-body">        <div class="field-content"> <div id="county-select"><select class="form-control" id="countySelect" onchange="if( this.options[this.selectedIndex].value != '' ) location.href=this.options[this.selectedIndex].value;"><option>Select a County</option> </select></div>
<script type="text/javascript">

  function initialize() {

    // Load a local GeoJSON file
    var script = document.createElement('script');
    script.setAttribute('src','/sites/all/themes/msuext/json/counties-geo2.json'); 
    document.getElementsByTagName('head')[0].appendChild(script);
  }

  // Defines the callback function referenced in the json file.
  function county_callback(data) {
    // sort list of counties from the json file alphabetically by name
    data.features.sort(function(a, b){
      var countyA = a.properties.name.toLowerCase(), countyB = b.properties.name.toLowerCase();
      if (countyA < countyB) // sort string ascending
        return -1;
      if (countyA > countyB)
        return 1;
      return 0;
    })

    // Loop through each county listed in the json file
    for (var i = 0; i < data.features.length; i++) {
      var countyShape = data.features[i];
      var polyIndex = i;
      // create options to add to select list
      createSelectList('countySelect', countyShape, polyIndex);
    }
  }

  function createSelectList(selectId, countyShape, polyIndex) {
    // Creates a select list
    var select = document.getElementById(selectId);
    var option = document.createElement('option');
    var title = countyShape.properties.name;
    var countyNameFormatted = countyShape.properties.name.replace(/\s+/g, '-').toLowerCase();
    option.id = "county" + polyIndex;
    option.value = "/" + "msu-extension-" + countyNameFormatted + "-county" ;
    option.innerHTML = title;
    select.appendChild(option);
  }

  initialize();
