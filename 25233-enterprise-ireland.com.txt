        function getVideoDetail(videolink) {
            var video = { id: 0, type: "" };
            var regExp = /^.*((youtu.be\/)|(v\/)|(\/u\/\w\/)|(embed\/)|(watch\?))\??v?=?([^#\&\?]*).*/;
            var match = videolink.match(regExp);

            if (videolink.indexOf("vimeo.com") > -1) {

                if (videolink.slice(-1) == "/") {
                    var str = videolink.substring(0, (videolink.length - 1));
                    video.id = str.substring(str.lastIndexOf("/") + 1);

                }
                else {
                    video.id = videolink.substring(videolink.lastIndexOf("/") + 1);
                }
                video.type = "vimeo";
            }
            else {
                if (match && match[7].length == 11) {
                    video.id = match[7];
                }
                video.type = "youtube";
            }
            return video;
        }

        function loadNewsDocuments(start, count) {
            $('.loadWrap').before('<div class="loader-holder"><img src="/images/loader/loader.gif" alt="loader" /></div>');
            $.ajax({
                type: "POST",
                url: "/Webservices/NewsService.asmx/getNews",
                contentType: "application/json; charset=utf-8",
                data: "{'start':'" + start + "', 'count':'" + count + "'}",
                dataType: "json",
                success: ajaxSuccess,
                error: ajaxFailed,
                complete: function () { }
            });
        }

        function ajaxSuccess(resultJson) {

            $(".loader-holder").remove();

            var response = JSON.parse(resultJson.d);
            var remainderItemNum;

            var total = $("#newsWrapper > .each-item").size();
            
            $(response).each(function (i, val) {
                var html = "<article class='each-item new-item'>";
                if (val.SummaryImage != '') {
                    html += "<a class='clearfix'" + "href='" + val.FullPath + "'>"
                         + "<div class='img-panel'><img src='" + "/apps/ImageResizer.ashx?file=" + val.SummaryImagePath + "&amp;w=405" + "'" + "/></div>"
                }
                else {
                    html += "<a class='clearfix no-img'" + "href='" + val.FullPath + "'>"
                        + "<div class='img-panel'><img src='/EI_Corporate/en/SiteLogo.jpg' alt='Enterprise Ireland logo image' /></div>";
                }

                if (val.Summary == '') {
                    html += "<div class='item-details'> <header class='no-summary'><h3>" + val.Name + "</h3></header>";
                }

                if (val.Summary != '') {
                    var summary = val.Summary;
                    var summary_length = summary.length;
                    if (summary_length > 380) {

                        var trimmed_summary = summary.substring(0, 380);

                        html += "<div class='item-details'> <header><h3>" + val.Name + "</h3></header>" + "<p class='summary'>" + trimmed_summary + "..." + "</p>";
                    }
                    else {
                        html += "<div class='item-details'> <header><h3>" + val.Name + "</h3></header>" + "<p class='summary'>" + val.Summary + "</p>";
                    }
                }

                html += "<p class='clearfix footer'>" + "<time datetime='" + val.Date + "'>" + val.Date + "</time>"
                    + "<span class='read-more'>READ MORE</span>" + "</p></div></a></article> ";
                $("#newsWrapper").append(html).slideDown(2000);
                remainderItemNum = val.Total - total;
            });

            var visibleItem = $("#newsWrapper > .each-item:visible").size();

            if (visibleItem > 3) {
                $("#loadLess").removeClass("hidden");
            }

            if (remainderItemNum >= 0 && remainderItemNum < 4) {
                $("#loadMore").addClass("hidden");
            }
        }
